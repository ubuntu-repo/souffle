52  /**
53  * Read and return the next tuple.
54  *
55  * Returns nullptr if no tuple was readable.
56  * @return
57  */
58  std::unique_ptr<RamDomain[]> readNextTuple() override {
59  if (file.eof()) {
60  return nullptr;
61  }
62  std::string line;
63  std::unique_ptr<RamDomain[]> tuple = std::make_unique<RamDomain[]>(symbolMask.size());
64 
65  if (!getline(file, line)) {
66  return nullptr;
67  }
68  // Handle Windows line endings on non-Windows systems
69  if (line.back() == '\r') {
70  line = line.substr(0, line.length() - 1);
71  }
72  ++lineNumber;
73 
74  size_t start = 0, end = 0, columnsFilled = 0;
75  for (uint32_t column = 0; columnsFilled < arity; column++) {
76  end = line.find(delimiter, start);
77  if (end == std::string::npos) {
78  end = line.length();
79  }
80  std::string element;
81  if (start <= end) {
82  element = line.substr(start, end - start);
83  } else {
84  std::stringstream errorMessage;
85  errorMessage << "Values missing in line " << lineNumber << "; ";
86  throw std::invalid_argument(errorMessage.str());
87  }
88  start = end + delimiter.size();
89  if (inputMap.count(column) == 0) {
90  continue;
91  }
92  ++columnsFilled;
93  if (symbolMask.at(column)) {
94  tuple[inputMap[column]] = symbolTable.unsafeLookup(element);
95  } else {
96  try {
97 #if RAM_DOMAIN_SIZE == 64
98  tuple[inputMap[column]] = std::stoll(element);
99 #else
100  tuple[inputMap[column]] = std::stoi(element);
101 #endif
102  } catch (...) {
103  std::stringstream errorMessage;
104  errorMessage << "Error converting number <" + element + "> in column " << column + 1
105  << " in line " << lineNumber << "; ";
106  throw std::invalid_argument(errorMessage.str());
107  }
108  }
109  }
110 
111  return tuple;
112  }

114  std::string getDelimiter(const IODirectives& ioDirectives) const {
115  if (ioDirectives.has("delimiter")) {
116  return ioDirectives.get("delimiter");
117  }
118  return "\t";
119  }

121  std::map<int, int> getInputColumnMap(const IODirectives& ioDirectives, const unsigned arity) const {
122  std::string columnString = "";
123  if (ioDirectives.has("columns")) {
124  columnString = ioDirectives.get("columns");
125  }
126  std::map<int, int> inputMap;
127 
128  if (!columnString.empty()) {
129  std::istringstream iss(columnString);
130  std::string mapping;
131  int index = 0;
132  while (std::getline(iss, mapping, ':')) {
133  inputMap[stoi(mapping)] = index++;
134  }
135  if (inputMap.size() < arity) {
136  throw std::invalid_argument("Invalid column set was given: <" + columnString + ">");
137  }
138  } else {
139  while (inputMap.size() < arity) {
140  int size = inputMap.size();
141  inputMap[size] = size;
142  }
143  }
144  return inputMap;
145  }
