83  : identifier(ident), element(elem), relation(std::move(rel)) {}
84 
85  Location(const Location& l) : identifier(l.identifier), element(l.element) {
86  if (l.relation != nullptr) {
87  relation = std::unique_ptr<RamRelationReference>(l.relation->clone());

89  }
90 
91  Location& operator=(Location other) {
92  identifier = other.identifier;

96  }

99  return identifier == loc.identifier && element == loc.element;
100  }
101 
102  bool operator!=(const Location& loc) const {
103  return !(*this == loc);
104  }
105 
106  bool operator<(const Location& loc) const {
107  return identifier < loc.identifier || (identifier == loc.identifier && element < loc.element);
