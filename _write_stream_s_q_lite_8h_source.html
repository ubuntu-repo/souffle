97  uint64_t getSymbolTableIDFromDB(int index) {
98  if (sqlite3_bind_text(symbolSelectStatement, 1, symbolTable.unsafeResolve(index).c_str(), -1,
99  SQLITE_TRANSIENT) != SQLITE_OK) {
100  throwError("SQLite error in sqlite3_bind_text: ");
101  }
102  if (sqlite3_step(symbolSelectStatement) != SQLITE_ROW) {
103  throwError("SQLite error in sqlite3_step: ");
104  }
105  uint64_t rowid = sqlite3_column_int64(symbolSelectStatement, 0);
106  sqlite3_clear_bindings(symbolSelectStatement);
107  sqlite3_reset(symbolSelectStatement);
108  return rowid;
109  }
110  uint64_t getSymbolTableID(int index) {
111  if (dbSymbolTable.count(index) != 0) {
112  return dbSymbolTable[index];
113  }
114 
115  if (sqlite3_bind_text(symbolInsertStatement, 1, symbolTable.unsafeResolve(index).c_str(), -1,
116  SQLITE_TRANSIENT) != SQLITE_OK) {
117  throwError("SQLite error in sqlite3_bind_text: ");
118  }
119  // Either the insert succeeds and we have a new row id or it already exists and a select is needed.
120  uint64_t rowid;
121  if (sqlite3_step(symbolInsertStatement) != SQLITE_DONE) {
122  // The symbol already exists so select it.
123  rowid = getSymbolTableIDFromDB(index);
124  } else {
125  rowid = sqlite3_last_insert_rowid(db);
126  }
127  sqlite3_clear_bindings(symbolInsertStatement);
128  sqlite3_reset(symbolInsertStatement);
129 
130  dbSymbolTable[index] = rowid;
131  return rowid;

203  void createRelationView() {
204  // Create view with symbol strings resolved
205  std::stringstream createViewText;
206  createViewText << "CREATE VIEW IF NOT EXISTS '" << relationName << "' AS ";
207  std::stringstream projectionClause;
208  std::stringstream fromClause;
209  fromClause << "'_" << relationName << "'";
210  std::stringstream whereClause;
211  bool firstWhere = true;
212  for (unsigned int i = 0; i < arity; i++) {
213  std::string columnName = std::to_string(i);
214  if (i != 0) {
215  projectionClause << ",";
216  }
217  if (!symbolMask.at(i)) {
218  projectionClause << "'_" << relationName << "'.'" << columnName << "'";
219  } else {
220  projectionClause << "'_symtab_" << columnName << "'.symbol AS '" << columnName << "'";
221  fromClause << ",'" << symbolTableName << "' AS '_symtab_" << columnName << "'";
222  if (!firstWhere) {
223  whereClause << " AND ";
224  } else {
225  firstWhere = false;
226  }
227  whereClause << "'_" << relationName << "'.'" << columnName << "' = "
228  << "'_symtab_" << columnName << "'.id";
229  }
230  }
231  createViewText << "SELECT " << projectionClause.str() << " FROM " << fromClause.str();
232  if (!firstWhere) {
233  createViewText << " WHERE " << whereClause.str();
234  }
235  createViewText << ";";
236  executeSQL(createViewText.str(), db);
237  }
