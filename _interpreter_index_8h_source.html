107  bool isCompatible(const InterpreterIndexOrder& other) const {

137  /* lexicographical comparison operation on two tuple pointers */
138  struct comparator {
139  const InterpreterIndexOrder& order;
140 
141  /* constructor to initialize state */
142  comparator(const InterpreterIndexOrder& order) : order(order) {}
143 
144  /* comparison function */
145  int operator()(const RamDomain* x, const RamDomain* y) const {
146  for (size_t i = 0; i < order.size(); i++) {
147  if (x[order[i]] < y[order[i]]) {
148  return -1;
149  }
150  if (x[order[i]] > y[order[i]]) {
151  return 1;
152  }
153  }
154  return 0;
155  }
156 
157  /* less comparison */
158  bool less(const RamDomain* x, const RamDomain* y) const {
159  return operator()(x, y) < 0;
160  }
161 
162  /* equal comparison */
163  bool equal(const RamDomain* x, const RamDomain* y) const {
164  for (size_t i = 0; i < order.size(); i++) {
165  if (x[order[i]] != y[order[i]]) {
166  return false;
167  }
168  }
169  return true;
170  }
171  };
