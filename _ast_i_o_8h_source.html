128  const std::string& inputString, const std::string& needle, const std::string& replacement) const {
129  std::string result = inputString;
130  size_t pos = 0;
131  while ((pos = result.find(needle, pos)) != std::string::npos) {
132  result = result.replace(pos, needle.length(), replacement);
