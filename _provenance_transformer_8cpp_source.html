62 std::unique_ptr<AstRelation> makeInfoRelation(
63  AstClause& originalClause, AstTranslationUnit& translationUnit) {
64  AstRelationIdentifier name =
65  makeRelationName(originalClause.getHead()->getName(), "@info", originalClause.getClauseNum());
66 
67  // initialise info relation
68  auto infoRelation = new AstRelation();
69  infoRelation->setName(name);
70 
71  // create new clause containing a single fact
72  auto infoClause = new AstClause();
73  auto infoClauseHead = new AstAtom();
74  infoClauseHead->setName(name);
75 
76  infoRelation->addAttribute(std::make_unique<AstAttribute>("clause_num", AstTypeIdentifier("number")));
77  infoClauseHead->addArgument(std::make_unique<AstNumberConstant>(originalClause.getClauseNum()));
78 
79  // add head relation as meta info
80  std::vector<std::string> headVariables;
81  visitDepthFirst(*(originalClause.getHead()), [&](const AstVariable& var) {
82  std::stringstream varName;
83  var.print(varName);
84  headVariables.push_back(varName.str());
85  });
86 
87  std::stringstream headVariableString;
88  headVariableString << join(headVariables, ",");
89 
90  infoRelation->addAttribute(
91  std::make_unique<AstAttribute>(std::string("head_vars"), AstTypeIdentifier("symbol")));
92  infoClauseHead->addArgument(
93  std::make_unique<AstStringConstant>(translationUnit.getSymbolTable(), headVariableString.str()));
94 
95  // visit all body literals and add to info clause head
96  for (size_t i = 0; i < originalClause.getBodyLiterals().size(); i++) {
97  auto lit = originalClause.getBodyLiterals()[i];
98  const AstAtom* atom = lit->getAtom();
99 
100  // add an attribute for atoms and binary constraints
101  if (atom != nullptr || dynamic_cast<AstBinaryConstraint*>(lit) != nullptr) {
102  infoRelation->addAttribute(std::make_unique<AstAttribute>(
103  std::string("rel_") + std::to_string(i), AstTypeIdentifier("symbol")));
104  }
105 
106  if (atom != nullptr) {
107  std::string relName = identifierToString(atom->getName());
108 
109  if (dynamic_cast<AstAtom*>(lit) != nullptr) {
110  std::string atomDescription = relName;
111 
112  // for each variable in the literal, add an extra field to the instrumented relation name
113  visitDepthFirst(*atom, [&](const AstVariable& var) {
114  std::stringstream varName;
115  var.print(varName);
116  atomDescription.append("," + varName.str());
117  });
118 
119  /*
120  for (auto& arg : atom->getArguments()) {
121  std::stringstream argDescription;
122  arg->print(argDescription);
123  atomDescription.append("," + argDescription.str());
124  }
125  */
126 
127  infoClauseHead->addArgument(std::make_unique<AstStringConstant>(
128  translationUnit.getSymbolTable(), atomDescription));
129  } else if (dynamic_cast<AstNegation*>(lit) != nullptr) {
130  infoClauseHead->addArgument(std::make_unique<AstStringConstant>(
131  translationUnit.getSymbolTable(), ("!" + relName)));
132  }
133  } else if (auto con = dynamic_cast<AstBinaryConstraint*>(lit)) {
134  infoClauseHead->addArgument(std::make_unique<AstStringConstant>(
135  translationUnit.getSymbolTable(), (toBinaryConstraintSymbol(con->getOperator()))));
136  }
137  }
138 
139  // generate and add clause representation
140  std::stringstream ss;
141  originalClause.print(ss);
142 
143  infoRelation->addAttribute(std::make_unique<AstAttribute>("clause_repr", AstTypeIdentifier("symbol")));
144  infoClauseHead->addArgument(
145  std::make_unique<AstStringConstant>(translationUnit.getSymbolTable(), ss.str()));
146 
147  // set clause head and add clause to info relation
148  infoClause->setHead(std::unique_ptr<AstAtom>(infoClauseHead));
149  infoRelation->addClause(std::unique_ptr<AstClause>(infoClause));
150 
151  return std::unique_ptr<AstRelation>(infoRelation);
152 }

154 /** Transform eqrel relations to explicitly define equivalence relations */
155 void transformEqrelRelation(AstRelation& rel) {
156  assert(rel.getRepresentation() == RelationRepresentation::EQREL &&
157  "attempting to transform non-eqrel relation");
158  assert(rel.getArity() == 2 && "eqrel relation not binary");
159 
160  rel.setQualifier(rel.getQualifier() - EQREL_RELATION);
161 
162  // transitivity
163  // transitive clause: A(x, z) :- A(x, y), A(y, z).
164  auto transitiveClause = new AstClause();
165  auto transitiveClauseHead = new AstAtom(rel.getName());
166  transitiveClauseHead->addArgument(std::make_unique<AstVariable>("x"));
167  transitiveClauseHead->addArgument(std::make_unique<AstVariable>("z"));
168 
169  auto transitiveClauseBody = new AstAtom(rel.getName());
170  transitiveClauseBody->addArgument(std::make_unique<AstVariable>("x"));
171  transitiveClauseBody->addArgument(std::make_unique<AstVariable>("y"));
172 
173  auto transitiveClauseBody2 = new AstAtom(rel.getName());
174  transitiveClauseBody2->addArgument(std::make_unique<AstVariable>("y"));
175  transitiveClauseBody2->addArgument(std::make_unique<AstVariable>("z"));
176 
177  transitiveClause->setHead(std::unique_ptr<AstAtom>(transitiveClauseHead));
178  transitiveClause->addToBody(std::unique_ptr<AstLiteral>(transitiveClauseBody));
179  transitiveClause->addToBody(std::unique_ptr<AstLiteral>(transitiveClauseBody2));
180  rel.addClause(std::unique_ptr<AstClause>(transitiveClause));
181 
182  // symmetric
183  // symmetric clause: A(x, y) :- A(y, x).
184  auto symClause = new AstClause();
185  auto symClauseHead = new AstAtom(rel.getName());
186  symClauseHead->addArgument(std::make_unique<AstVariable>("x"));
187  symClauseHead->addArgument(std::make_unique<AstVariable>("y"));
188 
189  auto symClauseBody = new AstAtom(rel.getName());
190  symClauseBody->addArgument(std::make_unique<AstVariable>("y"));
191  symClauseBody->addArgument(std::make_unique<AstVariable>("x"));
192 
193  symClause->setHead(std::unique_ptr<AstAtom>(symClauseHead));
194  symClause->addToBody(std::unique_ptr<AstLiteral>(symClauseBody));
195  rel.addClause(std::unique_ptr<AstClause>(symClause));
196 
197  // reflexivity
198  // reflexive clause: A(x, x) :- A(x, _).
199  auto reflexiveClause = new AstClause();
200  auto reflexiveClauseHead = new AstAtom(rel.getName());
201  reflexiveClauseHead->addArgument(std::make_unique<AstVariable>("x"));
202  reflexiveClauseHead->addArgument(std::make_unique<AstVariable>("x"));
203 
204  auto reflexiveClauseBody = new AstAtom(rel.getName());
205  reflexiveClauseBody->addArgument(std::make_unique<AstVariable>("x"));
206  reflexiveClauseBody->addArgument(std::make_unique<AstUnnamedVariable>());
207 
208  reflexiveClause->setHead(std::unique_ptr<AstAtom>(reflexiveClauseHead));
209  reflexiveClause->addToBody(std::unique_ptr<AstLiteral>(reflexiveClauseBody));
210  rel.addClause(std::unique_ptr<AstClause>(reflexiveClause));
211 }

213 bool ProvenanceTransformer::transform(AstTranslationUnit& translationUnit) {
214  auto program = translationUnit.getProgram();
215 
216  // get next level number
217  auto getNextLevelNumber = [&](std::vector<AstArgument*> levels) {
218  if (levels.empty()) {
219  return static_cast<AstArgument*>(new AstNumberConstant(0));
220  }
221 
222  if (levels.size() == 1) {
223  return static_cast<AstArgument*>(new AstIntrinsicFunctor(FunctorOp::ADD,
224  std::unique_ptr<AstArgument>(levels[0]), std::make_unique<AstNumberConstant>(1)));
225  }
226 
227  auto currentMax = new AstIntrinsicFunctor(FunctorOp::MAX, std::unique_ptr<AstArgument>(levels[0]),
228  std::unique_ptr<AstArgument>(levels[1]));
229 
230  for (size_t i = 2; i < levels.size(); i++) {
231  currentMax = new AstIntrinsicFunctor(FunctorOp::MAX, std::unique_ptr<AstArgument>(currentMax),
232  std::unique_ptr<AstArgument>(levels[i]));
233  }
234 
235  return static_cast<AstArgument*>(new AstIntrinsicFunctor(FunctorOp::ADD,
236  std::unique_ptr<AstArgument>(currentMax), std::make_unique<AstNumberConstant>(1)));
237  };
238 
239  for (auto relation : program->getRelations()) {
240  if (relation->getRepresentation() == RelationRepresentation::EQREL) {
241  transformEqrelRelation(*relation);
242  }
243 
244  // generate info relations for each clause
245  // do this before all other transformations so that we record
246  // the original rule without any instrumentation
247  size_t clauseNum = 1;
248  for (auto clause : relation->getClauses()) {
249  if (!clause->isFact()) {
250  clause->setClauseNum(clauseNum);
251 
252  // add info relation
253  program->addRelation(makeInfoRelation(*clause, translationUnit));
254 
255  clauseNum++;
256  }
257  }
258 
259  relation->addAttribute(
260  std::make_unique<AstAttribute>(std::string("@rule_number"), AstTypeIdentifier("number")));
261  relation->addAttribute(
262  std::make_unique<AstAttribute>(std::string("@level_number"), AstTypeIdentifier("number")));
263 
264  for (auto clause : relation->getClauses()) {
265  // mapper to add two provenance columns to atoms
266  struct M : public AstNodeMapper {
267  using AstNodeMapper::operator();
268 
269  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
270  // add provenance columns
271  if (auto atom = dynamic_cast<AstAtom*>(node.get())) {
272  atom->addArgument(std::make_unique<AstUnnamedVariable>());
273  atom->addArgument(std::make_unique<AstUnnamedVariable>());
274  } else if (auto neg = dynamic_cast<AstNegation*>(node.get())) {
275  auto atom = neg->getAtom();
276  atom->addArgument(std::make_unique<AstUnnamedVariable>());
277  atom->addArgument(std::make_unique<AstUnnamedVariable>());
278  }
279 
280  // otherwise - apply mapper recursively
281  node->apply(*this);
282  return node;
283  }
284  };
285 
286  // add unnamed vars to each atom nested in arguments of head
287  clause->getHead()->apply(M());
288 
289  // if fact, level number is 0
290  if (clause->isFact()) {
291  clause->getHead()->addArgument(std::make_unique<AstNumberConstant>(0));
292  clause->getHead()->addArgument(std::make_unique<AstNumberConstant>(0));
293  } else {
294  std::vector<AstArgument*> bodyLevels;
295 
296  for (size_t i = 0; i < clause->getBodyLiterals().size(); i++) {
297  auto lit = clause->getBodyLiterals()[i];
298 
299  // add unnamed vars to each atom nested in arguments of lit
300  lit->apply(M());
301 
302  // add two provenance columns to lit; first is rule num, second is level num
303  if (auto atom = dynamic_cast<AstAtom*>(lit)) {
304  atom->addArgument(std::make_unique<AstUnnamedVariable>());
305  atom->addArgument(std::make_unique<AstVariable>("@level_num_" + std::to_string(i)));
306  bodyLevels.push_back(new AstVariable("@level_num_" + std::to_string(i)));
307  }
308  }
309 
310  // add two provenance columns to head lit
311  clause->getHead()->addArgument(std::make_unique<AstNumberConstant>(clause->getClauseNum()));
312  clause->getHead()->addArgument(std::unique_ptr<AstArgument>(getNextLevelNumber(bodyLevels)));
313  }
314  }
315  }
316  return true;
317 }
