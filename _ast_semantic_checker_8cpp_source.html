71 void AstSemanticChecker::checkProgram(ErrorReport& report, const AstProgram& program,
72  const TypeEnvironment& typeEnv, const TypeAnalysis& typeAnalysis,
73  const PrecedenceGraph& precedenceGraph, const RecursiveClauses& recursiveClauses,
74  const IOType& ioTypes) {
75  // suppress warnings for given relations
76  if (Global::config().has("suppress-warnings")) {
77  std::vector<std::string> suppressedRelations =
78  splitString(Global::config().get("suppress-warnings"), ',');
79 
80  if (std::find(suppressedRelations.begin(), suppressedRelations.end(), "*") !=
81  suppressedRelations.end()) {
82  // mute all relations
83  for (AstRelation* rel : program.getRelations()) {
84  rel->setQualifier(rel->getQualifier() | SUPPRESSED_RELATION);
85  }
86  } else {
87  // mute only the given relations (if they exist)
88  for (auto& relname : suppressedRelations) {
89  const std::vector<std::string> comps = splitString(relname, '.');
90  if (!comps.empty()) {
91  // generate the relation identifier
92  AstRelationIdentifier relid(comps[0]);
93  for (size_t i = 1; i < comps.size(); i++) {
94  relid.append(comps[i]);
95  }
96 
97  // update suppressed qualifier if the relation is found
98  if (AstRelation* rel = program.getRelation(relid)) {
99  rel->setQualifier(rel->getQualifier() | SUPPRESSED_RELATION);
100  }
101  }
102  }
103  }
104  }
105 
106  // -- conduct checks --
107  // TODO: re-write to use visitors
108  checkTypes(report, program);
109  checkRules(report, typeEnv, program, recursiveClauses, ioTypes);
110  checkNamespaces(report, program);
111  checkIODirectives(report, program);
112  checkWitnessProblem(report, program);
113  checkInlining(report, program, precedenceGraph, ioTypes);
114 
115  // get the list of components to be checked
116  std::vector<const AstNode*> nodes;
117  for (const auto& rel : program.getRelations()) {
118  for (const auto& cls : rel->getClauses()) {
119  nodes.push_back(cls);
120  }
121  }
122 
123  // -- check grounded variables and records --
124  visitDepthFirst(nodes, [&](const AstClause& clause) {
125  // only interested in rules
126  if (clause.isFact()) {
127  return;
128  }
129 
130  // compute all grounded terms
131  auto isGrounded = getGroundedTerms(clause);
132 
133  // all terms in head need to be grounded
134  std::set<std::string> reportedVars;
135  for (const AstVariable* cur : getVariables(clause)) {
136  if (!isGrounded[cur] && reportedVars.insert(cur->getName()).second) {
137  report.addError("Ungrounded variable " + cur->getName(), cur->getSrcLoc());
138  }
139  }
140 
141  // all records need to be grounded
142  for (const AstRecordInit* cur : getRecords(clause)) {
143  if (!isGrounded[cur]) {
144  report.addError("Ungrounded record", cur->getSrcLoc());
145  }
146  }
147  });
148 
149  // -- type checks --
150 
151  // - variables -
152  visitDepthFirst(nodes, [&](const AstVariable& var) {
153  if (typeAnalysis.getTypes(&var).empty()) {
154  report.addError("Unable to deduce type for variable " + var.getName(), var.getSrcLoc());
155  }
156  });
157 
158  // - constants -
159 
160  // all string constants are used as symbols
161  visitDepthFirst(nodes, [&](const AstStringConstant& cnst) {
162  TypeSet types = typeAnalysis.getTypes(&cnst);
163  if (!isSymbolType(types)) {
164  report.addError("Symbol constant (type mismatch)", cnst.getSrcLoc());
165  }
166  });
167 
168  // all number constants are used as numbers
169  visitDepthFirst(nodes, [&](const AstNumberConstant& cnst) {
170  TypeSet types = typeAnalysis.getTypes(&cnst);
171  if (!isNumberType(types)) {
172  report.addError("Number constant (type mismatch)", cnst.getSrcLoc());
173  }
174  AstDomain idx = cnst.getIndex();
175  if (idx > MAX_AST_DOMAIN || idx < MIN_AST_DOMAIN) {
176  report.addError("Number constant not in range [" + std::to_string(MIN_AST_DOMAIN) + ", " +
177  std::to_string(MAX_AST_DOMAIN) + "]",
178  cnst.getSrcLoc());
179  }
180  });
181 
182  // all null constants are used as records
183  visitDepthFirst(nodes, [&](const AstNullConstant& cnst) {
184  // TODO (#467) remove the next line to enable subprogram compilation for record types
185  Global::config().unset("engine");
186  TypeSet types = typeAnalysis.getTypes(&cnst);
187  if (!isRecordType(types)) {
188  report.addError("Null constant used as a non-record", cnst.getSrcLoc());
189  }
190  });
191 
192  // record initializations have the same size as their types
193  visitDepthFirst(nodes, [&](const AstRecordInit& cnst) {
194  // TODO (#467) remove the next line to enable subprogram compilation for record types
195  Global::config().unset("engine");
196  TypeSet types = typeAnalysis.getTypes(&cnst);
197  if (isRecordType(types)) {
198  for (const Type& type : types) {
199  if (cnst.getArguments().size() !=
200  dynamic_cast<const RecordType*>(&type)->getFields().size()) {
201  report.addError("Wrong number of arguments given to record", cnst.getSrcLoc());
202  }
203  }
204  }
205  });
206 
207  // type casts name a valid type
208  visitDepthFirst(nodes, [&](const AstTypeCast& cast) {
209  if (!typeEnv.isType(cast.getType())) {
210  report.addError("Type cast is to undeclared type " + toString(cast.getType()), cast.getSrcLoc());
211  }
212  });
213 
214  // - intrinsic functors -
215  visitDepthFirst(nodes, [&](const AstIntrinsicFunctor& fun) {
216  // check type of result
217  if (fun.isNumerical() && !isNumberType(typeAnalysis.getTypes(&fun))) {
218  report.addError("Non-numeric use for numeric functor", fun.getSrcLoc());
219  }
220 
221  if (fun.isSymbolic() && !isSymbolType(typeAnalysis.getTypes(&fun))) {
222  report.addError("Non-symbolic use for symbolic functor", fun.getSrcLoc());
223  }
224 
225  // check types of arguments
226  if (fun.getFunction() == FunctorOp::ORD) {
227  return;
228  }
229 
230  for (size_t i = 0; i < fun.getArity(); i++) {
231  auto arg = fun.getArg(i);
232  if (fun.acceptsNumbers(i) && !isNumberType(typeAnalysis.getTypes(arg))) {
233  report.addError("Non-numeric argument for functor", arg->getSrcLoc());
234  }
235  if (fun.acceptsSymbols(i) && !isSymbolType(typeAnalysis.getTypes(arg))) {
236  report.addError("Non-symbolic argument for functor", arg->getSrcLoc());
237  }
238  }
239  });
240 
241  // - user-defined functors -
242  visitDepthFirst(nodes, [&](const AstUserDefinedFunctor& fun) {
243  const AstFunctorDeclaration* funDecl = program.getFunctorDeclaration(fun.getName());
244  if (funDecl == nullptr) {
245  report.addError("User-defined functor hasn't been declared", fun.getSrcLoc());
246  } else {
247  if (funDecl->getArgCount() != fun.getArgCount()) {
248  report.addError("Mismatching number of arguments of functor", fun.getSrcLoc());
249  }
250  // check return values of user-defined functor
251  if (funDecl->isNumerical() && !isNumberType(typeAnalysis.getTypes(&fun))) {
252  report.addError("Non-numeric use for numeric functor", fun.getSrcLoc());
253  }
254  if (funDecl->isSymbolic() && !isSymbolType(typeAnalysis.getTypes(&fun))) {
255  report.addError("Non-symbolic use for symbolic functor", fun.getSrcLoc());
256  }
257  for (size_t i = 0; i < fun.getArgCount(); i++) {
258  const AstArgument* arg = fun.getArg(i);
259  if (i < funDecl->getArgCount()) {
260  if (funDecl->acceptsNumbers(i) && !isNumberType(typeAnalysis.getTypes(arg))) {
261  report.addError("Non-numeric argument for functor", arg->getSrcLoc());
262  }
263  if (funDecl->acceptsSymbols(i) && !isSymbolType(typeAnalysis.getTypes(arg))) {
264  report.addError("Non-symbolic argument for functor", arg->getSrcLoc());
265  }
266  }
267  }
268  }
269  });
270 
271  // - binary relation -
272  visitDepthFirst(nodes, [&](const AstBinaryConstraint& constraint) {
273  // only interested in non-equal constraints
274  auto op = constraint.getOperator();
275  if (op == BinaryConstraintOp::EQ || op == BinaryConstraintOp::NE) {
276  return;
277  }
278 
279  // get left and right side
280  auto lhs = constraint.getLHS();
281  auto rhs = constraint.getRHS();
282 
283  if (constraint.isNumerical()) {
284  // check numeric type
285  if (!isNumberType(typeAnalysis.getTypes(lhs))) {
286  report.addError("Non-numerical operand for comparison", lhs->getSrcLoc());
287  }
288  if (!isNumberType(typeAnalysis.getTypes(rhs))) {
289  report.addError("Non-numerical operand for comparison", rhs->getSrcLoc());
290  }
291  } else if (constraint.isSymbolic()) {
292  // check symbolic type
293  if (!isSymbolType(typeAnalysis.getTypes(lhs))) {
294  report.addError("Non-string operand for operation", lhs->getSrcLoc());
295  }
296  if (!isSymbolType(typeAnalysis.getTypes(rhs))) {
297  report.addError("Non-string operand for operation", rhs->getSrcLoc());
298  }
299  }
300  });
301 
302  // - stratification --
303 
304  // check for cyclic dependencies
305  const Graph<const AstRelation*, AstNameComparison>& depGraph = precedenceGraph.graph();
306  for (const AstRelation* cur : depGraph.vertices()) {
307  if (depGraph.reaches(cur, cur)) {
308  AstRelationSet clique = depGraph.clique(cur);
309  for (const AstRelation* cyclicRelation : clique) {
310  // Negations and aggregations need to be stratified
311  const AstLiteral* foundLiteral = nullptr;
312  bool hasNegation = hasClauseWithNegatedRelation(cyclicRelation, cur, &program, foundLiteral);
313  if (hasNegation ||
314  hasClauseWithAggregatedRelation(cyclicRelation, cur, &program, foundLiteral)) {
315  std::string relationsListStr = toString(join(clique, ",",
316  [](std::ostream& out, const AstRelation* r) { out << r->getName(); }));
317  std::vector<DiagnosticMessage> messages;
318  messages.push_back(
319  DiagnosticMessage("Relation " + toString(cur->getName()), cur->getSrcLoc()));
320  std::string negOrAgg = hasNegation ? "negation" : "aggregation";
321  messages.push_back(
322  DiagnosticMessage("has cyclic " + negOrAgg, foundLiteral->getSrcLoc()));
323  report.addDiagnostic(Diagnostic(Diagnostic::ERROR,
324  DiagnosticMessage("Unable to stratify relation(s) {" + relationsListStr + "}"),
325  messages));
326  break;
327  }
328  }
329  }
330  }

352 static bool hasUnnamedVariable(const AstArgument* arg) {
353  if (dynamic_cast<const AstUnnamedVariable*>(arg)) {
354  return true;
355  }
356  if (dynamic_cast<const AstVariable*>(arg)) {
357  return false;
358  }
359  if (dynamic_cast<const AstConstant*>(arg)) {
360  return false;
361  }
362  if (dynamic_cast<const AstCounter*>(arg)) {
363  return false;
364  }
365  if (const auto* cast = dynamic_cast<const AstTypeCast*>(arg)) {
366  return hasUnnamedVariable(cast->getValue());
367  }
368  if (const auto* inf = dynamic_cast<const AstIntrinsicFunctor*>(arg)) {
369  return any_of(inf->getArguments(), (bool (*)(const AstArgument*))hasUnnamedVariable);
370  }
371  if (const auto* udf = dynamic_cast<const AstUserDefinedFunctor*>(arg)) {
372  return any_of(udf->getArguments(), (bool (*)(const AstArgument*))hasUnnamedVariable);
373  }
374  if (const auto* ri = dynamic_cast<const AstRecordInit*>(arg)) {
375  return any_of(ri->getArguments(), (bool (*)(const AstArgument*))hasUnnamedVariable);
376  }
377  if (dynamic_cast<const AstAggregator*>(arg)) {
378  return false;
379  }
380  std::cout << "Unsupported Argument type: " << typeid(*arg).name() << "\n";
381  assert(false && "Unsupported Argument Type!");
382  return false;

387  return any_of(at->getArguments(), (bool (*)(const AstArgument*))hasUnnamedVariable);
388  }

398  }

401  assert(false && "Unsupported Argument Type!");

523 void AstSemanticChecker::checkClause(ErrorReport& report, const AstProgram& program, const AstClause& clause,
524  const RecursiveClauses& recursiveClauses) {
525  // check head atom
526  checkAtom(report, program, *clause.getHead());
527 
528  // check for absence of underscores in head
529  if (hasUnnamedVariable(clause.getHead())) {
530  report.addError("Underscore in head of rule", clause.getHead()->getSrcLoc());
531  }
532 
533  // check body literals
534  for (AstLiteral* lit : clause.getAtoms()) {
535  checkLiteral(report, program, *lit);
536  }
537  for (AstNegation* neg : clause.getNegations()) {
538  checkLiteral(report, program, *neg);
539  }
540  for (AstLiteral* lit : clause.getConstraints()) {
541  checkLiteral(report, program, *lit);
542  }
543 
544  // check facts
545  if (clause.isFact()) {
546  checkFact(report, program, clause);
547  }
548 
549  // check for use-once variables
550  std::map<std::string, int> var_count;
551  std::map<std::string, const AstVariable*> var_pos;
552  visitDepthFirst(clause, [&](const AstVariable& var) {
553  var_count[var.getName()]++;
554  var_pos[var.getName()] = &var;
555  });
556 
557  // check for variables only occurring once
558  if (!clause.isGenerated()) {
559  for (const auto& cur : var_count) {
560  if (cur.second == 1 && cur.first[0] != '_') {
561  report.addWarning(
562  "Variable " + cur.first + " only occurs once", var_pos[cur.first]->getSrcLoc());
563  }
564  }
565  }
566 
567  // check execution plan
568  if (clause.getExecutionPlan()) {
569  auto numAtoms = clause.getAtoms().size();
570  for (const auto& cur : clause.getExecutionPlan()->getOrders()) {
571  if (cur.second->size() != numAtoms || !cur.second->isComplete()) {
572  report.addError("Invalid execution plan", cur.second->getSrcLoc());
573  }
574  }
575  }
576  // check auto-increment
577  if (recursiveClauses.recursive(&clause)) {
578  visitDepthFirst(clause, [&](const AstCounter& ctr) {
579  report.addError("Auto-increment functor in a recursive rule", ctr.getSrcLoc());
580  });
581  }
582 }

584 void AstSemanticChecker::checkRelationDeclaration(ErrorReport& report, const TypeEnvironment& typeEnv,
585  const AstProgram& program, const AstRelation& relation, const IOType& ioTypes) {
586  for (size_t i = 0; i < relation.getArity(); i++) {
587  AstAttribute* attr = relation.getAttribute(i);
588  AstTypeIdentifier typeName = attr->getTypeName();
589 
590  /* check whether type exists */
591  if (typeName != "number" && typeName != "symbol" && !program.getType(typeName)) {
592  report.addError("Undefined type in attribute " + attr->getAttributeName() + ":" +
593  toString(attr->getTypeName()),
594  attr->getSrcLoc());
595  }
596 
597  /* check whether name occurs more than once */
598  for (size_t j = 0; j < i; j++) {
599  if (attr->getAttributeName() == relation.getAttribute(j)->getAttributeName()) {
600  report.addError("Doubly defined attribute name " + attr->getAttributeName() + ":" +
601  toString(attr->getTypeName()),
602  attr->getSrcLoc());
603  }
604  }
605 
606  /* check whether type is a record type */
607  if (typeEnv.isType(typeName)) {
608  const Type& type = typeEnv.getType(typeName);
609  if (isRecordType(type)) {
610  // TODO (#467) remove the next line to enable subprogram compilation for record types

817 static const std::vector<SrcLocation> usesInvalidWitness(const std::vector<AstLiteral*>& literals,
818  const std::set<std::unique_ptr<AstArgument>>& groundedArguments) {
819  // Node-mapper that replaces aggregators with new (unique) variables
820  struct M : public AstNodeMapper {
821  // Variables introduced to replace aggregators
822  mutable std::set<std::string> aggregatorVariables;
823 
824  const std::set<std::string>& getAggregatorVariables() {
825  return aggregatorVariables;
826  }
827 
828  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
829  static int numReplaced = 0;
830  if (dynamic_cast<AstAggregator*>(node.get())) {
831  // Replace the aggregator with a variable
832  std::stringstream newVariableName;
833  newVariableName << "+aggr_var_" << numReplaced++;
834 
835  // Keep track of which variables are bound to aggregators
836  aggregatorVariables.insert(newVariableName.str());
837 
838  return std::make_unique<AstVariable>(newVariableName.str());
839  }
840  node->apply(*this);
841  return node;
842  }
843  };
844 
845  std::vector<SrcLocation> result;
846 
847  // Create two versions of the original clause
848 
849  // Clause 1 - will remain equivalent to the original clause in terms of variable groundedness
850  auto originalClause = std::make_unique<AstClause>();
851  originalClause->setHead(std::make_unique<AstAtom>("*"));
852 
853  // Clause 2 - will have aggregators replaced with intrinsically grounded variables
854  auto aggregatorlessClause = std::make_unique<AstClause>();
855  aggregatorlessClause->setHead(std::make_unique<AstAtom>("*"));
856 
857  // Construct both clauses in the same manner to match the original clause
858  // Must keep track of the subnode in Clause 1 that each subnode in Clause 2 matches to
859  std::map<const AstArgument*, const AstArgument*> identicalSubnodeMap;
860  for (const AstLiteral* lit : literals) {
861  auto firstClone = std::unique_ptr<AstLiteral>(lit->clone());
862  auto secondClone = std::unique_ptr<AstLiteral>(lit->clone());
863 
864  // Construct the mapping between equivalent literal subnodes
865  std::vector<const AstArgument*> firstCloneArguments;
866  visitDepthFirst(*firstClone, [&](const AstArgument& arg) { firstCloneArguments.push_back(&arg); });
867 
868  std::vector<const AstArgument*> secondCloneArguments;
869  visitDepthFirst(*secondClone, [&](const AstArgument& arg) { secondCloneArguments.push_back(&arg); });
870 
871  for (size_t i = 0; i < firstCloneArguments.size(); i++) {
872  identicalSubnodeMap[secondCloneArguments[i]] = firstCloneArguments[i];
873  }
874 
875  // Actually add the literal clones to each clause
876  originalClause->addToBody(std::move(firstClone));
877  aggregatorlessClause->addToBody(std::move(secondClone));
878  }
879 
880  // Replace the aggregators in Clause 2 with variables
881  M update;
882  aggregatorlessClause->apply(update);
883 
884  // Create a dummy atom to force certain arguments to be grounded in the aggregatorlessClause
885  auto groundingAtomAggregatorless = std::make_unique<AstAtom>("grounding_atom");
886  auto groundingAtomOriginal = std::make_unique<AstAtom>("grounding_atom");
887 
888  // Force the new aggregator variables to be grounded in the aggregatorless clause
889  const std::set<std::string>& aggregatorVariables = update.getAggregatorVariables();
890  for (const std::string& str : aggregatorVariables) {
891  groundingAtomAggregatorless->addArgument(std::make_unique<AstVariable>(str));
892  }
893 
894  // Force the given grounded arguments to be grounded in both clauses
895  for (const std::unique_ptr<AstArgument>& arg : groundedArguments) {
896  groundingAtomAggregatorless->addArgument(std::unique_ptr<AstArgument>(arg->clone()));
897  groundingAtomOriginal->addArgument(std::unique_ptr<AstArgument>(arg->clone()));
898  }
899 
900  aggregatorlessClause->addToBody(std::move(groundingAtomAggregatorless));
901  originalClause->addToBody(std::move(groundingAtomOriginal));
902 
903  // Compare the grounded analysis of both generated clauses
904  // All added arguments in Clause 2 were forced to be grounded, so if an ungrounded argument
905  // appears in Clause 2, it must also appear in Clause 1. Consequently, have two cases:
906  // - The argument is also ungrounded in Clause 1 - handled by another check
907  // - The argument is grounded in Clause 1 => the argument was grounded in the
908  // first clause somewhere along the line by an aggregator-body - not allowed!
909  std::set<std::unique_ptr<AstArgument>> newlyGroundedArguments;
910  std::map<const AstArgument*, bool> originalGrounded = getGroundedTerms(*originalClause);
911  std::map<const AstArgument*, bool> aggregatorlessGrounded = getGroundedTerms(*aggregatorlessClause);
912  for (auto pair : aggregatorlessGrounded) {
913  if (!pair.second && originalGrounded[identicalSubnodeMap[pair.first]]) {
914  result.push_back(pair.first->getSrcLoc());
915  }
916 
917  // Otherwise, it can now be considered grounded
918  newlyGroundedArguments.insert(std::unique_ptr<AstArgument>(pair.first->clone()));
919  }
920 
921  // All previously grounded are still grounded
922  for (const std::unique_ptr<AstArgument>& arg : groundedArguments) {
923  newlyGroundedArguments.insert(std::unique_ptr<AstArgument>(arg->clone()));
924  }
925 
926  // Everything on this level is fine, check subaggregators of each literal
927  for (const AstLiteral* lit : literals) {
928  visitDepthFirst(*lit, [&](const AstAggregator& aggr) {
929  // Check recursively if an invalid witness is used
930  std::vector<AstLiteral*> aggrBodyLiterals = aggr.getBodyLiterals();
931  std::vector<SrcLocation> subresult = usesInvalidWitness(aggrBodyLiterals, newlyGroundedArguments);
932  for (SrcLocation argloc : subresult) {
933  result.push_back(argloc);
934  }
935  });
936  }
937 
938  return result;
939 }

969 /**
970  * Find a cycle consisting entirely of inlined relations.
971  * If no cycle exists, then an empty vector is returned.
972  */
973 std::vector<AstRelationIdentifier> findInlineCycle(const PrecedenceGraph& precedenceGraph,
974  std::map<const AstRelation*, const AstRelation*>& origins, const AstRelation* current,
975  AstRelationSet& unvisited, AstRelationSet& visiting, AstRelationSet& visited) {
976  std::vector<AstRelationIdentifier> result;
977 
978  if (current == nullptr) {
979  // Not looking at any nodes at the moment, so choose any node from the unvisited list
980 
981  if (unvisited.empty()) {
982  // Nothing left to visit - so no cycles exist!
983  return result;
984  }
985 
986  // Choose any element from the unvisited set
987  current = *unvisited.begin();
988  origins[current] = nullptr;
989 
990  // Move it to "currently visiting"
991  unvisited.erase(current);
992  visiting.insert(current);
993 
994  // Check if we can find a cycle beginning from this node
995  std::vector<AstRelationIdentifier> subresult =
996  findInlineCycle(precedenceGraph, origins, current, unvisited, visiting, visited);
997 
998  if (subresult.empty()) {
999  // No cycle found, try again from another node
1000  return findInlineCycle(precedenceGraph, origins, nullptr, unvisited, visiting, visited);
1001  } else {
1002  // Cycle found! Return it
1003  return subresult;
1004  }
1005  }
1006 
1007  // Check neighbours
1008  const AstRelationSet& successors = precedenceGraph.graph().successors(current);
1009  for (const AstRelation* successor : successors) {
1010  // Only care about inlined neighbours in the graph
1011  if (successor->isInline()) {
1012  if (visited.find(successor) != visited.end()) {
1013  // The neighbour has already been visited, so move on
1014  continue;
1015  }
1016 
1017  if (visiting.find(successor) != visiting.end()) {
1018  // Found a cycle!!
1019  // Construct the cycle in reverse
1020  while (current != nullptr) {
1021  result.push_back(current->getName());
1022  current = origins[current];
1023  }
1024  return result;
1025  }
1026 
1027  // Node has not been visited yet
1028  origins[successor] = current;
1029 
1030  // Move from unvisited to visiting
1031  unvisited.erase(successor);
1032  visiting.insert(successor);
1033 
1034  // Visit recursively and check if a cycle is formed
1035  std::vector<AstRelationIdentifier> subgraphCycle =
1036  findInlineCycle(precedenceGraph, origins, successor, unvisited, visiting, visited);
1037 
1038  if (!subgraphCycle.empty()) {
1039  // Found a cycle!
1040  return subgraphCycle;
1041  }
1042  }
1043  }
1044 
1045  // Visited all neighbours with no cycle found, so done visiting this node.
1046  visiting.erase(current);
1047  visited.insert(current);
1048  return result;
1049 }

1051 void AstSemanticChecker::checkInlining(ErrorReport& report, const AstProgram& program,
1052  const PrecedenceGraph& precedenceGraph, const IOType& ioTypes) {
1053  // Find all inlined relations
1054  AstRelationSet inlinedRelations;
1055  for (const auto& relation : program.getRelations()) {
1056  if (relation->isInline()) {
1057  inlinedRelations.insert(relation);
1058  if (ioTypes.isIO(relation)) {
1059  report.addError("IO relation " + toString(relation->getName()) + " cannot be inlined",
1060  relation->getSrcLoc());
1061  }
1062  }
1063  }
1064 
1065  // Check 1:
1066  // Let G' be the subgraph of the precedence graph G containing only those nodes
1067  // which are marked with the inline directive.
1068  // If G' contains a cycle, then inlining cannot be performed.
1069 
1070  AstRelationSet unvisited; // nodes that have not been visited yet
1071  AstRelationSet visiting; // nodes that we are currently visiting
1072  AstRelationSet visited; // nodes that have been completely explored
1073 
1074  // All nodes are initially unvisited
1075  for (const AstRelation* rel : inlinedRelations) {
1076  unvisited.insert(rel);
1077  }
1078 
1079  // Remember the parent node of each visited node to construct the found cycle
1080  std::map<const AstRelation*, const AstRelation*> origins;
1081 
1082  std::vector<AstRelationIdentifier> result =
1083  findInlineCycle(precedenceGraph, origins, nullptr, unvisited, visiting, visited);
1084 
1085  // If the result contains anything, then a cycle was found
1086  if (!result.empty()) {
1087  AstRelation* cycleOrigin = program.getRelation(result[result.size() - 1]);
1088 
1089  // Construct the string representation of the cycle
1090  std::stringstream cycle;
1091  cycle << "{" << cycleOrigin->getName();
1092 
1093  // Print it backwards to preserve the initial cycle order
1094  for (int i = result.size() - 2; i >= 0; i--) {
1095  cycle << ", " << result[i];
1096  }
1097 
1098  cycle << "}";
1099 
1100  report.addError(
1101  "Cannot inline cyclically dependent relations " + cycle.str(), cycleOrigin->getSrcLoc());
1102  }
1103 
1104  // Check 2:
1105  // Cannot use the counter argument ('$') in inlined relations
1106 
1107  // Check if an inlined literal ever takes in a $
1108  visitDepthFirst(program, [&](const AstAtom& atom) {
1109  AstRelation* associatedRelation = program.getRelation(atom.getName());
1110  if (associatedRelation != nullptr && associatedRelation->isInline()) {
1111  visitDepthFirst(atom, [&](const AstArgument& arg) {
1112  if (dynamic_cast<const AstCounter*>(&arg)) {
1113  report.addError(
1114  "Cannot inline literal containing a counter argument '$'", arg.getSrcLoc());
1115  }
1116  });
1117  }
1118  });
1119 
1120  // Check if an inlined clause ever contains a $
1121  for (const AstRelation* rel : inlinedRelations) {
1122  for (AstClause* clause : rel->getClauses()) {
1123  visitDepthFirst(*clause, [&](const AstArgument& arg) {
1124  if (dynamic_cast<const AstCounter*>(&arg)) {
1125  report.addError(
1126  "Cannot inline clause containing a counter argument '$'", arg.getSrcLoc());
1127  }
1128  });
1129  }
1130  }
1131 
1132  // Check 3:
1133  // Suppose the relation b is marked with the inline directive, but appears negated
1134  // in a clause. Then, if b introduces a new variable in its body, we cannot inline
1135  // the relation b.
1136 
1137  // Find all relations with the inline declarative that introduce new variables in their bodies
1138  AstRelationSet nonNegatableRelations;
1139  for (const AstRelation* rel : inlinedRelations) {
1140  bool foundNonNegatable = false;
1141  for (const AstClause* clause : rel->getClauses()) {
1142  // Get the variables in the head
1143  std::set<std::string> headVariables;
1144  visitDepthFirst(
1145  *clause->getHead(), [&](const AstVariable& var) { headVariables.insert(var.getName()); });
1146 
1147  // Get the variables in the body
1148  std::set<std::string> bodyVariables;
1149  visitDepthFirst(clause->getBodyLiterals(),
1150  [&](const AstVariable& var) { bodyVariables.insert(var.getName()); });
1151 
1152  // Check if all body variables are in the head
1153  // Do this separately to the above so only one error is printed per variable
1154  for (const std::string& var : bodyVariables) {
1155  if (headVariables.find(var) == headVariables.end()) {
1156  nonNegatableRelations.insert(rel);
1157  foundNonNegatable = true;
1158  break;
1159  }
1160  }
1161 
1162  if (foundNonNegatable) {
1163  break;
1164  }
1165  }
1166  }
1167 
1168  // Check that these relations never appear negated
1169  visitDepthFirst(program, [&](const AstNegation& neg) {
1170  AstRelation* associatedRelation = program.getRelation(neg.getAtom()->getName());
1171  if (associatedRelation != nullptr &&
1172  nonNegatableRelations.find(associatedRelation) != nonNegatableRelations.end()) {
1173  report.addError(
1174  "Cannot inline negated relation which may introduce new variables", neg.getSrcLoc());
1175  }
1176  });
1177 
1178  // Check 4:
1179  // Don't support inlining atoms within aggregators at this point.
1180 
1181  // Reasoning: Suppose we have an aggregator like `max X: a(X)`, where `a` is inlined to `a1` and `a2`.
1182  // Then, `max X: a(X)` will become `max( max X: a1(X), max X: a2(X) )`. Suppose further that a(X) has
1183  // values X where it is true, while a2(X) does not. Then, the produced argument
1184  // `max( max X: a1(X), max X: a2(X) )` will not return anything (as one of its arguments fails), while
1185  // `max X: a(X)` will.
1186  // Can work around this with emptiness checks (e.g. `!a1(_), ... ; !a2(_), ... ; ...`)
1187 
1188  // This corner case prevents generalising aggregator inlining with the current set up.
1189 
1190  visitDepthFirst(program, [&](const AstAggregator& aggr) {
1191  visitDepthFirst(aggr, [&](const AstAtom& subatom) {
1192  const AstRelation* rel = program.getRelation(subatom.getName());
1193  if (rel != nullptr && rel->isInline()) {
1194  report.addError("Cannot inline relations that appear in aggregator", subatom.getSrcLoc());
1195  }
1196  });
1197  });
1198 
1199  // Check 5:
1200  // Suppose a relation `a` is inlined, appears negated in a clause, and contains a
1201  // (possibly nested) unnamed variable in its arguments. Then, the atom can't be
1202  // inlined, as unnamed variables are named during inlining (since they may appear
1203  // multiple times in an inlined-clause's body) => ungroundedness!
1204 
1205  // Exception: It's fine if the unnamed variable appears in a nested aggregator, as
1206  // the entire aggregator will automatically be grounded.
1207

1277 bool AstExecutionPlanChecker::transform(AstTranslationUnit& translationUnit) {
1278  auto* relationSchedule = translationUnit.getAnalysis<RelationSchedule>();
1279  auto* recursiveClauses = translationUnit.getAnalysis<RecursiveClauses>();
1280 
1281  for (const RelationScheduleStep& step : relationSchedule->schedule()) {
1282  const std::set<const AstRelation*>& scc = step.computed();
1283  for (const AstRelation* rel : scc) {
1284  for (const AstClause* clause : rel->getClauses()) {
1285  if (!recursiveClauses->recursive(clause)) {
1286  continue;
1287  }
1288  if (!clause->getExecutionPlan()) {
1289  continue;
1290  }
1291  int version = 0;
1292  for (const AstAtom* atom : clause->getAtoms()) {
1293  if (scc.count(getAtomRelation(atom, translationUnit.getProgram()))) {
1294  version++;
1295  }
1296  }
1297  if (version <= clause->getExecutionPlan()->getMaxVersion()) {
1298  for (const auto& cur : clause->getExecutionPlan()->getOrders()) {
1299  if (cur.first >= version) {
1300  translationUnit.getErrorReport().addDiagnostic(Diagnostic(Diagnostic::ERROR,
1301  DiagnosticMessage(
1302  "execution plan for version " + std::to_string(cur.first),
1303  cur.second->getSrcLoc()),
1304  {DiagnosticMessage("only versions 0.." + std::to_string(version - 1) +
1305  " permitted")}));
1306  }
1307  }
1308  }
1309  }
1310  }
1311  }
