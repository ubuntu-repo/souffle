67 /**
68  * Executes a binary file.
69  */
70 void executeBinary(const std::string& binaryFilename
71 #ifdef USE_MPI
72  ,
73  const int numberOfProcesses
74 #endif
75 ) {
76  assert(!binaryFilename.empty() && "binary filename cannot be blank");
77 
78  // check whether the executable exists
79  if (!isExecutable(binaryFilename)) {
80  throw std::invalid_argument("Generated executable <" + binaryFilename + "> could not be found");
81  }
82 
83  // run the executable
84  int exitCode;
85 #ifdef USE_MPI
86  if (Global::config().get("engine") == "mpi") {
87  std::stringstream ss;
88  ss << "mpiexec";
89  if (Global::config().has("hostfile")) {
90  ss << " --hostfile " << Global::config().get("hostfile");
91  }
92  ss << " -n " << std::to_string(numberOfProcesses);
93  ss << " " << binaryFilename;
94  exitCode = system(ss.str().c_str());
95  } else
96 #endif
97  {
98  exitCode = system(binaryFilename.c_str());
99  }
100 
101  if (Global::config().get("dl-program").empty()) {
102  remove(binaryFilename.c_str());
103  remove((binaryFilename + ".cpp").c_str());
104  }
105 
106  // exit with same code as executable
107  if (exitCode != 0) {
108  exit(exitCode);
109  }
110 }
