66  /** Insert tuple */
67  virtual void insert(const RamDomain* tuple) {
68  // check for null-arity
69  if (arity == 0) {
70  // set number of tuples to one -- that's it
71  num_tuples = 1;
72  return;
73  }
74 
75  assert(tuple);
76 
77  // make existence check
78  if (exists(tuple)) {
79  return;
80  }
81 
82  int blockIndex = num_tuples / (BLOCK_SIZE / arity);
83  int tupleIndex = (num_tuples % (BLOCK_SIZE / arity)) * arity;
84 
85  if (tupleIndex == 0) {
86  blockList.push_back(std::make_unique<RamDomain[]>(BLOCK_SIZE));
87  }
88 
89  RamDomain* newTuple = &blockList[blockIndex][tupleIndex];
90  for (size_t i = 0; i < arity; ++i) {
91  newTuple[i] = tuple[i];
92  }
93 
94  // update all indexes with new tuple
95  for (const auto& cur : indices) {
96  cur.second->insert(newTuple);
97  }
98 
99  // increment relation size
100  num_tuples++;
101  }

129  /** get index for a given set of keys. Keys are encoded as bits for each column */
130  InterpreterIndex* getIndex(const SearchColumns& key) const {
131  // suffix for order, if no matching prefix exists
132  std::vector<unsigned char> suffix;
133  suffix.reserve(getArity());
134 
135  // convert to order
136  InterpreterIndexOrder order;
137  for (size_t k = 1, i = 0; i < getArity(); i++, k *= 2) {
138  if (key & k) {
139  order.append(i);
140  } else {
141  suffix.push_back(i);
142  }
143  }
144 
145  // see whether there is an order with a matching prefix
146  InterpreterIndex* res = nullptr;
147  {
148  auto lease = lock.acquire();
149  (void)lease;
150 
151  for (auto it = indices.begin(); !res && it != indices.end(); ++it) {
152  if (order.isCompatible(it->first)) {
153  res = it->second.get();
154  }
155  }
156  }
157  // if found, use compatible index
158  if (res) {
159  return res;
160  }
161 
162  // extend index to full index
163  for (auto cur : suffix) {
164  order.append(cur);
165  }
166  assert(order.isComplete());
167 
168  // get a new index
169  return getIndex(order);
170  }

172  /** get index for a given order. Keys are encoded as bits for each column */
173  InterpreterIndex* getIndex(const InterpreterIndexOrder& order) const {
174  // TODO: improve index usage by re-using indices with common prefix
175  InterpreterIndex* res = nullptr;
176  {
177  auto lease = lock.acquire();
178  (void)lease;
179  auto pos = indices.find(order);
180  if (pos == indices.end()) {
181  std::unique_ptr<InterpreterIndex>& newIndex = indices[order];
182  newIndex = std::make_unique<InterpreterIndex>(order);
183  newIndex->insert(this->begin(), this->end());
184  res = newIndex.get();
185  } else {
186  res = pos->second.get();
187  }
188  }
189  return res;
190  }

366  relevantStored.push_back(vals);
