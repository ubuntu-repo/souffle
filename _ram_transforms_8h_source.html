72 /**
73  * @class HoistConditionsTransformer
74  * @brief Hosts conditions in a loop-nest to the most-outer/semantically-correct loop
75  *
76  * Hoists the conditions to the earliest point in the loop nest where their
77  * evaluation is still semantically correct.
78  *
79  * The transformations assumes that filter operations are stored verbose,
80  * i.e. a conjunction is expressed by two consecutive filter operations.
81  * For example ..
82  *
83  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
84  * QUERY
85  * ...
86  * IF C1 /\ C2 then
87  * ...
88  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
89  *
90  * should be rewritten / or produced by the translator as
91  *
92  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
93  * QUERY
94  * ...
95  * IF C1
96  * IF C2
97  * ...
98  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
99  *
100  * otherwise the levelling becomes imprecise, i.e., for both conditions
101  * the most outer-level is sought rather than considered separately.
102  *
103  * If there are transformers prior to hoistConditions() that introduce
104  * conjunction, another transformer is required that splits the
105  * filter operations. However, at the moment this is not necessary
106  * because the translator delivers already the right RAM format.
107  *
108  * TODO: break-up conditions while transforming so that this requirement
109  * is removed.
110  */
111 class HoistConditionsTransformer : public RamTransformer {
112 public:
113  std::string getName() const override {
114  return "HoistConditionsTransformer";
115  }
116 
117  /**
118  * @brief Hoist filter operations.
119  * @param Program that is transformed
120  * @return Flag showing whether the program has been changed by the transformation
121  *
122  * There are two types of conditions in
123  * filter operations. The first type depends on tuples of
124  * RamSearch operations. The second type are independent of
125  * tuple access. Both types of conditions will be hoisted to
126  * the most out-scope such that the program is still valid.
127  */
128  bool hoistConditions(RamProgram& program);
129 
130 protected:
131  RamLevelAnalysis* rla{nullptr};
132 
133  bool transform(RamTranslationUnit& translationUnit) override {
134  rla = translationUnit.getAnalysis<RamLevelAnalysis>();
135  return hoistConditions(*translationUnit.getProgram());
136  }
137 };

312  * ...

314  *
