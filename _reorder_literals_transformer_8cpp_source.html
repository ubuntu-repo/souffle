45 /**
46  * Counts the number of bound arguments in a given atom.
47  */
48 unsigned int numBoundArguments(const AstAtom* atom, const std::set<std::string>& boundVariables) {
49  int count = 0;
50 
51  for (const AstArgument* arg : atom->getArguments()) {
52  // argument is bound iff all contained variables are bound
53  bool isBound = true;
54 
55  visitDepthFirst(*arg, [&](const AstVariable& var) {
56  if (boundVariables.find(var.getName()) == boundVariables.end()) {
57  // found an unbound variable, so argument is unbound
58  isBound = false;
59  }
60  });
61 
62  if (isBound) {
63  count++;
64  }
65  }
66 
67  return count;
68 }

70 /**
71  * Returns a SIPS function based on the SIPS option provided.
72  * The SIPS function will return the index of the appropriate atom in a clause
73  * given a goal.
74  *
75  * E.g. the 'max-bound' SIPS function will return the atom in the clause with
76  * the maximum number of bound arguments.
77  */
78 sips_t getSipsFunction(const std::string& sipsChosen) {
79  // --- Create the appropriate SIPS function ---
80 
81  // Each SIPS function has a priority metric (e.g. max-bound atoms).
82  // Arguments:
83  // - a vector of atoms to choose from (nullpointers in the vector will be ignored)
84  // - the set of variables bound so far
85  // Returns: the index of the atom maximising the priority metric
86  sips_t getNextAtomSips;
87 
88  if (sipsChosen == "naive") {
89  // Goal: choose the first atom with at least one bound argument, or with no arguments
90  getNextAtomSips = [&](std::vector<AstAtom*> atoms, const std::set<std::string>& boundVariables) {
91  for (unsigned int i = 0; i < atoms.size(); i++) {
92  const AstAtom* currAtom = atoms[i];
93 
94  if (currAtom == nullptr) {
95  // already processed - move on
96  continue;
97  }
98 
99  if (isProposition(currAtom)) {
100  // no arguments
101  return i;
102  }
103 
104  if (numBoundArguments(currAtom, boundVariables) >= 1) {
105  // at least one bound argument
106  return i;
107  }
108  }
109 
110  // none found; choose the first non-null
111  for (unsigned int i = 0; i < atoms.size(); i++) {
112  if (atoms[i] != nullptr) {
113  return i;
114  }
115  }
116 
117  // fall back to the first
118  return 0U;
119  };
120  } else if (sipsChosen == "all-bound") {
121  // Goal: prioritise atoms with all arguments bound
122  getNextAtomSips = [&](std::vector<AstAtom*> atoms, const std::set<std::string>& boundVariables) {
123  for (unsigned int i = 0; i < atoms.size(); i++) {
124  const AstAtom* currAtom = atoms[i];
125 
126  if (currAtom == nullptr) {
127  // already processed - move on
128  continue;
129  }
130 
131  if (isProposition(currAtom)) {
132  // no arguments, so all are trivially bound
133  return i;
134  }
135 
136  int arity = currAtom->getArity();
137  int numBound = numBoundArguments(currAtom, boundVariables);
138  if (numBound == arity) {
139  // all arguments are bound!
140  return i;
141  }
142  }
143 
144  // none found; choose the first non-null
145  for (unsigned int i = 0; i < atoms.size(); i++) {
146  if (atoms[i] != nullptr) {
147  return i;
148  }
149  }
150 
151  // fall back to the first
152  return 0U;
153  };
154  } else if (sipsChosen == "max-bound") {
155  // Goal: choose the atom with the maximum number of bound variables
156  // - exception: propositions should be prioritised
157  getNextAtomSips = [&](std::vector<AstAtom*> atoms, const std::set<std::string>& boundVariables) {
158  int currMaxBound = -1;
159  unsigned int currMaxIdx = 0U;
160 
161  for (unsigned int i = 0; i < atoms.size(); i++) {
162  const AstAtom* currAtom = atoms[i];
163 
164  if (currAtom == nullptr) {
165  // already processed - move on
166  continue;
167  }
168 
169  if (isProposition(currAtom)) {
170  // propositions should be prioritised
171  return i;
172  }
173 
174  int numBound = numBoundArguments(currAtom, boundVariables);
175  if (numBound > currMaxBound) {
176  currMaxBound = numBound;
177  currMaxIdx = i;
178  }
179  }
180 
181  return currMaxIdx;
182  };
183  } else if (sipsChosen == "max-ratio") {
184  // Goal: choose the atom with the maximum ratio of bound to unbound
185  getNextAtomSips = [&](std::vector<AstAtom*> atoms, const std::set<std::string>& boundVariables) {
186  auto isLargerRatio = [&](std::pair<int, int> lhs, std::pair<int, int> rhs) {
187  return (lhs.first * rhs.second > lhs.second * rhs.first);
188  };
189 
190  auto currMaxRatio = std::pair<int, int>(-1, 1); // set to -1
191  unsigned int currMaxIdx = 0U;
192 
193  for (unsigned int i = 0; i < atoms.size(); i++) {
194  const AstAtom* currAtom = atoms[i];
195 
196  if (currAtom == nullptr) {
197  // already processed - move on
198  continue;
199  }
200 
201  if (isProposition(currAtom)) {
202  // propositions are as bound as possible
203  return i;
204  }
205 
206  int numBound = numBoundArguments(currAtom, boundVariables);
207  int numArgs = currAtom->getArity();
208  auto currRatio = std::pair<int, int>(numBound, numArgs);
209  if (isLargerRatio(currRatio, currMaxRatio)) {
210  currMaxRatio = currRatio;
211  currMaxIdx = i;
212  }
213  }
214 
215  return currMaxIdx;
216  };
217  } else if (sipsChosen == "least-free") {
218  // Goal: choose the atom with the least number of unbound arguments
219  getNextAtomSips = [&](std::vector<AstAtom*> atoms, const std::set<std::string>& boundVariables) {
220  int currLeastFree = -1;
221  unsigned int currLeastIdx = 0U;
222 
223  for (unsigned int i = 0; i < atoms.size(); i++) {
224  const AstAtom* currAtom = atoms[i];
225 
226  if (currAtom == nullptr) {
227  // already processed - move on
228  continue;
229  }
230 
231  if (isProposition(currAtom)) {
232  // propositions have 0 unbound arguments, which is minimal
233  return i;
234  }
235 
236  int numBound = numBoundArguments(currAtom, boundVariables);
237  int numFree = currAtom->getArity() - numBound;
238  if (currLeastFree == -1 || numFree < currLeastFree) {
239  currLeastFree = numFree;
240  currLeastIdx = i;
241  }
242  }
243 
244  return currLeastIdx;
245  };
246  } else if (sipsChosen == "least-free-vars") {
247  // Goal: choose the atom with the least amount of unbound variables
248  getNextAtomSips = [&](std::vector<AstAtom*> atoms, const std::set<std::string>& boundVariables) {
249  int currLeastFree = -1;
250  unsigned int currLeastIdx = 0U;
251 
252  for (unsigned int i = 0; i < atoms.size(); i++) {
253  const AstAtom* currAtom = atoms[i];
254 
255  if (currAtom == nullptr) {
256  // already processed - move on
257  continue;
258  }
259 
260  if (isProposition(currAtom)) {
261  // propositions have 0 unbound variables, which is minimal
262  return i;
263  }
264 
265  // use a set to hold all free variables to avoid double-counting
266  std::set<std::string> freeVars;
267  visitDepthFirst(*currAtom, [&](const AstVariable& var) {
268  if (boundVariables.find(var.getName()) == boundVariables.end()) {
269  freeVars.insert(var.getName());
270  }
271  });
272 
273  int numFreeVars = freeVars.size();
274  if (currLeastFree == -1 || numFreeVars < currLeastFree) {
275  currLeastFree = numFreeVars;
276  currLeastIdx = i;
277  }
278  }
279 
280  return currLeastIdx;
281  };
282  } else {
283  // chosen SIPS is not implemented, so keep the same order
284  // Goal: leftmost atom first
285  getNextAtomSips = [&](std::vector<AstAtom*> atoms, const std::set<std::string>& boundVariables) {
286  for (unsigned int i = 0; i < atoms.size(); i++) {
287  if (atoms[i] == nullptr) {
288  // already processed - move on
289  continue;
290  }
291 
292  return i;
293  }
294 
295  // fall back to the first
296  return 0U;
297  };
298  }
299 
300  return getNextAtomSips;
301 }

303 /**
304  * Finds the new ordering of a vector of atoms after the given SIPS is applied.
305  */
306 std::vector<unsigned int> applySips(sips_t sipsFunction, std::vector<AstAtom*> atoms) {
307  std::set<std::string> boundVariables;
308  std::vector<unsigned int> newOrder(atoms.size());
309 
310  unsigned int numAdded = 0;
311  while (numAdded < atoms.size()) {
312  // grab the next atom, based on the SIPS function
313  unsigned int nextIdx = sipsFunction(atoms, boundVariables);
314  AstAtom* nextAtom = atoms[nextIdx];
315 
316  // set all arguments that are variables as bound
317  // note: arguments that are functors, etc., do not newly bind anything
318  for (AstArgument* arg : nextAtom->getArguments()) {
319  if (auto* var = dynamic_cast<AstVariable*>(arg)) {
320  boundVariables.insert(var->getName());
321  }
322  }
323 
324  newOrder[numAdded] = nextIdx; // add to the ordering
325  atoms[nextIdx] = nullptr; // mark as done
326  numAdded++; // move on
327  }
328 
329  return newOrder;
330 }

332 bool reorderClauseWithSips(sips_t sipsFunction, AstClause* clause) {
333  // ignore clauses with fixed execution plans
334  if (clause->getExecutionPlan() != nullptr) {
335  return false;
336  }
337 
338  // get the ordering corresponding to the SIPS
339  std::vector<unsigned int> newOrdering = applySips(sipsFunction, clause->getAtoms());
340 
341  // reorder the clause accordingly
342  clause->reorderAtoms(newOrdering);
343 
344  // check if we have a change
345  for (unsigned int i = 0; i < newOrdering.size(); i++) {
346  if (newOrdering[i] != i) {
347  // changed
348  return true;
349  }
350  }
351 
352  // unchanged
353  return false;
354 }

356 bool ReorderLiteralsTransformer::transform(AstTranslationUnit& translationUnit) {
357  bool changed = false;
358  AstProgram& program = *translationUnit.getProgram();
359 
360  // --- SIPS-based static reordering ---
361  // ordering is based on the given SIPS
362 
363  // default SIPS to choose is 'all-bound'
364  std::string sipsChosen = "all-bound";
365  if (Global::config().has("SIPS")) {
366  sipsChosen = Global::config().get("SIPS");
367  }
368  auto sipsFunction = getSipsFunction(sipsChosen);
369 
370  // literal reordering is a rule-local transformation
371  for (const AstRelation* rel : program.getRelations()) {
372  for (AstClause* clause : rel->getClauses()) {
373  changed |= reorderClauseWithSips(sipsFunction, clause);
374  }
375  }
376 
377  // --- profile-guided reordering ---
378  if (Global::config().has("profile-use")) {
379  // parse supplied profile information
380  auto* profileUse = translationUnit.getAnalysis<AstProfileUse>();
381 
382  auto profilerSips = [&](std::vector<AstAtom*> atoms, const std::set<std::string>& boundVariables) {
383  // Goal: reorder based on the given profiling information
384  // Metric: cost(atom_R) = log(|atom_R|) * #free/#args
385  // - exception: propositions are prioritised
386 
387  double currOptimalVal = -1;
388  unsigned int currOptimalIdx = 0U;
389  bool set = false;
390 
391  for (unsigned int i = 0; i < atoms.size(); i++) {
392  const AstAtom* currAtom = atoms[i];
393 
394  if (currAtom == nullptr) {
395  // already processed - move on
396  continue;
397  }
398 
399  if (isProposition(currAtom)) {
400  // prioritise propositions
401  return i;
402  }
403 
404  // calculate log(|R|) * #free/#args
405  int numBound = numBoundArguments(currAtom, boundVariables);
406  int numArgs = currAtom->getArity();
407  int numFree = numArgs - numBound;
408  double value = log(profileUse->getRelationSize(currAtom->getName()));
409  value *= (numFree * 1.0) / numArgs;
410 
411  if (!set || value < currOptimalVal) {
412  set = true;
413  currOptimalVal = value;
414  currOptimalIdx = i;
415  }
416  }
417 
418  return currOptimalIdx;
419  };
420 
421  for (const AstRelation* rel : program.getRelations()) {
422  for (AstClause* clause : rel->getClauses()) {
423  changed |= reorderClauseWithSips(profilerSips, clause);
424  }
425  }
426  }
427 
428  return changed;
429 }
