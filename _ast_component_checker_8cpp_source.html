96 void AstComponentChecker::checkComponent(ErrorReport& report, const AstComponent* enclosingComponent,
97  const ComponentLookup& componentLookup, const AstComponent& component, const TypeBinding& binding) {
98  // -- inheritance --
99 
100  // Update type binding:
101  // Since we are not compiling, i.e. creating concrete instance of the
102  // components with type parameters, we are only interested in whether
103  // component references refer to existing components or some type parameter.
104  // Type parameter for us here is unknown type that will be bound at the template
105  // instantiation time.
106  auto parentTypeParameters = component.getComponentType()->getTypeParameters();
107  std::vector<AstTypeIdentifier> actualParams(parentTypeParameters.size(), "<type parameter>");
108  TypeBinding activeBinding = binding.extend(parentTypeParameters, actualParams);
109 
110  // check parents of component
111  for (const auto& cur : component.getBaseComponents()) {
112  checkComponentReference(
113  report, enclosingComponent, componentLookup, *cur, component.getSrcLoc(), activeBinding);
114 
115  // Note: type parameters can also be atomic types like number, or anything defined through .type
116  // The original semantic check permitted any identifier (existing or non-existing) to be actual
117  // parameter
118  // In order to maintain the compatibility, we do not check the actual parameters
119 
120  // for (const std::string& param : cur.getTypeParameters()) {
121  // checkComponentNameReference(report, scope, param, component.getSrcLoc(), activeBinding);
122  //}
123  }
124 
125  // get all parents
126  std::set<const AstComponent*> parents;
127  std::function<void(const AstComponent&)> collectParents = [&](const AstComponent& cur) {
128  for (const auto& base : cur.getBaseComponents()) {
129  auto c = componentLookup.getComponent(enclosingComponent, base->getName(), binding);
130  if (!c) {
131  continue;
132  }
133  if (parents.insert(c).second) {
134  collectParents(*c);
135  }
136  }
137  };
138  collectParents(component);
139 
140  // check overrides
141  for (const AstRelation* relation : component.getRelations()) {
142  if (component.getOverridden().count(relation->getName().getNames()[0])) {
143  report.addError("Override of non-inherited relation " + relation->getName().getNames()[0] +
144  " in component " + component.getComponentType()->getName(),
145  component.getSrcLoc());
146  }
147  }
148  for (const AstComponent* parent : parents) {
149  for (const AstRelation* relation : parent->getRelations()) {
150  if (component.getOverridden().count(relation->getName().getNames()[0]) &&
151  !relation->isOverridable()) {
152  report.addError("Override of non-overridable relation " + relation->getName().getNames()[0] +
153  " in component " + component.getComponentType()->getName(),
154  component.getSrcLoc());
155  }
156  }
157  }
158 
159  // check for a cycle
160  if (parents.find(&component) != parents.end()) {
161  report.addError(
162  "Invalid cycle in inheritance for component " + component.getComponentType()->getName(),
163  component.getSrcLoc());
164  }
165 
166  // -- nested components --
167 
168  // check nested components
169  for (const auto& cur : component.getComponents()) {
170  checkComponent(report, &component, componentLookup, *cur, activeBinding);
171  }
172 
173  // check nested instantiations
174  for (const auto& cur : component.getInstantiations()) {
175  checkComponentInit(report, &component, componentLookup, *cur, activeBinding);
176  }
177 }

190 // Check that component names are disjoint from type and relation names.
191 void AstComponentChecker::checkComponentNamespaces(ErrorReport& report, const AstProgram& program) {
192  std::map<std::string, SrcLocation> names;
193 
194  // Type and relation name error reporting performed by the AstSemanticChecker instead
195 
196  // Find all names and report redeclarations as we go.
197  for (const auto& type : program.getTypes()) {
198  const std::string name = toString(type->getName());
199  if (!names.count(name)) {
200  names[name] = type->getSrcLoc();
201  }
202  }
203 
204  for (const auto& rel : program.getRelations()) {
205  const std::string name = toString(rel->getName());
206  if (!names.count(name)) {
207  names[name] = rel->getSrcLoc();
208  }
209  }
210 
211  // Note: Nested component and instance names are not obtained.
212  for (const auto& comp : program.getComponents()) {
213  const std::string name = toString(comp->getComponentType()->getName());
214  if (names.count(name)) {
215  report.addError("Name clash on component " + name, comp->getSrcLoc());
216  } else {
217  names[name] = comp->getSrcLoc();
218  }
219  }
220 
221  for (const auto& inst : program.getComponentInstantiations()) {
222  const std::string name = toString(inst->getInstanceName());
223  if (names.count(name)) {
224  report.addError("Name clash on instantiation " + name, inst->getSrcLoc());
225  } else {
226  names[name] = inst->getSrcLoc();
227  }
228  }
229 }
