35 /**
36  * A utility function utilized by the finishParsing member function to update a type environment
37  * out of a given list of types in the AST
38  *
39  * @param types the types specified in the input file, contained in the AST
40  * @param env the type environment to be updated
41  */
42 void TypeEnvironmentAnalysis::updateTypeEnvironment(const AstProgram& program) {
43  // build up new type system based on defined types
44 
45  // create all type symbols in a first step
46  for (const auto& cur : program.getTypes()) {
47  // support faulty codes with multiple definitions
48  if (env.isType(cur->getName())) {
49  continue;
50  }
51 
52  // create type within type environment
53  if (auto* t = dynamic_cast<const AstPrimitiveType*>(cur)) {
54  if (t->isNumeric()) {
55  env.createNumericType(cur->getName());
56  } else {
57  env.createSymbolType(cur->getName());
58  }
59  } else if (dynamic_cast<const AstUnionType*>(cur)) {
60  // initialize the union
61  env.createUnionType(cur->getName());
62  } else if (dynamic_cast<const AstRecordType*>(cur)) {
63  // initialize the record
64  env.createRecordType(cur->getName());
65  } else {
66  std::cout << "Unsupported type construct: " << typeid(cur).name() << "\n";
67  assert(false && "Unsupported Type Construct!");
68  }
69  }
70 
71  // link symbols in a second step
72  for (const auto& cur : program.getTypes()) {
73  Type* type = env.getModifiableType(cur->getName());
74  assert(type && "It should be there!");
75 
76  if (dynamic_cast<const AstPrimitiveType*>(cur)) {
77  // nothing to do here
78  } else if (auto* t = dynamic_cast<const AstUnionType*>(cur)) {
79  // get type as union type
80  auto* ut = dynamic_cast<UnionType*>(type);
81  if (!ut) {
82  continue; // support faulty input
83  }
84 
85  // add element types
86  for (const auto& cur : t->getTypes()) {
87  if (env.isType(cur)) {
88  ut->add(env.getType(cur));
89  }
90  }
91  } else if (auto* t = dynamic_cast<const AstRecordType*>(cur)) {
92  // get type as record type
93  auto* rt = dynamic_cast<RecordType*>(type);
94  if (!rt) {
95  continue; // support faulty input
96  }
97 
98  // add fields
99  for (const auto& f : t->getFields()) {
100  if (env.isType(f.type)) {
101  rt->add(f.name, env.getType(f.type));
102  }
103  }
104  } else {
105  std::cout << "Unsupported type construct: " << typeid(cur).name() << "\n";
106  assert(false && "Unsupported Type Construct!");
107  }
108  }
109 }
