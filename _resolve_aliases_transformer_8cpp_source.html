43 /**
44  * A utility class for the unification process required to eliminate aliases.
45  * A substitution maps variables to terms and can be applied as a transformation
46  * to AstArguments.
47  */
48 class Substitution {
49  // map type used for internally storing var->term mappings
50  // - note: variables are identified by their names
51  using map_t = std::map<std::string, std::unique_ptr<AstArgument>>;
52 
53  // the mapping of variables to terms
54  map_t varToTerm;
55 
56 public:
57  // -- Constructors/Destructors --
58 
59  Substitution() = default;
60 
61  Substitution(const std::string& var, const AstArgument* arg) {
62  varToTerm.insert(std::make_pair(var, std::unique_ptr<AstArgument>(arg->clone())));
63  }
64 
65  ~Substitution() = default;
66 
67  /**
68  * Applies this substitution to the given argument and returns a pointer
69  * to the modified argument.
70  *
71  * @param node the node to be transformed
72  * @return a pointer to the modified or replaced node
73  */
74  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const {
75  // create a substitution mapper
76  struct M : public AstNodeMapper {
77  const map_t& map;
78 
79  M(const map_t& map) : map(map) {}
80 
81  using AstNodeMapper::operator();
82 
83  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
84  // see whether it is a variable to be substituted
85  if (auto var = dynamic_cast<AstVariable*>(node.get())) {
86  auto pos = map.find(var->getName());
87  if (pos != map.end()) {
88  return std::unique_ptr<AstNode>(pos->second->clone());
89  }
90  }
91 
92  // otherwise, apply the mapper recursively
93  node->apply(*this);
94  return node;
95  }
96  };
97 
98  // apply the mapper
99  return M(varToTerm)(std::move(node));
100  }
101 
102  /**
103  * A generic, type consistent wrapper of the transformation operation above.
104  */
105  template <typename T>
106  std::unique_ptr<T> operator()(std::unique_ptr<T> node) const {
107  std::unique_ptr<AstNode> resPtr = (*this)(std::unique_ptr<AstNode>(node.release()));
108  assert(nullptr != dynamic_cast<T*>(resPtr.get()) && "Invalid node type mapping.");
109  return std::unique_ptr<T>(dynamic_cast<T*>(resPtr.release()));
110  }
111 
112  /**
113  * Appends the given substitution s to this substitution t such that the
114  * result t' is s composed with t (s o t).
115  * i.e.,
116  * - if t(x) = y, then t'(x) = s(y)
117  * - if s(x) = y, and x is not mapped by t, then t'(x) = y
118  */
119  void append(const Substitution& sub) {
120  // apply substitution on the rhs of all current mappings
121  for (auto& pair : varToTerm) {
122  pair.second = sub(std::move(pair.second));
123  }
124 
125  // append unseen variables to the end
126  for (const auto& pair : sub.varToTerm) {
127  if (varToTerm.find(pair.first) == varToTerm.end()) {
128  // not seen yet, add it in
129  varToTerm.insert(
130  std::make_pair(pair.first, std::unique_ptr<AstArgument>(pair.second->clone())));
131  }
132  }
133  }
134 
135  /** A print function (for debugging) */
136  void print(std::ostream& out) const {
137  out << "{"
138  << join(varToTerm, ",",
139  [](std::ostream& out,
140  const std::pair<const std::string, std::unique_ptr<AstArgument>>& cur) {
141  out << cur.first << " -> " << *cur.second;
142  })
143  << "}";
144  }
145 
146  friend std::ostream& operator<<(std::ostream& out, const Substitution& s) __attribute__((unused)) {
147  s.print(out);
148  return out;
149  }
150 };

201 std::unique_ptr<AstClause> ResolveAliasesTransformer::resolveAliases(const AstClause& clause) {
202  // -- utilities --
203 
204  // tests whether something is a variable
205  auto isVar = [&](const AstArgument& arg) { return dynamic_cast<const AstVariable*>(&arg); };
206 
207  // tests whether something is a record
208  auto isRec = [&](const AstArgument& arg) { return dynamic_cast<const AstRecordInit*>(&arg); };
209 
210  // tests whether a value `a` occurs in a term `b`
211  auto occurs = [](const AstArgument& a, const AstArgument& b) {
212  bool res = false;
213  visitDepthFirst(b, [&](const AstArgument& arg) { res = res || (arg == a); });
214  return res;
215  };
216 
217  // find all variables appearing as functorless arguments in grounding atoms
218  // these variables are the source of groundedness
219  // e.g. a(y) :- b(x), y = x + 1. -- y is only grounded because x appears in b(x)
220  std::set<std::string> baseGroundedVariables;
221  for (const AstAtom* atom : clause.getAtoms()) {
222  for (const AstArgument* arg : atom->getArguments()) {
223  if (const auto* var = dynamic_cast<const AstVariable*>(arg)) {
224  baseGroundedVariables.insert(var->getName());
225  }
226  }
227  }
228 
229  // I) extract equations
230  std::vector<Equation> equations;
231  visitDepthFirst(clause, [&](const AstBinaryConstraint& constraint) {
232  if (constraint.getOperator() == BinaryConstraintOp::EQ) {
233  equations.push_back(Equation(constraint.getLHS(), constraint.getRHS()));
234  }
235  });
236 
237  // II) compute unifying substitution
238  Substitution substitution;
239 
240  // a utility for processing newly identified mappings
241  auto newMapping = [&](const std::string& var, const AstArgument* term) {
242  // found a new substitution
243  Substitution newMapping(var, term);
244 
245  // apply substitution to all remaining equations
246  for (auto& equation : equations) {
247  equation.apply(newMapping);
248  }
249 
250  // add mapping v -> t to substitution
251  substitution.append(newMapping);
252  };
253 
254  while (!equations.empty()) {
255  // get next equation to compute
256  Equation equation = equations.back();
257  equations.pop_back();
258 
259  // shortcuts for left/right
260  const AstArgument& lhs = *equation.lhs;
261  const AstArgument& rhs = *equation.rhs;
262 
263  // #1: t = t => skip
264  if (lhs == rhs) {
265  continue;
266  }
267 
268  // #2: [..] = [..] => decompose
269  if (isRec(lhs) && isRec(rhs)) {
270  // get arguments
271  const auto& lhs_args = static_cast<const AstRecordInit&>(lhs).getArguments();
272  const auto& rhs_args = static_cast<const AstRecordInit&>(rhs).getArguments();
273 
274  // make sure sizes are identical
275  assert(lhs_args.size() == rhs_args.size() && "Record lengths not equal");
276 
277  // create new equalities
278  for (size_t i = 0; i < lhs_args.size(); i++) {
279  equations.push_back(Equation(lhs_args[i], rhs_args[i]));
280  }
281 
282  continue;
283  }
284 
285  // #3: neither is a variable => skip
286  if (!isVar(lhs) && !isVar(rhs)) {
287  continue;
288  }
289 
290  // #4: v = w => add mapping
291  if (isVar(lhs) && isVar(rhs)) {
292  auto& var = static_cast<const AstVariable&>(lhs);
293  newMapping(var.getName(), &rhs);
294  continue;
295  }
296 
297  // #5: t = v => swap
298  if (!isVar(lhs)) {
299  equations.push_back(Equation(rhs, lhs));
300  continue;
301  }
302 
303  // now we know lhs is a variable
304  assert(isVar(lhs));
305 
306  // therefore, we have v = t
307  const auto& v = static_cast<const AstVariable&>(lhs);
308  const AstArgument& t = rhs;
309 
310  // #6: v occurs in t => skip
311  if (occurs(v, t)) {
312  continue;
313  }
314 
315  assert(!occurs(v, t));
316 
317  // #7: t is a record => add mapping
318  if (isRec(t)) {
319  newMapping(v.getName(), &t);
320  continue;
321  }
322 
323  // #8: v is already grounded => skip
324  auto pos = baseGroundedVariables.find(v.getName());
325  if (pos != baseGroundedVariables.end()) {
326  continue;
327  }
328 
329  // add new mapping
330  newMapping(v.getName(), &t);
331  }
332 
333  // III) compute resulting clause
334  return substitution(std::unique_ptr<AstClause>(clause.clone()));
335 }

337 std::unique_ptr<AstClause> ResolveAliasesTransformer::removeTrivialEquality(const AstClause& clause) {
338  std::unique_ptr<AstClause> res(clause.cloneHead());
339 
340  // add all literals, except filtering out t = t constraints
341  for (AstLiteral* literal : clause.getBodyLiterals()) {
342  if (auto* constraint = dynamic_cast<AstBinaryConstraint*>(literal)) {
343  if (constraint->getOperator() == BinaryConstraintOp::EQ) {
344  if (*constraint->getLHS() == *constraint->getRHS()) {
345  continue; // skip this one
346  }
347  }
348  }
349 
350  res->addToBody(std::unique_ptr<AstLiteral>(literal->clone()));
351  }
352 
353  // done
354  return res;
355 }

357 std::unique_ptr<AstClause> ResolveAliasesTransformer::removeComplexTermsInAtoms(const AstClause& clause) {
358  std::unique_ptr<AstClause> res(clause.clone());
359 
360  // get list of atoms
361  std::vector<AstAtom*> atoms = res->getAtoms();
362 
363  // find all functors in atoms
364  std::vector<const AstArgument*> terms;
365  for (const AstAtom* atom : atoms) {
366  for (const AstArgument* arg : atom->getArguments()) {
367  // ignore if not a functor
368  if (dynamic_cast<const AstFunctor*>(arg) == nullptr) {
369  continue;
370  }
371 
372  // add this functor if not seen yet
373  if (!any_of(terms, [&](const AstArgument* cur) { return *cur == *arg; })) {
374  terms.push_back(arg);
375  }
376  }
377  }
378 
379  // substitute them with new variables (a real map would compare pointers)
380  using substitution_map =
381  std::vector<std::pair<std::unique_ptr<AstArgument>, std::unique_ptr<AstVariable>>>;
382  substitution_map termToVar;
383 
384  int varCounter = 0;
385  for (const AstArgument* arg : terms) {
386  // create a new mapping for this term
387  auto term = std::unique_ptr<AstArgument>(arg->clone());
388  auto newVariable = std::make_unique<AstVariable>(" _tmp_" + toString(varCounter++));
389  termToVar.push_back(std::make_pair(std::move(term), std::move(newVariable)));
390  }
391 
392  // apply mapping to replace the terms with the variables
393  struct Update : public AstNodeMapper {
394  const substitution_map& map;
395 
396  Update(const substitution_map& map) : map(map) {}
397 
398  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
399  // check whether node needs to be replaced
400  for (const auto& pair : map) {
401  auto& term = pair.first;
402  auto& variable = pair.second;
403 
404  if (*term == *node) {
405  return std::unique_ptr<AstNode>(variable->clone());
406  }
407  }
408 
409  // continue recursively
410  node->apply(*this);
411  return node;
412  }
413  };
414 
415  // update atoms
416  Update update(termToVar);
417  for (AstAtom* atom : atoms) {
418  atom->apply(update);
419  }
420 
421  // add the necessary variable constraints to the clause
422  for (const auto& pair : termToVar) {
423  auto& term = pair.first;
424  auto& variable = pair.second;
425 
426  res->addToBody(std::make_unique<AstBinaryConstraint>(BinaryConstraintOp::EQ,
427  std::unique_ptr<AstArgument>(variable->clone()),
428  std::unique_ptr<AstArgument>(term->clone())));
429  }
430 
431  return res;
432 }

434 bool ResolveAliasesTransformer::transform(AstTranslationUnit& translationUnit) {
435  bool changed = false;
436  AstProgram& program = *translationUnit.getProgram();
437 
438  // get all clauses
439  std::vector<const AstClause*> clauses;
440  visitDepthFirst(program, [&](const AstRelation& rel) {
441  for (const auto& clause : rel.getClauses()) {
442  clauses.push_back(clause);
443  }
444  });
445 
446  // clean all clauses
447  for (const AstClause* clause : clauses) {
448  // -- Step 1 --
449  // get rid of aliases
450  std::unique_ptr<AstClause> noAlias = resolveAliases(*clause);
451 
452  // clean up equalities
453  std::unique_ptr<AstClause> cleaned = removeTrivialEquality(*noAlias);
454 
455  // -- Step 2 --
456  // restore simple terms in atoms
457  std::unique_ptr<AstClause> normalised = removeComplexTermsInAtoms(*cleaned);
458 
459  // swap if changed
460  if (*normalised != *clause) {
461  changed = true;
462  program.removeClause(clause);
463  program.appendClause(std::move(normalised));
464  }
465  }
466 
467  return changed;
468 }
