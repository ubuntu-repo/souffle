31 /**
32  * Class that encapsulates std::map of types binding that comes from .init c = Comp<MyType>
33  * Type binding in this example would be T->MyType if the component code is .comp Comp<T> ...
34  */
35 class TypeBinding {
36 public:
37  /**
38  * Returns binding for given name or empty string if such binding does not exist.
39  */
40  const AstTypeIdentifier& find(const AstTypeIdentifier& name) const {
41  const static AstTypeIdentifier unknown;
42  auto pos = binding.find(name);
43  if (pos == binding.end()) {
44  return unknown;
45  }
46  return pos->second;
47  }
48 
49  TypeBinding extend(const std::vector<AstTypeIdentifier>& formalParams,
50  const std::vector<AstTypeIdentifier>& actualParams) const {
51  TypeBinding result;
52  if (formalParams.size() != actualParams.size()) {
53  return *this; // invalid init => will trigger a semantic error
54  }
55 
56  for (std::size_t i = 0; i < formalParams.size(); i++) {
57  auto pos = binding.find(actualParams[i]);
58  if (pos != binding.end()) {
59  result.binding[formalParams[i]] = pos->second;
60  } else {
61  result.binding[formalParams[i]] = actualParams[i];
62  }
63  }
64 
65  return result;
66  }
67 
68 private:
69  /**
70  * Key value pair. Keys are names that should be forwarded to value,
71  * which is the actual name. Example T->MyImplementation.
72  */
73  std::map<AstTypeIdentifier, AstTypeIdentifier> binding;
74 };
