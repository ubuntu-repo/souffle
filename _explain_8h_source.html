223  // parse relation, split into relation name and values
224  std::pair<std::string, std::vector<std::string>> parseTuple(const std::string& str) {
225  std::string relName;
226  std::vector<std::string> args;
227 
228  // regex for matching tuples
229  // values matches numbers or strings enclosed in quotation marks
230  std::regex relRegex(
231  "([a-zA-Z0-9_.-]*)[[:blank:]]*\\(([[:blank:]]*([0-9]+|\"[^\"]*\")([[:blank:]]*,[[:blank:]]*(["
232  "0-"
233  "9]+|\"[^\"]*\"))*)?\\)",
234  std::regex_constants::extended);
235  std::smatch relMatch;
236 
237  // first check that format matches correctly
238  // and extract relation name
239  if (!std::regex_match(str, relMatch, relRegex) || relMatch.size() < 3) {
240  return std::make_pair(relName, args);
241  }
242 
243  // set relation name
244  relName = relMatch[1];
245 
246  // extract each argument
247  std::string argsList = relMatch[2];
248  std::smatch argsMatcher;
249  std::regex argRegex(R"([0-9]+|"[^"]*")", std::regex_constants::extended);
250 
251  while (std::regex_search(argsList, argsMatcher, argRegex)) {
252  // match the start of the arguments
253  std::string currentArg = argsMatcher[0];
254  args.push_back(currentArg);
255 
256  // use the rest of the arguments
257  argsList = argsMatcher.suffix().str();
258  }
259 
260  return std::make_pair(relName, args);
261  }

317  // allow scrolling of provenance tree
318  void scrollTree(int maxx, int maxy) {
319  int x = 0;
320  int y = 0;
321 
322  while (true) {
323  int ch = wgetch(treePad);
324 
325  if (ch == KEY_LEFT) {
326  if (x > 2) x -= 3;
327  } else if (ch == KEY_RIGHT) {
328  if (x < MAX_TREE_WIDTH - 3) x += 3;
329  } else if (ch == KEY_UP) {
330  if (y > 0) y -= 1;
331  } else if (ch == KEY_DOWN) {
332  if (y < MAX_TREE_HEIGHT - 1) y += 1;
333  } else {
334  ungetch(ch);
335  break;
336  }
337 
338  prefresh(treePad, y, x, 0, 0, maxy - 3, maxx - 1);
