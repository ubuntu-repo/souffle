111  return interpreter.getSymbolTable().lookup(std::to_string(visit(args[0])));

114  case FunctorOp::ADD: {
115  return visit(args[0]) + visit(args[1]);
116  }
117  case FunctorOp::SUB: {
118  return visit(args[0]) - visit(args[1]);
119  }
120  case FunctorOp::MUL: {
121  return visit(args[0]) * visit(args[1]);
122  }

124  return visit(args[0]) / visit(args[1]);

129  case FunctorOp::MOD: {

131  }
132  case FunctorOp::BAND: {
133  return visit(args[0]) & visit(args[1]);
134  }
135  case FunctorOp::BOR: {
136  return visit(args[0]) | visit(args[1]);
137  }
138  case FunctorOp::BXOR: {
139  return visit(args[0]) ^ visit(args[1]);
140  }
141  case FunctorOp::LAND: {
142  return visit(args[0]) && visit(args[1]);
143  }
144  case FunctorOp::LOR: {
