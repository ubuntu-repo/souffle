113 /**
114  * The class utilized to model user-defined execution plans for various

117 class AstExecutionPlan : public AstNode {
118 public:
119  using const_iterator = typename std::map<int, AstExecutionOrder>::const_iterator;
120 
121  AstExecutionPlan() = default;
122 
123  /** Updates the execution order for a special version of a rule */
124  void setOrderFor(int version, std::unique_ptr<AstExecutionOrder> plan) {
125  plans[version] = std::move(plan);
126  if (version > maxVersion) {
127  maxVersion = version;
128  }
129  }
130 
131  /** Determines whether for the given version a plan has been specified */
132  bool hasOrderFor(int version) const {
133  return plans.find(version) != plans.end();
134  }
135 
136  /** get maximal version number */
137  const int getMaxVersion() const {
138  return maxVersion;
139  }
140 
141  /** Obtains the order defined for the given version */
142  const AstExecutionOrder& getOrderFor(int version) const {
143  assert(hasOrderFor(version));
144  return *plans.find(version)->second;
145  }
146 
147  /** Tests whether there has any order been defined */
148  bool empty() const {
149  return plans.empty();
150  }
151 
152  std::map<int, const AstExecutionOrder*> getOrders() const {
153  std::map<int, const AstExecutionOrder*> result;
154  for (auto& plan : plans) {
155  result.insert(std::make_pair(plan.first, plan.second.get()));
156  }
157  return result;
158  }
159 
160  void print(std::ostream& out) const override {
161  if (!plans.empty()) {
162  out << "\n\n .plan ";
163  bool first = true;
164  for (const auto& plan : plans) {
165  if (first) {
166  first = false;
167  } else {
168  out << ",";
169  }
170  out << plan.first << ":";
171  plan.second->print(out);
172  }
173  }
174  }
175 
176  AstExecutionPlan* clone() const override {
177  auto res = new AstExecutionPlan();
178  res->setSrcLoc(getSrcLoc());
179  for (auto& plan : plans) {
180  res->setOrderFor(plan.first, std::unique_ptr<AstExecutionOrder>(plan.second->clone()));
181  }
182  return res;
183  }
184 
185  void apply(const AstNodeMapper& map) override {
186  for (auto& plan : plans) {
187  plan.second = map(std::move(plan.second));
188  }
189  }
190 
191  /** Obtains a list of all embedded child nodes */
192  std::vector<const AstNode*> getChildNodes() const override {
193  std::vector<const AstNode*> childNodes;
194  for (auto& plan : plans) {
195  childNodes.push_back(plan.second.get());
196  }
197  return childNodes;
198  }
199 
200 protected:
201  /** Implements the node comparison for this node type */
202  bool equal(const AstNode& node) const override {
203  assert(nullptr != dynamic_cast<const AstExecutionPlan*>(&node));
204  const auto& other = static_cast<const AstExecutionPlan&>(node);
205  if (maxVersion != other.maxVersion) {
206  return false;
207  }
208  if (plans.size() != other.plans.size()) {
209  return false;
210  }
211  auto iter = plans.begin();
212  auto otherIter = other.plans.begin();
213  for (; iter != plans.end(); ++iter, ++otherIter) {
214  if (iter->first != otherIter->first) {
215  return false;
216  }
217  if (*iter->second != *otherIter->second) {
218  return false;
219  }
220  }
221  return true;
222  }
223 
224 private:
225  /** Mapping versions of clauses to execution plans */
226  std::map<int, std::unique_ptr<AstExecutionOrder>> plans;
227 
228  /** remember maximal version number */
229  int maxVersion = -1;
230 };

232 /**
233  * Intermediate representation of a datalog clause.
234  *
235  * A clause can either be:
236  * - a fact - a clause with no body (e.g., X(a,b))
237  * - a rule - a clause with a head and a body (e.g., Y(a,b) -: X(a,b))
238  *
239  * TODO: Currently Clause object is used to represent 2 different types of datalog
240  * clauses, such as rules, queries and facts. This solution was to quickly
