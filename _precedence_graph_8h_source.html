158  std::set<size_t> successorSccs;

184  const std::set<const AstRelation*> getExternalOutputPredecessorRelations(const size_t scc) const {
185  std::set<const AstRelation*> externOutPreds;
186  for (const auto& relation : getInternalRelations(scc)) {

233  /** Get all internal relations of a given SCC with external successors. */
234  const std::set<const AstRelation*> getInternalRelationsWithExternalSuccessors(const size_t scc) const {
235  std::set<const AstRelation*> internsWithExternSuccs;
236  for (const auto& relation : getInternalRelations(scc)) {
237  for (const auto& successor : precedenceGraph->graph().successors(relation)) {

251  for (const auto& relation : getInternalRelations(scc)) {
252  if (!ioType->isOutput(relation)) {
253  for (const auto& successor : precedenceGraph->graph().successors(relation)) {
254  if (relationToScc.at(successor) != scc) {
255  internNonOutsWithExternSuccs.insert(relation);
256  break;
257  }
258  }
259  }
260  }
261  return internNonOutsWithExternSuccs;
262  }
263 
264  /** Get all internal input relations of a given SCC. */
265  const std::set<const AstRelation*> getInternalInputRelations(const size_t scc) const {
266  std::set<const AstRelation*> internIns;
267  for (const auto& relation : getInternalRelations(scc)) {

275  /** Return if the given SCC is recursive. */
276  bool isRecursive(const size_t scc) const {
277  const std::set<const AstRelation*>& sccRelations = sccToRelation.at(scc);
278  if (sccRelations.size() == 1) {
279  const AstRelation* singleRelation = *sccRelations.begin();
280  if (!precedenceGraph->graph().predecessors(singleRelation).count(singleRelation)) {
281  return false;
282  }
283  }
284  return true;
285  }
