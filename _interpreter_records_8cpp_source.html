32 class RecordMap {

42 public:
43  RecordMap(int arity) : arity(arity), i2r(1) {} // note: index 0 element left free

45  /**
46  * Packs the given tuple -- and may create a new reference if necessary.
47  */
48  RamDomain pack(const RamDomain* tuple) {
49  vector<RamDomain> tmp(arity);
50  for (int i = 0; i < arity; i++) {
51  tmp[i] = tuple[i];
52  }
53 
54  RamDomain index;
55 #pragma omp critical(record_pack)
56  {
57  auto pos = r2i.find(tmp);
58  if (pos != r2i.end()) {
59  index = pos->second;
60  } else {
61 #pragma omp critical(record_unpack)
62  {
63  i2r.push_back(tmp);
64  index = i2r.size() - 1;
65  r2i[tmp] = index;
66 
67  // assert that new index is smaller than the range
68  assert(index != std::numeric_limits<RamDomain>::max());
69  }
70  }
71  }
72 
73  return index;
74  }
75 
76  /**
77  * Obtains a pointer to the tuple addressed by the given index.
78  */
79  RamDomain* unpack(RamDomain index) {
80  RamDomain* res;
81 
82 #pragma omp critical(record_unpack)
83  res = &(i2r[index][0]);
84 
85  return res;
86  }
87 };
