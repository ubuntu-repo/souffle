32 void AstClause::addToBody(std::unique_ptr<AstLiteral> l) {
33  if (dynamic_cast<AstAtom*>(l.get())) {
34  atoms.emplace_back(static_cast<AstAtom*>(l.release()));
35  } else if (dynamic_cast<AstNegation*>(l.get())) {
36  negations.emplace_back(static_cast<AstNegation*>(l.release()));
37  } else if (dynamic_cast<AstProvenanceNegation*>(l.get())) {
38  provNegations.emplace_back(static_cast<AstProvenanceNegation*>(l.release()));
39  } else if (dynamic_cast<AstConstraint*>(l.get())) {
40  constraints.emplace_back(static_cast<AstConstraint*>(l.release()));
41  } else {
42  assert(false && "Unsupported literal type!");
43  }
44 }

85 bool AstClause::isFact() const {
86  // there must be a head
87  if (head == nullptr) {
88  return false;
89  }
90  // there must not be any body clauses
91  if (getBodySize() != 0) {
92  return false;

114 void AstClause::reorderAtoms(const std::vector<unsigned int>& newOrder) {
115  // Validate given order
116  assert(newOrder.size() == atoms.size());
117  std::vector<unsigned int> nopOrder;
118  for (unsigned int i = 0; i < atoms.size(); i++) {
119  nopOrder.push_back(i);
120  }
121  assert(std::is_permutation(nopOrder.begin(), nopOrder.end(), newOrder.begin()));
122 
123  // Reorder atoms

125  atoms.swap(oldAtoms);
126  for (unsigned int i = 0; i < newOrder.size(); i++) {
127  atoms[i].swap(oldAtoms[newOrder[i]]);
128  }
