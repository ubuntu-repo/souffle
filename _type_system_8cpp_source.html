119  * A visitor for Types.

121 template <typename R>
122 struct TypeVisitor {
123  virtual ~TypeVisitor() = default;
124 
125  R operator()(const Type& type) const {
126  return visit(type);
127  }
128 
129  virtual R visit(const Type& type) const {
130  // check all kinds of types and dispatch
131  if (auto* t = dynamic_cast<const PredefinedType*>(&type)) {
132  return visitPredefinedType(*t);
133  }
134  if (auto* t = dynamic_cast<const PrimitiveType*>(&type)) {
135  return visitPrimitiveType(*t);
136  }
137  if (auto* t = dynamic_cast<const UnionType*>(&type)) {
138  return visitUnionType(*t);
139  }
140  if (auto* t = dynamic_cast<const RecordType*>(&type)) {
141  return visitRecordType(*t);
142  }
143  assert(false && "Unsupported type encountered!");
144  return R();
145  }
146 
147  virtual R visitPredefinedType(const PredefinedType& type) const {
148  return visitType(type);
149  }
150 
151  virtual R visitPrimitiveType(const PrimitiveType& type) const {
152  return visitType(type);
153  }
154 
155  virtual R visitUnionType(const UnionType& type) const {
156  return visitType(type);
157  }
158 
159  virtual R visitRecordType(const RecordType& type) const {
160  return visitType(type);
161  }
162 
163  virtual R visitType(const Type& /*type*/) const {
164  return R();
165  }

168 /**
169  * A visitor for types visiting each type only once (effectively breaking
170  * recursive cycles).
171  */
172 template <typename R>
173 class VisitOnceTypeVisitor : public TypeVisitor<R> {
174 protected:
175  mutable std::map<const Type*, R> seen;
176 
177 public:
178  R visit(const Type& type) const override {
179  auto pos = seen.find(&type);
180  if (pos != seen.end()) {
181  return pos->second;
182  }
183  auto& res = seen[&type]; // mark as seen
184  return res = TypeVisitor<R>::visit(type);
185  }
186 };

201 bool isOfRootType(const Type& type, const Type& root) {
202  struct visitor : public VisitOnceTypeVisitor<bool> {
203  const Type& root;
204 
205  visitor(const Type& root) : root(root) {}
206 
207  bool visitPredefinedType(const PredefinedType& type) const override {
208  return type == root;
209  }
210  bool visitPrimitiveType(const PrimitiveType& type) const override {
211  return type == root || type.getBaseType() == root || isOfRootType(type.getBaseType(), root);
212  }
213  bool visitUnionType(const UnionType& type) const override {
214  if (type.getElementTypes().empty()) {
215  return false;
216  }
217  auto fit = [&](const Type* cur) { return visit(*cur); };
218  return all_of(type.getElementTypes(), fit);
219  }
220  bool visitType(const Type& /*unused*/) const override {
221  return false;
222  }
223  };
224 
225  return visitor(root).visit(type);
226 }

336  const Type& trg;
337  visitor(const Type& trg) : trg(trg) {}
338  bool visit(const Type& type) const override {

348  bool visitRecordType(const RecordType& type) const override {
349  auto reachesTrg = [&](const RecordType::Field& cur) { return this->visit(cur.type); };
350  return any_of(type.getFields(), reachesTrg);
351  }
352  };

355  if (const auto* r = dynamic_cast<const RecordType*>(&type)) {
356  auto reachesOrigin = visitor(type);
357  return any_of(r->getFields(),

364 bool isSubtypeOf(const Type& a, const Type& b) {
365  // make sure they are both in the same environment
366  auto& environment = a.getTypeEnvironment();
367  assert(environment.isType(a) && environment.isType(b));
368 
369  // first check - a type is a sub-type of itself
370  if (a == b) {
371  return true;
372  }
373 
374  // check for predefined types
375  if (b == environment.getNumberType()) {
376  return isNumberType(a);
377  }
378  if (b == environment.getSymbolType()) {
379  return isSymbolType(a);
380  }
381 
382  // check primitive type chains
383  if (isA<PrimitiveType>(a)) {
384  if (isSubtypeOf(as<PrimitiveType>(a).getBaseType(), b)) {
385  return true;
386  }
387  }
388 
389  // next - if b is a union type
390  if (isUnion(b)) {
391  return isSubType(a, as<UnionType>(b));
392  }
393 
394  // done
395  return false;
396 }

409 TypeSet getLeastCommonSupertypes(const Type& a, const Type& b) {
410  // make sure they are in the same type environment
411  assert(a.getTypeEnvironment().isType(a) && a.getTypeEnvironment().isType(b));
412 
413  // if they are equal it is easy
414  if (a == b) {
415  return TypeSet(a);
416  }
417 
418  // equally simple - check whether one is a sub-type of the other
419  if (isSubtypeOf(a, b)) {
420  return TypeSet(b);
421  }
422  if (isSubtypeOf(b, a)) {
423  return TypeSet(a);
424  }
425 
426  // harder: no obvious relation => hard way
427  TypeSet superTypes;
428  TypeSet all = a.getTypeEnvironment().getAllTypes();
429  for (const Type& cur : all) {
430  if (isSubtypeOf(a, cur) && isSubtypeOf(b, cur)) {
431  superTypes.insert(cur);
432  }
433  }
434 
435  // filter out non-least super types
436  TypeSet res;
437  for (const Type& cur : superTypes) {
438  bool least = !any_of(superTypes, [&](const Type& t) { return t != cur && isSubtypeOf(t, cur); });
439  if (least) {
440  res.insert(cur);
441  }
442  }
443 
444  return res;
445 }

504  // make sure they are in the same type environment
505  assert(a.getTypeEnvironment().isType(a) && a.getTypeEnvironment().isType(b));
506
