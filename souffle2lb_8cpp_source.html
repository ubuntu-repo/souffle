114  /**
115  * The entry point for the conversion of a program, converting the basic top-level structure.
116  */
117  void visitProgram(const AstProgram& program, std::ostream& out) override {
118  // declarations
119  for (const auto& rel : program.getRelations()) {
120  // process the relation declaration
121  visit(*rel, out);
122  }
123  out << "\n";
124 
125  // rules
126  for (const auto& rel : program.getRelations()) {
127  for (const auto& clause : rel->getClauses()) {
128  visit(*clause, out);
129  }
130  }
131  out << "\n";
132  }

189  /**
190  * Converting a clause.
191  */
192  void visitClause(const AstClause& clause, std::ostream& out) override {
193  visit(*clause.getHead(), out);
194 
195  // if it is a fact, that's it
196  if (clause.isFact()) {
197  // there must not be any additons
198  if (!extra_literals.empty()) {
199  throw UnsupportedConstructException("Unsupported fact: " + toString(clause));
200  }
201  out << ".\n";
202  return;
203  }
204 
205  // convert the body
206  out << " <- ";
207  out << join(
208  clause.getBodyLiterals(), ",", [&](std::ostream& os, AstLiteral* cur) { visit(*cur, os); });
209 
210  // add extra_literals
211  for (const auto& cur : extra_literals) {
212  out << "," << cur;
213  }
214  extra_literals.clear();
215 
216  // done
217  out << ".\n";
218  }

257  void visitFunctor(const AstFunctor& fun, std::ostream& out) override {
258  // only intrinsic binary operators supported
259  const auto* inf = dynamic_cast<const AstIntrinsicFunctor*>(&fun);
260  if (inf != nullptr && inf->getArity() == 2) {
261  std::string sym = getSymbolForFunctorOp(inf->getFunction());
262  out << "(";
263  visit(*inf->getArg(0), out);
264  out << sym;
265  visit(*inf->getArg(1), out);
266  out << ")";
267  } else {
268  throw UnsupportedConstructException("Unsupported function: " + toString(fun));
269  }
270  }

330 
331  // ------ command line arguments -------------
332 
333  /* for the help option, if given simply print the help text then exit */
334  if (!Global::config().has("") || Global::config().has("help")) {
335  std::cerr << Global::config().help();
336  return 0;
337  }
338 
339  /* check that Datalog program exists */
340  if (!existFile(Global::config().get(""))) {
341  throw std::runtime_error("cannot open file " + std::string(Global::config().get("")));
342  }
343 
344  /* collect all input directories for the c pre-processor */
345  if (Global::config().has("include-dir")) {
346  std::string currentInclude = "";
347  std::string allIncludes = "";
348  for (const char& ch : Global::config().get("include-dir")) {
349  if (ch == ' ') {
350  if (!existDir(currentInclude)) {
351  throw std::runtime_error("include directory " + currentInclude + " does not exists");
352  } else {
353  allIncludes += " -I";
354  allIncludes += currentInclude;
355  currentInclude = "";
356  }
357  } else {
358  currentInclude += ch;
359  }
360  }
361  allIncludes += " -I" + currentInclude;
362  Global::config().set("include-dir", allIncludes);
363  }
364 
365  // ------ start souffle -------------
366 
367  /* Create the pipe to establish a communication between cpp and souffle */
368  std::string cmd = ::which("mcpp");
369 
370  if (!isExecutable(cmd)) {
371  throw std::runtime_error("failed to locate mcpp pre-processor");
372  }
373 
374  cmd += " -W0 " + Global::config().get("include-dir") + " " + Global::config().get("");
375  FILE* in = popen(cmd.c_str(), "r");
376 
377  /* Time taking for parsing */
378  auto parser_start = std::chrono::high_resolution_clock::now();
379 
380  // ------- parse program -------------
381 
382  // parse file
383  SymbolTable symTab;
384  ErrorReport errReport(Global::config().has("no-warn"));
385  DebugReport debugReport;
386  std::unique_ptr<AstTranslationUnit> astTranslationUnit =
387  ParserDriver::parseTranslationUnit("<stdin>", in, symTab, errReport, debugReport);
388 
389  // close input pipe
390  int preprocessor_status = pclose(in);
391  if (preprocessor_status == -1) {
392  perror(nullptr);
393  throw std::runtime_error("failed to close pre-processor pipe");
394  }
395 
396  /* Report run-time of the parser if verbose flag is set */
397  if (Global::config().has("verbose")) {
398  auto parser_end = std::chrono::high_resolution_clock::now();

400  << "sec\n";
401  }
402 
403  // ------- check for parse errors -------------
404  if (astTranslationUnit->getErrorReport().getNumErrors() != 0) {
405  std::cerr << astTranslationUnit->getErrorReport();

408  << std::endl;
409  exit(1);
410  }
411 
412  // ------- rewriting / optimizations -------------
413 
414  /* set up additional global options based on pragma declaratives */
415  (std::make_unique<AstPragmaChecker>())->apply(*astTranslationUnit);
416 
417  /* construct the transformation pipeline */
418 
419  // Magic-Set pipeline
420  auto magicPipeline = std::make_unique<ConditionalTransformer>(Global::config().has("magic-transform"),
421  std::make_unique<PipelineTransformer>(std::make_unique<NormaliseConstraintsTransformer>(),
422  std::make_unique<MagicSetTransformer>(),

431  std::make_unique<AstSemanticChecker>(),
432  std::make_unique<RemoveBooleanConstraintsTransformer>(),
433  std::make_unique<InlineRelationsTransformer>(),
434  std::make_unique<ReduceExistentialsTransformer>(),
435  std::make_unique<PartitionBodyLiteralsTransformer>(),
436  std::make_unique<RemoveRelationCopiesTransformer>(),
437  std::make_unique<MaterializeAggregationQueriesTransformer>(),
438  std::make_unique<RemoveEmptyRelationsTransformer>(),
439  std::make_unique<RemoveRedundantRelationsTransformer>(), std::move(magicPipeline),
440  std::make_unique<AstExecutionPlanChecker>());
441 
442  // Set up the debug report if necessary
443  if (!Global::config().get("debug-report").empty()) {
444  auto parser_end = std::chrono::high_resolution_clock::now();
445  std::string runtimeStr =
446  "(" + std::to_string(std::chrono::duration<double>(parser_end - parser_start).count()) +
447  "s)";

449 
450  pipeline->setDebugReport();

454  pipeline->setVerbosity(Global::config().has("verbose"));
455

458
