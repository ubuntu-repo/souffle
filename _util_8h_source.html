244 /**
245  * A utility class enabling representation of ranges by pairing
246  * two iterator instances marking lower and upper boundaries.
247  */
248 template <typename Iter>
249 struct range {
250  // the lower and upper boundary
251  Iter a, b;
252 
253  // a constructor accepting a lower and upper boundary
254  range(Iter a, Iter b) : a(std::move(a)), b(std::move(b)) {}
255 
256  // default copy / move and assignment support
257  range(const range&) = default;
258  range(range&&) = default;
259  range& operator=(const range&) = default;
260 
261  // get the lower boundary (for for-all loop)
262  Iter& begin() {
263  return a;
264  }
265  const Iter& begin() const {
266  return a;
267  }
268 
269  // get the upper boundary (for for-all loop)
270  Iter& end() {
271  return b;
272  }
273  const Iter& end() const {
274  return b;
275  }
276 
277  // emptiness check
278  bool empty() const {
279  return a == b;
280  }
281 
282  // splits up this range into the given number of partitions
283  std::vector<range> partition(int np = 100) {
284  // obtain the size
285  int n = 0;
286  for (auto i = a; i != b; ++i) n++;
287 
288  // split it up
289  auto s = n / np;
290  auto r = n % np;
291  std::vector<range> res;
292  res.reserve(np);
293  auto cur = a;
294  auto last = cur;
295  int i = 0;
296  int p = 0;
297  while (cur != b) {
298  ++cur;
299  i++;
300  if (i >= (s + (p < r ? 1 : 0))) {
301  res.push_back({last, cur});
302  last = cur;
303  p++;
304  i = 0;
305  }
306  }
307  if (cur != last) {
308  res.push_back({last, cur});
309  }
310  return res;
311  }
312 };

535 /**
536  * A auxiliary class to be returned by the join function aggregating the information
537  * required to print a list of elements as well as the implementation of the printing
538  * itself.
539  */
540 template <typename Iter, typename Printer>
541 class joined_sequence {
542  /** The begin of the range to be printed */
543  Iter begin;
544 
545  /** The end of the range to be printed */
546  Iter end;
547 
548  /** The seperator to be utilized between elements */
549  std::string sep;
550 
551  /** A functor printing an element */
552  Printer p;
553 
554 public:
555  /** A constructor setting up all fields of this class */
556  joined_sequence(const Iter& a, const Iter& b, std::string sep, Printer p)
557  : begin(a), end(b), sep(std::move(sep)), p(std::move(p)) {}
558 
559  /** The actual print method */
560  friend std::ostream& operator<<(std::ostream& out, const joined_sequence& s) {
561  auto cur = s.begin;
562  if (cur == s.end) {
563  return out;
564  }
565 
566  s.p(out, *cur);
567  ++cur;
568  for (; cur != s.end; ++cur) {
569  out << s.sep;
570  s.p(out, *cur);
571  }
572  return out;
573  }
574 };

1221 /* begin reference implementation
1222  * http://www.open-std.org/jtc1/sc22/wg21/docs/papers/2007/n2406.html#shared_mutex */
1223 // This simply exists as we do not compile using C++17. If we change standard >=C++17,
1224 // souffle::shared_mutex should be exchanged with std::shared_mutex
1225 // Slight cosmetic adjustments have been made
1226 class shared_mutex {
1227  std::mutex mut_;
1228  std::condition_variable gate1_;
1229  std::condition_variable gate2_;
1230  unsigned state_ = 0;
1231 
1232  static const unsigned write_entered_ = 1U << (sizeof(unsigned) * CHAR_BIT - 1);
1233  static const unsigned n_readers_ = ~write_entered_;
1234 
1235 public:
1236  shared_mutex() = default;
1237 
1238  // Exclusive ownership
1239  void lock() {
1240  std::unique_lock<std::mutex> lk(mut_);
1241  while (state_ & write_entered_) gate1_.wait(lk);
1242  state_ |= write_entered_;
1243  while (state_ & n_readers_) gate2_.wait(lk);
1244  }
1245 
1246  bool try_lock() {
1247  std::unique_lock<std::mutex> lk(mut_, std::try_to_lock);
1248  if (lk.owns_lock() && state_ == 0) {
1249  state_ = write_entered_;
1250  return true;
1251  }
1252  return false;
1253  }
1254 
1255  void unlock() {
1256  {
1257  std::lock_guard<std::mutex> _(mut_);
1258  state_ = 0;
1259  }
1260  gate1_.notify_all();
1261  }
1262 
1263  // Shared ownership
1264  void lock_shared() {
1265  std::unique_lock<std::mutex> lk(mut_);
1266  while ((state_ & write_entered_) || (state_ & n_readers_) == n_readers_) gate1_.wait(lk);
1267  unsigned num_readers = (state_ & n_readers_) + 1;
1268  state_ &= ~n_readers_;
1269  state_ |= num_readers;
1270  }
1271 
1272  bool try_lock_shared() {
1273  std::unique_lock<std::mutex> lk(mut_, std::try_to_lock);
1274  unsigned num_readers = state_ & n_readers_;
1275 
1276  if (lk.owns_lock() && !(state_ & write_entered_) && num_readers != n_readers_) {
1277  ++num_readers;
1278  state_ &= ~n_readers_;
1279  state_ |= num_readers;
1280  return true;
1281  }
1282  return false;
1283  }
1284 
1285  void unlock_shared() {
1286  std::lock_guard<std::mutex> _(mut_);
1287  unsigned num_readers = (state_ & n_readers_) - 1;
1288  state_ &= ~n_readers_;
1289  state_ |= num_readers;
1290 
1291  if (state_ & write_entered_) {
1292  if (num_readers == 0) gate2_.notify_one();
1293  } else {
1294  if (num_readers == n_readers_ - 1) gate1_.notify_one();
1295  }
1296  }
1297 };

1323 /**
1324  * An Least-Recently-Used cache for arbitrary element types. Elements can be signaled
1325  * to be accessed and iterated through in their LRU order.
1326  */
1327 template <typename T, unsigned size = 1>
1328 class LRUCache {
1329  // the list of pointers maintained
1330  std::array<T, size> entries;
1331 
1332  // pointer to predecessor / successor in the entries list
1333  std::array<std::size_t, size> priv; // < predecessor of element i
1334  std::array<std::size_t, size> post; // < successor of element i
1335 
1336  std::size_t first{0}; // < index of the first element
1337  std::size_t last{size - 1}; // < index of the last element
1338 
1339 public:
1340  // creates a new, empty cache
1341  LRUCache(const T& val = T()) {
1342  for (unsigned i = 0; i < size; i++) {
1343  entries[i] = val;
1344  priv[i] = i - 1;
1345  post[i] = i + 1;
1346  }
1347  priv[first] = last;
1348  post[last] = first;
1349  }
1350 
1351  // clears the content of this cache
1352  void clear(const T& val = T()) {
1353  for (auto& cur : entries) {
1354  cur = val;
1355  }
1356  }
1357 
1358  // registers an access to the given element
1359  void access(const T& val) {
1360  // test whether it is contained
1361  for (std::size_t i = 0; i < size; i++) {
1362  if (entries[i] != val) continue;
1363 
1364  // -- move this one to the front --
1365 
1366  // if it is the first, nothing to handle
1367  if (i == first) return;
1368 
1369  // if this is the last, just first and last need to change
1370  if (i == last) {
1371  auto tmp = last;
1372  last = priv[last];
1373  first = tmp;
1374  return;
1375  }
1376 
1377  // otherwise we need to update the linked list
1378 
1379  // remove from current position
1380  post[priv[i]] = post[i];
1381  priv[post[i]] = priv[i];
1382 
1383  // insert in first position
1384  post[i] = first;
1385  priv[i] = last;
1386  priv[first] = i;
1387  post[last] = i;
1388 
1389  // update first pointer
1390  first = i;
1391  return;
1392  }
1393  // not present => drop last, make it first
1394  entries[last] = val;
1395  auto tmp = last;
1396  last = priv[last];
1397  first = tmp;
1398  }
1399 
1400  /**
1401  * Iterates over the elements within this cache in LRU order.
1402  * The operator is applied on each element. If the operation
1403  * returns false, iteration is continued. If the operator return
1404  * true, iteration is stopped -- similar to the any operator.
1405  *
1406  * @param op the operator to be applied on every element
1407  * @return true if op returned true for any entry, false otherwise
1408  */
1409  template <typename Op>
1410  bool forEachInOrder(const Op& op) const {
1411  std::size_t i = first;
1412  while (i != last) {
1413  if (op(entries[i])) return true;
1414  i = post[i];
1415  }
1416  return op(entries[i]);
1417  }
1418 
1419  // equivalent to forEachInOrder
1420  template <typename Op>
1421  bool any(const Op& op) const {
1422  return forEachInOrder(op);
1423  }
1424 };

1428  bool first = true;

1430  if (!first) out << ",";
1431  first = false;
1432  out << val;
1433  return false;
1434  });
1435  return out;
