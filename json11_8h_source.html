1 /* json11
2  *
3  * json11 is a tiny JSON library for C++11, providing JSON parsing and serialization.
4  *
5  * The core object provided by the library is json11::Json. A Json object represents any JSON
6  * value: null, bool, number (int or double), string (std::string), array (std::vector), or
7  * object (std::map).
8  *
9  * Json objects act like values: they can be assigned, copied, moved, compared for equality or
10  * order, etc. There are also helper methods Json::dump, to serialize a Json to a string, and
11  * Json::parse (static) to parse a std::string as a Json object.
12  *
13  * Internally, the various types of Json object are represented by the JsonValue class
14  * hierarchy.
15  *
16  * A note on numbers - JSON specifies the syntax of number formatting but not its semantics,
17  * so some JSON implementations distinguish between integers and floating-point numbers, while
18  * some don't. In json11, we choose the latter. Because some JSON implementations (namely
19  * Javascript itself) treat all numbers as the same type, distinguishing the two leads
20  * to JSON that will be *silently* changed by a round-trip through those implementations.
21  * Dangerous! To avoid that risk, json11 stores all numbers as double internally, but also
22  * provides integer helpers.
23  *
24  * Fortunately, double-precision IEEE754 ('double') can precisely store any integer in the
25  * range +/-2^53, which includes every 'int' on most systems. (Timestamps often use int64
26  * or long long to avoid the Y2038K problem; a double storing microseconds since some epoch
27  * will be exact for +/- 275 years.)
28  */

322 static void dump(const string& value, string& out) {
323  out += '"';
324  for (size_t i = 0; i < value.length(); i++) {
325  const char ch = value[i];
326  if (ch == '\\') {
327  out += "\\\\";
328  } else if (ch == '"') {
329  out += "\\\"";
330  } else if (ch == '\b') {
331  out += "\\b";
332  } else if (ch == '\f') {
333  out += "\\f";
334  } else if (ch == '\n') {
335  out += "\\n";
336  } else if (ch == '\r') {
337  out += "\\r";
338  } else if (ch == '\t') {
339  out += "\\t";
340  } else if (static_cast<uint8_t>(ch) <= 0x1f) {
341  char buf[8];
342  snprintf(buf, sizeof buf, "\\u%04x", ch);
343  out += buf;
344  } else if (static_cast<uint8_t>(ch) == 0xe2 && static_cast<uint8_t>(value[i + 1]) == 0x80 &&
345  static_cast<uint8_t>(value[i + 2]) == 0xa8) {
346  out += "\\u2028";
347  i += 2;
348  } else if (static_cast<uint8_t>(ch) == 0xe2 && static_cast<uint8_t>(value[i + 1]) == 0x80 &&
349  static_cast<uint8_t>(value[i + 2]) == 0xa9) {
350  out += "\\u2029";
351  i += 2;
352  } else {
353  out += ch;
354  }
355  }
356  out += '"';
357 }

719  i++;

760  /* encode_utf8(pt, out)
761  *
762  * Encode pt as UTF-8 and add it to out.
763  */
764  void encode_utf8(long pt, string& out) {
765  if (pt < 0) return;
766 
767  if (pt < 0x80) {
768  out += static_cast<char>(pt);
769  } else if (pt < 0x800) {
770  out += static_cast<char>((pt >> 6) | 0xC0);
771  out += static_cast<char>((pt & 0x3F) | 0x80);
772  } else if (pt < 0x10000) {
773  out += static_cast<char>((pt >> 12) | 0xE0);
774  out += static_cast<char>(((pt >> 6) & 0x3F) | 0x80);
775  out += static_cast<char>((pt & 0x3F) | 0x80);
776  } else {
777  out += static_cast<char>((pt >> 18) | 0xF0);
778  out += static_cast<char>(((pt >> 12) & 0x3F) | 0x80);
779  out += static_cast<char>(((pt >> 6) & 0x3F) | 0x80);
780  out += static_cast<char>((pt & 0x3F) | 0x80);
781  }

784  /* parse_string()
785  *
786  * Parse a string, starting at the current position.
787  */
788  string parse_string() {
789  string out;
790  long last_escaped_codepoint = -1;
791  while (true) {
792  if (i == str.size()) return fail("unexpected end of input in string", "");
793 
794  char ch = str[i++];
795 
796  if (ch == '"') {
797  encode_utf8(last_escaped_codepoint, out);
798  return out;
799  }
800 
801  if (in_range(ch, 0, 0x1f)) return fail("unescaped " + esc(ch) + " in string", "");
802 
803  // The usual case: non-escaped characters
804  if (ch != '\\') {
805  encode_utf8(last_escaped_codepoint, out);
806  last_escaped_codepoint = -1;
807  out += ch;
808  continue;
809  }
810 
811  // Handle escapes
812  if (i == str.size()) return fail("unexpected end of input in string", "");
813 
814  ch = str[i++];
815 
816  if (ch == 'u') {
817  // Extract 4-byte escape sequence
818  string esc = str.substr(i, 4);
819  // Explicitly check length of the substring. The following loop
820  // relies on std::string returning the terminating NUL when
821  // accessing str[length]. Checking here reduces brittleness.
822  if (esc.length() < 4) {
823  return fail("bad \\u escape: " + esc, "");
824  }
825  for (size_t j = 0; j < 4; j++) {
826  if (!in_range(esc[j], 'a', 'f') && !in_range(esc[j], 'A', 'F') &&
827  !in_range(esc[j], '0', '9'))
828  return fail("bad \\u escape: " + esc, "");
829  }
830 
831  long codepoint = strtol(esc.data(), nullptr, 16);
832 
833  // JSON specifies that characters outside the BMP shall be encoded as a pair
834  // of 4-hex-digit \u escapes encoding their surrogate pair components. Check
835  // whether we're in the middle of such a beast: the previous codepoint was an
836  // escaped lead (high) surrogate, and this is a trail (low) surrogate.
837  if (in_range(last_escaped_codepoint, 0xD800, 0xDBFF) && in_range(codepoint, 0xDC00, 0xDFFF)) {
838  // Reassemble the two surrogate pairs into one astral-plane character, per
839  // the UTF-16 algorithm.
840  encode_utf8((((last_escaped_codepoint - 0xD800) << 10) | (codepoint - 0xDC00)) + 0x10000,
841  out);
842  last_escaped_codepoint = -1;
843  } else {
844  encode_utf8(last_escaped_codepoint, out);
845  last_escaped_codepoint = codepoint;
846  }
847 
848  i += 4;
849  continue;
850  }
851 
852  encode_utf8(last_escaped_codepoint, out);
853  last_escaped_codepoint = -1;
854 
855  if (ch == 'b') {
856  out += '\b';
857  } else if (ch == 'f') {
858  out += '\f';
859  } else if (ch == 'n') {
860  out += '\n';
861  } else if (ch == 'r') {
862  out += '\r';
863  } else if (ch == 't') {
864  out += '\t';
865  } else if (ch == '"' || ch == '\\' || ch == '/') {
866  out += ch;
867  } else {
868  return fail("invalid escape character " + esc(ch), "");
869  }
870  }
871  }

873  /* parse_number()
874  *
875  * Parse a double.
876  */
877  Json parse_number() {
878  size_t start_pos = i;
879 
880  if (str[i] == '-') i++;
881 
882  // Integer part
883  if (str[i] == '0') {
884  i++;
885  if (in_range(str[i], '0', '9')) return fail("leading 0s not permitted in numbers");
886  } else if (in_range(str[i], '1', '9')) {
887  i++;
888  while (in_range(str[i], '0', '9')) i++;
889  } else {
890  return fail("invalid " + esc(str[i]) + " in number");
891  }
892 
893  if (str[i] != '.' && str[i] != 'e' && str[i] != 'E' &&
894  (i - start_pos) <= static_cast<size_t>(std::numeric_limits<int>::digits10)) {
895  return std::atoll(str.c_str() + start_pos);
896  }
897 
898  // Decimal part
899  if (str[i] == '.') {
900  i++;
901  if (!in_range(str[i], '0', '9')) return fail("at least one digit required in fractional part");
902 
903  while (in_range(str[i], '0', '9')) i++;
904  }
905 
906  // Exponent part
907  if (str[i] == 'e' || str[i] == 'E') {
908  i++;
909 
910  if (str[i] == '+' || str[i] == '-') i++;
911 
912  if (!in_range(str[i], '0', '9')) return fail("at least one digit required in exponent");
913 
914  while (in_range(str[i], '0', '9')) i++;
915  }
916 
917  return std::strtod(str.c_str() + start_pos, nullptr);
918  }

936  /* parse_json()
937  *
938  * Parse a JSON object.
939  */
940  Json parse_json(int depth) {
941  if (depth > max_depth) {
942  return fail("exceeded maximum nesting depth");
943  }
944 
945  char ch = get_next_token();
946  if (failed) return Json();
947 
948  if (ch == '-' || (ch >= '0' && ch <= '9')) {
949  i--;
950  return parse_number();
951  }
952 
953  if (ch == 't') return expect("true", true);
954 
955  if (ch == 'f') return expect("false", false);
956 
957  if (ch == 'n') return expect("null", Json());
958 
959  if (ch == '"') return parse_string();
960 
961  if (ch == '{') {
962  map<string, Json> data;
963  ch = get_next_token();
964  if (ch == '}') return data;
965 
966  while (true) {
967  if (ch != '"') return fail("expected '\"' in object, got " + esc(ch));
968 
969  string key = parse_string();
970  if (failed) return Json();
971 
972  ch = get_next_token();
973  if (ch != ':') return fail("expected ':' in object, got " + esc(ch));
974 
975  data[std::move(key)] = parse_json(depth + 1);
976  if (failed) return Json();
977 
978  ch = get_next_token();
979  if (ch == '}') break;
980  if (ch != ',') return fail("expected ',' in object, got " + esc(ch));
981 
982  ch = get_next_token();
983  }
984  return data;
985  }
986 
987  if (ch == '[') {
988  vector<Json> data;
989  ch = get_next_token();
990  if (ch == ']') return data;
991 
992  while (true) {
993  i--;
994  data.push_back(parse_json(depth + 1));
995  if (failed) return Json();
996 
997  ch = get_next_token();
998  if (ch == ']') break;
999  if (ch != ',') return fail("expected ',' in list, got " + esc(ch));
1000 
1001  ch = get_next_token();
1002  (void)ch;
1003  }
1004  return data;
1005  }
1006 
1007  return fail("expected value, got " + esc(ch));
1008  }
