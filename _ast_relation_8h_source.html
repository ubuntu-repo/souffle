117  /** Set qualifier associated with this relation */
118  void setQualifier(int q) {
119  qualifier = q;
120  if (q & EQREL_RELATION) {
121  representation = RelationRepresentation::EQREL;
122  } else if (q & BRIE_RELATION) {
123  representation = RelationRepresentation::BRIE;

128  if (q & INPUT_RELATION) {
129  loads.emplace_back(new AstLoad());
130  loads.back()->setName(getName());

132  }
133  if (q & OUTPUT_RELATION) {
134  stores.emplace_back(new AstStore());
135  stores.back()->setName(getName());
136  stores.back()->setSrcLoc(getSrcLoc());
137  }
138  if (q & PRINTSIZE_RELATION) {

140  stores.back()->setName(getName());
141  stores.back()->setSrcLoc(getSrcLoc());
142  }
