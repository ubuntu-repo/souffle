75 bool RemoveRelationCopiesTransformer::removeRelationCopies(AstTranslationUnit& translationUnit) {
76  using alias_map = std::map<AstRelationIdentifier, AstRelationIdentifier>;
77 
78  // tests whether something is a variable
79  auto isVar = [&](const AstArgument& arg) { return dynamic_cast<const AstVariable*>(&arg); };
80 
81  // tests whether something is a record
82  auto isRec = [&](const AstArgument& arg) { return dynamic_cast<const AstRecordInit*>(&arg); };
83 
84  // collect aliases
85  alias_map isDirectAliasOf;
86 
87  IOType* ioType = translationUnit.getAnalysis<IOType>();
88 
89  AstProgram& program = *translationUnit.getProgram();
90 
91  // search for relations only defined by a single rule ..
92  for (AstRelation* rel : program.getRelations()) {
93  if (!ioType->isIO(rel) && rel->getClauses().size() == 1u) {
94  // .. of shape r(x,y,..) :- s(x,y,..)
95  AstClause* cl = rel->getClause(0);
96  if (!cl->isFact() && cl->getBodySize() == 1u && cl->getAtoms().size() == 1u) {
97  AstAtom* atom = cl->getAtoms()[0];
98  if (equal_targets(cl->getHead()->getArguments(), atom->getArguments())) {
99  // we have a match but have to check that all arguments are either
100  // variables or records containing variables
101  bool onlyVars = true;
102  auto args = cl->getHead()->getArguments();
103  while (!args.empty()) {
104  const auto& cur = args.back();
105  args.pop_back();
106  if (!isVar(*cur)) {
107  if (isRec(*cur)) {
108  // records are decomposed and their arguments are checked
109  const auto& rec_args = static_cast<const AstRecordInit&>(*cur).getArguments();
110  for (auto rec_arg : rec_args) {
111  args.push_back(rec_arg);
112  }
113  } else {
114  onlyVars = false;
115  break;
116  }
117  }
118  }
119  if (onlyVars) {
120  // all arguments are either variables or records containing variables
121  isDirectAliasOf[cl->getHead()->getName()] = atom->getName();
122  }
123  }
124  }
125  }
126  }
127 
128  // map each relation to its ultimate alias (could be transitive)
129  alias_map isAliasOf;
130 
131  // track any copy cycles; cyclic rules are effectively empty
132  std::set<AstRelationIdentifier> cycle_reps;
133 
134  for (std::pair<AstRelationIdentifier, AstRelationIdentifier> cur : isDirectAliasOf) {
135  // compute replacement
136 
137  std::set<AstRelationIdentifier> visited;
138  visited.insert(cur.first);
139  visited.insert(cur.second);
140 
141  auto pos = isDirectAliasOf.find(cur.second);
142  while (pos != isDirectAliasOf.end()) {
143  if (visited.count(pos->second)) {
144  cycle_reps.insert(cur.second);
145  break;
146  }
147  cur.second = pos->second;
148  pos = isDirectAliasOf.find(cur.second);
149  }
150  isAliasOf[cur.first] = cur.second;
151  }
152 
153  if (isAliasOf.empty()) {
154  return false;
155  }
156 
157  // replace usage of relations according to alias map
158  visitDepthFirst(program, [&](const AstAtom& atom) {
159  auto pos = isAliasOf.find(atom.getName());
160  if (pos != isAliasOf.end()) {
161  const_cast<AstAtom&>(atom).setName(pos->second);
162  }
163  });
164 
165  // break remaining cycles
166  for (const auto& rep : cycle_reps) {
167  auto rel = program.getRelation(rep);
168  rel->removeClause(rel->getClause(0));
169  }
170 
171  // remove unused relations
172  for (const auto& cur : isAliasOf) {
173  if (!cycle_reps.count(cur.first)) {
174  program.removeRelation(program.getRelation(cur.first)->getName());
175  }
176  }
177 
178  return true;
179 }

213 bool MaterializeAggregationQueriesTransformer::materializeAggregationQueries(
214  AstTranslationUnit& translationUnit) {
215  bool changed = false;
216 
217  AstProgram& program = *translationUnit.getProgram();
218  const TypeEnvironment& env = translationUnit.getAnalysis<TypeEnvironmentAnalysis>()->getTypeEnvironment();
219 
220  // if an aggregator has a body consisting of more than an atom => create new relation
221  int counter = 0;
222  visitDepthFirst(program, [&](const AstClause& clause) {
223  visitDepthFirstPostOrder(clause, [&](const AstAggregator& agg) {
224  // check whether a materialization is required
225  if (!needsMaterializedRelation(agg)) {
226  return;
227  }
228 
229  changed = true;
230

233  visitDepthFirst(agg, [&](const AstVariable& var) { vars.insert(var.getName()); });

272  head->addArgument(std::unique_ptr<AstArgument>(res->clone()));
273 
274  // return replacement
275  return std::unique_ptr<AstNode>(res);
276  }));
277  }
278  }
279 
280  // -- build relation --
281 
282  auto* rel = new AstRelation();
283  rel->setName(relName);
284  // add attributes
285  std::map<const AstArgument*, TypeSet> argTypes =
286  TypeAnalysis::analyseTypes(env, *aggClause, &program);
287  for (const auto& cur : head->getArguments()) {
288  rel->addAttribute(std::make_unique<AstAttribute>(
289  toString(*cur), (isNumberType(argTypes[cur])) ? "number" : "symbol"));
290  }
291 
292  rel->addClause(std::unique_ptr<AstClause>(aggClause));
293  program.appendRelation(std::unique_ptr<AstRelation>(rel));
294 
295  // -- update aggregate --
296  AstAtom* aggAtom = head->clone();
297 
298  // count the usage of variables in the clause
299  // outside of aggregates. Note that the visitor
300  // is exhaustive hence double counting occurs
301  // which needs to be deducted for variables inside
302  // the aggregators and variables in the expression
303  // of aggregate need to be added. Counter is zero
304  // if the variable is local to the aggregate
305  std::map<std::string, int> varCtr;
306  visitDepthFirst(clause, [&](const AstArgument& arg) {
307  if (const auto* a = dynamic_cast<const AstAggregator*>(&arg)) {
308  visitDepthFirst(arg, [&](const AstVariable& var) { varCtr[var.getName()]--; });
309  if (a->getTargetExpression() != nullptr) {
310  visitDepthFirst(*a->getTargetExpression(),
311  [&](const AstVariable& var) { varCtr[var.getName()]++; });
312  }
313  } else {
314  visitDepthFirst(arg, [&](const AstVariable& var) { varCtr[var.getName()]++; });
315  }
316  });
317  for (size_t i = 0; i < aggAtom->getArity(); i++) {
318  if (auto* var = dynamic_cast<AstVariable*>(aggAtom->getArgument(i))) {
319  // replace local variable by underscore if local
320  if (varCtr[var->getName()] == 0) {
321  aggAtom->setArgument(i, std::make_unique<AstUnnamedVariable>());
322  }
323  }
324  }
325  const_cast<AstAggregator&>(agg).clearBodyLiterals();
326  const_cast<AstAggregator&>(agg).addBodyLiteral(std::unique_ptr<AstLiteral>(aggAtom));
327  });
328  });
329 
330  return changed;
331 }

334  // everything with more than 1 body literal => materialize
335  if (agg.getBodyLiterals().size() > 1) {
336  return true;
337  }
338 
339  // inspect remaining atom more closely
340  const AstAtom* atom = dynamic_cast<const AstAtom*>(agg.getBodyLiterals()[0]);
341  if (!atom) {
342  // no atoms, so materialize
343  return true;
344  }
345 
346  // if the same variable occurs several times => materialize
347  bool duplicates = false;
348  std::set<std::string> vars;
349  visitDepthFirst(*atom,
350  [&](const AstVariable& var) { duplicates = duplicates | !vars.insert(var.getName()).second; });
351 
352  // if there are duplicates a materialization is required
353  if (duplicates) {
354  return true;
355  }
356 
357  // for all others the materialization can be skipped
358  return false;

390 bool RemoveEmptyRelationsTransformer::removeEmptyRelationUses(
391  AstTranslationUnit& translationUnit, AstRelation* emptyRelation) {
392  AstProgram& program = *translationUnit.getProgram();
393  bool changed = false;
394 
395  //
396  // (1) drop rules from the program that have empty relations in their bodies.
397  // (2) drop negations of empty relations
398  //
399  // get all clauses
400  std::vector<const AstClause*> clauses;
401  visitDepthFirst(program, [&](const AstClause& cur) { clauses.push_back(&cur); });
402 
403  // clean all clauses
404  for (const AstClause* cl : clauses) {
405  // check for an atom whose relation is the empty relation
406 
407  bool removed = false;
408  for (AstLiteral* lit : cl->getBodyLiterals()) {
409  if (auto* arg = dynamic_cast<AstAtom*>(lit)) {
410  if (getAtomRelation(arg, &program) == emptyRelation) {
411  program.removeClause(cl);
412  removed = true;
413  changed = true;
414  break;
415  }
416  }
417  }
418 
419  if (!removed) {
420  // check whether a negation with empty relations exists
421 
422  bool rewrite = false;
423  for (AstLiteral* lit : cl->getBodyLiterals()) {
424  if (auto* neg = dynamic_cast<AstNegation*>(lit)) {
425  if (getAtomRelation(neg->getAtom(), &program) == emptyRelation) {
426  rewrite = true;
427  break;
428  }
429  }
430  }
431 
432  if (rewrite) {
433  // clone clause without negation for empty relations
434 
435  auto res = std::unique_ptr<AstClause>(cl->cloneHead());
436 
437  for (AstLiteral* lit : cl->getBodyLiterals()) {
438  if (auto* neg = dynamic_cast<AstNegation*>(lit)) {
439  if (getAtomRelation(neg->getAtom(), &program) != emptyRelation) {
440  res->addToBody(std::unique_ptr<AstLiteral>(lit->clone()));
441  }
442  } else {
443  res->addToBody(std::unique_ptr<AstLiteral>(lit->clone()));
444  }
445  }
446 
447  program.removeClause(cl);
448  program.appendClause(std::move(res));
449  changed = true;
450  }
451  }
452  }
453 
454  return changed;
455 }

572 bool PartitionBodyLiteralsTransformer::transform(AstTranslationUnit& translationUnit) {
573  bool changed = false;
574  AstProgram& program = *translationUnit.getProgram();
575 
576  /* Process:
577  * Go through each clause and construct a variable dependency graph G.
578  * The nodes of G are the variables. A path between a and b exists iff
579  * a and b appear in a common body literal.
580  *
581  * Using the graph, we can extract the body literals that are not associated
582  * with the arguments in the head atom into new relations. Depending on
583  * variable dependencies among these body literals, the literals can
584  * be partitioned into multiple separate new propositional clauses.
585  *
586  * E.g. a(x) <- b(x), c(y), d(y), e(z), f(z). can be transformed into:
587  * - a(x) <- b(x), newrel0(), newrel1().
588  * - newrel0() <- c(y), d(y).
589  * - newrel1() <- e(z), f(z).
590  *
591  * Note that only one pass through the clauses is needed:
592  * - All arguments in the body literals of the transformed clause cannot be
593  * independent of the head arguments (by construction)
594  * - The new relations holding the disconnected body literals are propositional,
595  * hence having no head arguments, and so the transformation does not apply.
596  */
597 
598  // Store clauses to add and remove after analysing the program
599  std::vector<AstClause*> clausesToAdd;
600  std::vector<const AstClause*> clausesToRemove;
601 
602  // The transformation is local to each rule, so can visit each independently
603  visitDepthFirst(program, [&](const AstClause& clause) {
604  // Create the variable dependency graph G
605  Graph<std::string> variableGraph = Graph<std::string>();
606  std::set<std::string> ruleVariables;
607 
608  // Add in the nodes
609  // The nodes of G are the variables in the rule
610  visitDepthFirst(clause, [&](const AstVariable& var) {
611  variableGraph.insert(var.getName());
612  ruleVariables.insert(var.getName());
613  });
614 
615  // Add in the edges
616  // Since we are only looking at reachability in the final graph, it is
617  // enough to just add in an (undirected) edge from the first variable
618  // in the literal to each of the other variables.
619  std::vector<AstLiteral*> literalsToConsider = clause.getBodyLiterals();
620  literalsToConsider.push_back(clause.getHead());
621 
622  for (AstLiteral* clauseLiteral : literalsToConsider) {
623  std::set<std::string> literalVariables;
624 
625  // Store all variables in the literal
626  visitDepthFirst(
627  *clauseLiteral, [&](const AstVariable& var) { literalVariables.insert(var.getName()); });
628 
629  // No new edges if only one variable is present
630  if (literalVariables.size() > 1) {
631  std::string firstVariable = *literalVariables.begin();
632  literalVariables.erase(literalVariables.begin());
633 
634  // Create the undirected edge
635  for (const std::string& var : literalVariables) {
636  variableGraph.insert(firstVariable, var);
637  variableGraph.insert(var, firstVariable);
638  }
639  }
640  }
641 
642  // Find the connected components of G
643  std::set<std::string> seenNodes;
644 
645  // Find the connected component associated with the head
646  std::set<std::string> headComponent;
647  visitDepthFirst(
648  *clause.getHead(), [&](const AstVariable& var) { headComponent.insert(var.getName()); });
649 
650  if (!headComponent.empty()) {
651  variableGraph.visitDepthFirst(*headComponent.begin(), [&](const std::string& var) {
652  headComponent.insert(var);
653  seenNodes.insert(var);
654  });
655  }
656 
657  // Compute all other connected components in the graph G
658  std::set<std::set<std::string>> connectedComponents;
659 
660  for (std::string var : ruleVariables) {
661  if (seenNodes.find(var) != seenNodes.end()) {
662  // Node has already been added to a connected component
663  continue;
664  }
665 
666  // Construct the connected component
667  std::set<std::string> component;
668  variableGraph.visitDepthFirst(var, [&](const std::string& child) {
669  component.insert(child);
670  seenNodes.insert(child);
671  });
672  connectedComponents.insert(component);
673  }
674 
675  if (connectedComponents.empty()) {
676  // No separate connected components, so no point partitioning
677  return;
678  }
679 
680  // Need to extract some disconnected lits!
681  changed = true;
682  std::vector<AstAtom*> replacementAtoms;
683 
684  // Construct the new rules
685  for (const std::set<std::string>& component : connectedComponents) {
686  // Come up with a unique new name for the relation
687  static int disconnectedCount = 0;
688  std::stringstream nextName;
689  nextName << "+disconnected" << disconnectedCount;
690  AstRelationIdentifier newRelationName = nextName.str();
691  disconnectedCount++;
692 
693  // Create the extracted relation and clause for the component
694  // newrelX() <- disconnectedLiterals(x).
695  auto newRelation = std::make_unique<AstRelation>();
696  newRelation->setName(newRelationName);
697  program.appendRelation(std::move(newRelation));
698 
699  auto* disconnectedClause = new AstClause();
700  disconnectedClause->setSrcLoc(clause.getSrcLoc());
701  disconnectedClause->setHead(std::make_unique<AstAtom>(newRelationName));
702 
703  // Find the body literals for this connected component
704  std::vector<AstLiteral*> associatedLiterals;
705  for (AstLiteral* bodyLiteral : clause.getBodyLiterals()) {
706  bool associated = false;
707  visitDepthFirst(*bodyLiteral, [&](const AstVariable& var) {
708  if (component.find(var.getName()) != component.end()) {
709  associated = true;
710  }
711  });
712  if (associated) {
713  disconnectedClause->addToBody(std::unique_ptr<AstLiteral>(bodyLiteral->clone()));
714  }
715  }
716 
717  // Create the atom to replace all these literals
718  replacementAtoms.push_back(new AstAtom(newRelationName));
719 
720  // Add the clause to the program
721  clausesToAdd.push_back(disconnectedClause);
722  }
723 
724  // Create the replacement clause
725  // a(x) <- b(x), c(y), d(z). --> a(x) <- newrel0(), newrel1(), b(x).
726  auto* replacementClause = new AstClause();
727  replacementClause->setSrcLoc(clause.getSrcLoc());
728  replacementClause->setHead(std::unique_ptr<AstAtom>(clause.getHead()->clone()));
729 
730  // Add the new propositions to the clause first
731  for (AstAtom* newAtom : replacementAtoms) {
732  replacementClause->addToBody(std::unique_ptr<AstLiteral>(newAtom));
733  }
734 
735  // Add the remaining body literals to the clause
736  for (AstLiteral* bodyLiteral : clause.getBodyLiterals()) {
737  bool associated = false;
738  bool hasVariables = false;
739  visitDepthFirst(*bodyLiteral, [&](const AstVariable& var) {
740  hasVariables = true;
741  if (headComponent.find(var.getName()) != headComponent.end()) {
742  associated = true;
743  }
744  });
745  if (associated || !hasVariables) {
746  replacementClause->addToBody(std::unique_ptr<AstLiteral>(bodyLiteral->clone()));
747  }
748  }
749 
750  // Replace the old clause with the new one
751  clausesToRemove.push_back(&clause);
752  clausesToAdd.push_back(replacementClause);
753  });
754 
755  // Adjust the program
756  for (AstClause* newClause : clausesToAdd) {
757  program.appendClause(std::unique_ptr<AstClause>(newClause));
758  }
759 
760  for (const AstClause* oldClause : clausesToRemove) {
761  program.removeClause(oldClause);
762  }
763 
764  return changed;

767 bool ReduceExistentialsTransformer::transform(AstTranslationUnit& translationUnit) {
768  AstProgram& program = *translationUnit.getProgram();
769 
770  // Checks whether a given clause is recursive
771  auto isRecursiveClause = [&](const AstClause& clause) {
772  AstRelationIdentifier relationName = clause.getHead()->getName();
773  bool recursive = false;
774  visitDepthFirst(clause.getBodyLiterals(), [&](const AstAtom& atom) {
775  if (atom.getName() == relationName) {
776  recursive = true;
777  }
778  });
779  return recursive;
780  };
781 
782  // Checks whether an atom is of the form a(_,_,...,_)
783  auto isExistentialAtom = [&](const AstAtom& atom) {
784  for (AstArgument* arg : atom.getArguments()) {
785  if (!dynamic_cast<AstUnnamedVariable*>(arg)) {
786  return false;
787  }
788  }
789  return true;
790  };
791 
792  // Construct a dependency graph G where:
793  // - Each relation is a node
794  // - An edge (a,b) exists iff a uses b "non-existentially" in one of its *recursive* clauses
795  // This way, a relation can be transformed into an existential form
796  // if and only if all its predecessors can also be transformed.
797  Graph<AstRelationIdentifier> relationGraph = Graph<AstRelationIdentifier>();
798 
799  // Add in the nodes
800  for (AstRelation* relation : program.getRelations()) {
801  relationGraph.insert(relation->getName());
802  }
803 
804  // Keep track of all relations that cannot be transformed
805  std::set<AstRelationIdentifier> minimalIrreducibleRelations;
806 
807  IOType* ioType = translationUnit.getAnalysis<IOType>();
808 
809  for (AstRelation* relation : program.getRelations()) {
810  // No I/O relations can be transformed
811  if (ioType->isIO(relation)) {
812  minimalIrreducibleRelations.insert(relation->getName());
813  }
814  for (AstClause* clause : relation->getClauses()) {
815  bool recursive = isRecursiveClause(*clause);
816  visitDepthFirst(*clause, [&](const AstAtom& atom) {
817  if (atom.getName() == clause->getHead()->getName()) {
818  return;
819  }
820 
821  if (!isExistentialAtom(atom)) {
822  if (recursive) {
823  // Clause is recursive, so add an edge to the dependency graph
824  relationGraph.insert(clause->getHead()->getName(), atom.getName());
825  } else {
826  // Non-existential apperance in a non-recursive clause, so
827  // it's out of the picture
828  minimalIrreducibleRelations.insert(atom.getName());
829  }
830  }
831  });
832  }
833  }
834 
835  // TODO (see issue #564): Don't transform relations appearing in aggregators
836  // due to aggregator issues with unnamed variables.
837  visitDepthFirst(program, [&](const AstAggregator& aggr) {
838  visitDepthFirst(
839  aggr, [&](const AstAtom& atom) { minimalIrreducibleRelations.insert(atom.getName()); });
840  });
841 
842  // Run a DFS from each 'bad' source
843  // A node is reachable in a DFS from an irreducible node if and only if it is
844  // also an irreducible node
845  std::set<AstRelationIdentifier> irreducibleRelations;
846  for (AstRelationIdentifier relationName : minimalIrreducibleRelations) {
847  relationGraph.visitDepthFirst(relationName,
848  [&](const AstRelationIdentifier& subRel) { irreducibleRelations.insert(subRel); });
849  }
850 
851  // All other relations are necessarily existential
852  std::set<AstRelationIdentifier> existentialRelations;
853  for (AstRelation* relation : program.getRelations()) {
854  if (!relation->getClauses().empty() && relation->getArity() != 0 &&
855  irreducibleRelations.find(relation->getName()) == irreducibleRelations.end()) {
856  existentialRelations.insert(relation->getName());
857  }
858  }
859 
860  // Reduce the existential relations
861  for (AstRelationIdentifier relationName : existentialRelations) {
862  AstRelation* originalRelation = program.getRelation(relationName);
863 
864  std::stringstream newRelationName;
865  newRelationName << "+?exists_" << relationName;
866 
867  auto newRelation = std::make_unique<AstRelation>();
868  newRelation->setName(newRelationName.str());
869  newRelation->setSrcLoc(originalRelation->getSrcLoc());
870 
871  // EqRel relations require two arguments, so remove it from the qualifier
872  newRelation->setQualifier(originalRelation->getQualifier() & ~(EQREL_RELATION));
873 
874  // Keep all non-recursive clauses
875  for (AstClause* clause : originalRelation->getClauses()) {
876  if (!isRecursiveClause(*clause)) {
877  auto newClause = std::make_unique<AstClause>();
878 
879  newClause->setSrcLoc(clause->getSrcLoc());
880  if (const AstExecutionPlan* plan = clause->getExecutionPlan()) {
881  newClause->setExecutionPlan(std::unique_ptr<AstExecutionPlan>(plan->clone()));
882  }
883  newClause->setGenerated(clause->isGenerated());
884  newClause->setFixedExecutionPlan(clause->hasFixedExecutionPlan());
885  newClause->setHead(std::make_unique<AstAtom>(newRelationName.str()));
886  for (AstLiteral* lit : clause->getBodyLiterals()) {
887  newClause->addToBody(std::unique_ptr<AstLiteral>(lit->clone()));
888  }
889 
890  newRelation->addClause(std::move(newClause));
891  }
892  }
893 
894  program.appendRelation(std::move(newRelation));
895  }

897  // Mapper that renames the occurrences of marked relations with their existential
898  // counterparts
899  struct renameExistentials : public AstNodeMapper {
900  const std::set<AstRelationIdentifier>& relations;
901 
902  renameExistentials(std::set<AstRelationIdentifier>& relations) : relations(relations) {}
903 
904  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
905  if (auto* clause = dynamic_cast<AstClause*>(node.get())) {
906  if (relations.find(clause->getHead()->getName()) != relations.end()) {
907  // Clause is going to be removed, so don't rename it
908  return node;
909  }
910  } else if (auto* atom = dynamic_cast<AstAtom*>(node.get())) {
911  if (relations.find(atom->getName()) != relations.end()) {
912  // Relation is now existential, so rename it
913  std::stringstream newName;
914  newName << "+?exists_" << atom->getName();
915  return std::make_unique<AstAtom>(newName.str());
916  }
917  }
918  node->apply(*this);
919  return node;
920  }
921  };
922 
923  renameExistentials update(existentialRelations);
924  program.apply(update);
925 
926  bool changed = !existentialRelations.empty();
927  return changed;
928 }

935  // Node-mapper to replace a set of singletons with unnamed variables
936  struct replaceSingletons : public AstNodeMapper {
937  std::set<std::string>& singletons;
938 
939  replaceSingletons(std::set<std::string>& singletons) : singletons(singletons) {}
940 
941  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
942  if (auto* var = dynamic_cast<AstVariable*>(node.get())) {
943  if (singletons.find(var->getName()) != singletons.end()) {
944  return std::make_unique<AstUnnamedVariable>();

946  }
947  node->apply(*this);
948  return node;
949  }

960  // Variable seen before, so not a singleton variable
961  nonsingletons.insert(name);
962  } else {

964  }
965  });
966

973  visitDepthFirst(rec, [&](const AstVariable& var) { ignoredVars.insert(var.getName()); });
974  });
975 
976  // Don't unname singleton variables occuring in constraints.
977  std::set<std::string> constraintVars;
978  visitDepthFirst(*clause, [&](const AstConstraint& cons) {
979  visitDepthFirst(cons, [&](const AstVariable& var) { ignoredVars.insert(var.getName()); });
980  });

983  for (auto& var : vars) {
984  if ((nonsingletons.find(var) == nonsingletons.end()) &&
985  (ignoredVars.find(var) == ignoredVars.end())) {
986  changed = true;
987  singletons.insert(var);
988  }
989  }
990 
991  // Replace the singletons found with underscores
992  replaceSingletons update(singletons);
993  clause->apply(update);
994  }
995  }
996 
997  return changed;

1000 bool NormaliseConstraintsTransformer::transform(AstTranslationUnit& translationUnit) {
1001  bool changed = false;
1002 
1003  // set a prefix for variables bound by magic-set for identification later
1004  // prepended by + to avoid conflict with user-defined variables
1005  static constexpr const char* boundPrefix = "+abdul";
1006 
1007  AstProgram& program = *translationUnit.getProgram();
1008 
1009  /* Create a node mapper that recursively replaces all constants and underscores
1010  * with named variables.
1011  *
1012  * The mapper keeps track of constraints that should be added to the original
1013  * clause it is being applied on in a given constraint set.
1014  */
1015  struct constraintNormaliser : public AstNodeMapper {
1016  std::set<AstBinaryConstraint*>& constraints;
1017  mutable int changeCount;
1018 
1019  constraintNormaliser(std::set<AstBinaryConstraint*>& constraints, int changeCount)
1020  : constraints(constraints), changeCount(changeCount) {}
1021 
1022  bool hasChanged() const {
1023  return changeCount > 0;
1024  }
1025 
1026  int getChangeCount() const {
1027  return changeCount;
1028  }
1029 
1030  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
1031  if (auto* stringConstant = dynamic_cast<AstStringConstant*>(node.get())) {
1032  // string constant found
1033  changeCount++;
1034 
1035  // create new variable name (with appropriate suffix)
1036  std::string constantValue = stringConstant->getConstant();
1037  std::stringstream newVariableName;
1038  newVariableName << boundPrefix << changeCount << "_" << constantValue << "_s";
1039 
1040  // create new constraint (+abdulX = constant)
1041  auto newVariable = std::make_unique<AstVariable>(newVariableName.str());
1042  constraints.insert(new AstBinaryConstraint(BinaryConstraintOp::EQ,
1043  std::unique_ptr<AstArgument>(newVariable->clone()),
1044  std::unique_ptr<AstArgument>(stringConstant->clone())));
1045 
1046  // update constant to be the variable created
1047  return std::move(newVariable);
1048  } else if (auto* numberConstant = dynamic_cast<AstNumberConstant*>(node.get())) {
1049  // number constant found
1050  changeCount++;
1051 
1052  // create new variable name (with appropriate suffix)
1053  AstDomain constantValue = numberConstant->getIndex();
1054  std::stringstream newVariableName;
1055  newVariableName << boundPrefix << changeCount << "_" << constantValue << "_n";
1056 
1057  // create new constraint (+abdulX = constant)
1058  auto newVariable = std::make_unique<AstVariable>(newVariableName.str());
1059  constraints.insert(new AstBinaryConstraint(BinaryConstraintOp::EQ,
1060  std::unique_ptr<AstArgument>(newVariable->clone()),
1061  std::unique_ptr<AstArgument>(numberConstant->clone())));
1062 
1063  // update constant to be the variable created
1064  return std::move(newVariable);
1065  } else if (dynamic_cast<AstUnnamedVariable*>(node.get())) {
1066  // underscore found
1067  changeCount++;
1068 
1069  // create new variable name
1070  std::stringstream newVariableName;
1071  newVariableName << "+underscore" << changeCount;
1072 
1073  return std::make_unique<AstVariable>(newVariableName.str());
1074  }
1075 
1076  node->apply(*this);
1077  return node;
1078  }
1079  };
1080 
1081  int changeCount = 0; // number of constants and underscores seen so far
1082 
1083  // apply the change to all clauses in the program
1084  for (AstRelation* rel : program.getRelations()) {
1085  for (AstClause* clause : rel->getClauses()) {
1086  if (clause->isFact()) {
1087  continue; // don't normalise facts
1088  }
1089 
1090  std::set<AstBinaryConstraint*> constraints;
1091  constraintNormaliser update(constraints, changeCount);
1092  clause->apply(update);
1093 
1094  changeCount = update.getChangeCount();
1095  changed = changed || update.hasChanged();
1096 
1097  for (AstBinaryConstraint* constraint : constraints) {
1098  clause->addToBody(std::unique_ptr<AstBinaryConstraint>(constraint));
1099  }
1100  }
1101  }
1102 
1103  return changed;
1104 }
