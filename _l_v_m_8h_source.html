48  *
49  * RAM is transferred into an equivalent Bytecode Representation (LVMCode) by LVMGenerator.

72  /** Execute the subroutine */
73  void executeSubroutine(const std::string& name, const std::vector<RamDomain>& arguments,
74  std::vector<RamDomain>& returnValues, std::vector<bool>& returnErrors) {
75  InterpreterContext ctxt;
76  ctxt.setReturnValues(returnValues);
77  ctxt.setReturnErrors(returnErrors);
78  ctxt.setArguments(arguments);
79 
80  if (subroutines.find(name) != subroutines.cend()) {
81  execute(subroutines.at(name), ctxt);
82  } else {
83  // Parse and cache the program
84  LVMGenerator generator(
85  translationUnit.getSymbolTable(), translationUnit.getProgram()->getSubroutine(name));
86  subroutines.emplace(std::make_pair(name, generator.getCodeStream()));
87  execute(subroutines.at(name), ctxt);
88  }
89  }
