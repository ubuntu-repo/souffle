57 /**
58  * Subclass of Literal that represents the use of a relation
59  * either in the head or in the body of a Clause, e.g., parent(x,y).

90  arguments.push_back(std::move(arg));

94  AstArgument* getArgument(size_t idx) const {
95  return arguments[idx].get();

99  void setArgument(size_t idx, std::unique_ptr<AstArgument> newArg) {
100  arguments[idx].swap(newArg);

102

104  std::vector<AstArgument*> getArguments() const {
105  return toPtrVector(arguments);
106  }
107 
108  /** Return the number of arguments */
109  size_t argSize() const {
110  return arguments.size();
111  }
112 
113  /** Output to a given stream */
114  void print(std::ostream& os) const override {
115  os << getName() << "(";
116 
117  for (size_t i = 0; i < arguments.size(); ++i) {
118  if (i != 0) {
119  os << ",";
120  }
121  if (arguments[i] != nullptr) {
122  arguments[i]->print(os);
123  } else {
124  os << "_";
125  }
126  }
127  os << ")";
128  }
129 
130  /** Creates a clone of this AST sub-structure */
131  AstAtom* clone() const override {
132  auto res = new AstAtom(name);
133  res->setSrcLoc(getSrcLoc());
134  for (const auto& cur : arguments) {
135  res->arguments.emplace_back(cur->clone());
136  }
137  return res;
138  }
139 
140  /** Mutates this node */
141  void apply(const AstNodeMapper& map) override {
142  for (auto& arg : arguments) {
143  arg = map(std::move(arg));
144  }
145  }
146 
147  /** Obtains a list of all embedded child nodes */
148  std::vector<const AstNode*> getChildNodes() const override {
149  std::vector<const AstNode*> res;
150  for (auto& cur : arguments) {
151  res.push_back(cur.get());
152  }
153  return res;
154  }
155 
156 protected:
157  /** Name of the atom */
158  AstRelationIdentifier name;
159 
160  /** Arguments of the atom */
161  std::vector<std::unique_ptr<AstArgument>> arguments;
162 
163  /** Implements the node comparison for this node type */
164  bool equal(const AstNode& node) const override {
165  assert(nullptr != dynamic_cast<const AstAtom*>(&node));
166  const auto& other = static_cast<const AstAtom&>(node);
167  return name == other.name && equal_targets(arguments, other.arguments);
168  }
