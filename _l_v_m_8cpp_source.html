60 void LVM::executeMain() {
61  const RamStatement& main = *translationUnit.getProgram()->getMain();
62  if (mainProgram.get() == nullptr) {
63  LVMGenerator generator(translationUnit.getSymbolTable(), main);
64  mainProgram = generator.getCodeStream();
65  }
66  InterpreterContext ctxt;
67  SignalHandler::instance()->set();
68  if (Global::config().has("verbose")) {
69  SignalHandler::instance()->enableLogging();
70  }
71 
72  if (!Global::config().has("profile")) {
73  execute(mainProgram, ctxt);
74  } else {
75  ProfileEventSingleton::instance().setOutputFile(Global::config().get("profile"));
76  // Prepare the frequency table for threaded use
77  visitDepthFirst(main, [&](const RamSearch& node) {
78  if (!node.getProfileText().empty()) {
79  frequencies.emplace(node.getProfileText(), std::map<size_t, size_t>());
80  }
81  });
82  // Enable profiling for execution of main
83  ProfileEventSingleton::instance().startTimer();
84  ProfileEventSingleton::instance().makeTimeEvent("@time;starttime");
85  // Store configuration
86  for (const auto& cur : Global::config().data()) {
87  ProfileEventSingleton::instance().makeConfigRecord(cur.first, cur.second);
88  }
89  // Store count of relations
90  size_t relationCount = 0;
91  visitDepthFirst(main, [&](const RamCreate& create) {
92  if (create.getRelation().getName()[0] != '@') {
93  ++relationCount;
94  reads[create.getRelation().getName()] = 0;
95  }
96  });
97  ProfileEventSingleton::instance().makeConfigRecord("relationCount", std::to_string(relationCount));
98 
99  // Store count of rules
100  size_t ruleCount = 0;
101  visitDepthFirst(main, [&](const RamQuery& rule) { ++ruleCount; });
102  ProfileEventSingleton::instance().makeConfigRecord("ruleCount", std::to_string(ruleCount));
103 
104  execute(mainProgram, ctxt);
105  ProfileEventSingleton::instance().stopTimer();
106  for (auto const& cur : frequencies) {
107  for (auto const& iter : cur.second) {
108  ProfileEventSingleton::instance().makeQuantityEvent(cur.first, iter.second, iter.first);
109  }
110  }
111  for (auto const& cur : reads) {
112  ProfileEventSingleton::instance().makeQuantityEvent(
113  "@relation-reads;" + cur.first, cur.second, 0);
114  }
115  }
116  SignalHandler::instance()->reset();
117 }

119 void LVM::execute(std::unique_ptr<LVMCode>& codeStream, InterpreterContext& ctxt, size_t ip) {
120  std::stack<RamDomain> stack;
121  const auto& code = codeStream->getCode();
122  auto& symbolTable = codeStream->getSymbolTable();
123  while (true) {
124  switch (code[ip]) {
125  case LVM_Number:
126  stack.push(code[ip + 1]);
127  ip += 2;
128  break;
129  case LVM_ElementAccess:
130  stack.push(ctxt[code[ip + 1]][code[ip + 2]]);
131  ip += 3;
132  break;
133  case LVM_AutoIncrement:

136  ip += 1;

138  case LVM_OP_ORD:

146  ip += 1;
147  break;
148  }

150  RamDomain val = stack.top();
151  stack.pop();
152  stack.push(-val);
153  ip += 1;
154  break;
155  }
156  case LVM_OP_BNOT: {
157  RamDomain val = stack.top();
158  stack.pop();
159  stack.push(~val);
160  ip += 1;
161  break;
162  }
163  case LVM_OP_LNOT: {
164  RamDomain val = stack.top();
165  stack.pop();
166  stack.push(!val);
167  ip += 1;
168  break;

346  }

349  stack.pop();

367  stack.pop();

371  ip += 1;
372  break;

376  stack.pop();
377  RamDomain lhs = stack.top();
378  stack.pop();
379  stack.push(lhs <= rhs);
380  ip += 1;
381  break;
382  }

385  stack.pop();
386  RamDomain lhs = stack.top();
387  stack.pop();
388  stack.push(lhs > rhs);
389  ip += 1;
390  break;

394  stack.pop();
395  RamDomain lhs = stack.top();
396  stack.pop();
397  stack.push(lhs >= rhs);
398  ip += 1;
399  break;

403  stack.pop();
404  RamDomain lhs = stack.top();
405  stack.pop();
406

423  stack.pop();
424  RamDomain lhs = stack.top();
425  stack.pop();
426

428  const std::string& text = symbolTable.resolve(rhs);

432  result = !std::regex_match(text, std::regex(pattern));

434  std::cerr << "warning: wrong pattern provided for match(\"" << pattern << "\",\"" << text

436  }
437  stack.push(result);
438  ip += 1;
439  break;
440  }

443  stack.pop();
444  RamDomain lhs = stack.top();
445  stack.pop();
446  const std::string& pattern = symbolTable.resolve(lhs);
447  const std::string& text = symbolTable.resolve(rhs);

451  }

454  stack.pop();
455  RamDomain lhs = stack.top();
456  stack.pop();
