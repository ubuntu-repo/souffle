73 /**
74  * A constraint factory for a constraint ensuring that the value assigned to the
75  * given variable is (at least) {true}
76  */
77 BoolDisjunctConstraint isTrue(const BoolDisjunctVar& var) {
78  struct C : public Constraint<BoolDisjunctVar> {
79  BoolDisjunctVar var;
80  C(BoolDisjunctVar var) : var(std::move(var)) {}
81  bool update(Assignment<BoolDisjunctVar>& ass) const override {
82  auto res = !ass[var];
83  ass[var] = true;
84  return res;
85  }
86  void print(std::ostream& out) const override {
87  out << var << " is true";
88  }
89  };
90  return std::make_shared<C>(var);
91 }

93 /**
94  * A constraint factory for a constraint ensuring the constraint
95  *
96  * a ⇒ b
97  *
98  * Hence, whenever a is mapped to {true}, so is b.
99  */
100 BoolDisjunctConstraint imply(const BoolDisjunctVar& a, const BoolDisjunctVar& b) {
101  return sub(a, b, "⇒");
102 }

104 /**
105  * A constraint factory for a constraint ensuring the constraint
106  *
107  * vars[0] ∧ vars[1] ∧ ... ∧ vars[n] ⇒ res
108  *
109  * Hence, whenever all variables vars[i] are mapped to true, so is res.
110  */
111 BoolDisjunctConstraint imply(const std::vector<BoolDisjunctVar>& vars, const BoolDisjunctVar& res) {
112  struct C : public Constraint<BoolDisjunctVar> {
113  BoolDisjunctVar res;
114  std::vector<BoolDisjunctVar> vars;
115 
116  C(BoolDisjunctVar res, std::vector<BoolDisjunctVar> vars)
117  : res(std::move(res)), vars(std::move(vars)) {}
118 
119  bool update(Assignment<BoolDisjunctVar>& ass) const override {
120  bool r = ass[res];
121  if (r) {
122  return false;
123  }
124 
125  for (const auto& cur : vars) {
126  if (!ass[cur]) {
127  return false;
128  }
129  }
130 
131  ass[res] = true;
132  return true;
133  }
134 
135  void print(std::ostream& out) const override {
136  out << join(vars, " ∧ ") << " ⇒ " << res;
137  }
138  };
139 
140  return std::make_shared<C>(res, vars);
141 }

148 std::map<const AstArgument*, bool> getGroundedTerms(const AstClause& clause) {
149  struct Analysis : public AstConstraintAnalysis<BoolDisjunctVar> {
150  std::set<const AstAtom*> ignore;
151 
152  // atoms are producing grounded variables
153  void visitAtom(const AstAtom& cur) override {
154  // some atoms need to be skipped (head or negation)
155  if (ignore.find(&cur) != ignore.end()) {
156  return;
157  }
158 
159  // all arguments are grounded
160  for (const auto& arg : cur.getArguments()) {
161  addConstraint(isTrue(getVar(arg)));
162  }
163  }
164 
165  // negations need to be skipped
166  void visitNegation(const AstNegation& cur) override {
167  // add nested atom to black-list
168  ignore.insert(cur.getAtom());
169  }
170 
171  // also skip head
172  void visitClause(const AstClause& clause) override {
173  // ignore head
174  ignore.insert(clause.getHead());
175  }
176 
177  // binary equality relations propagates groundness
178  void visitBinaryConstraint(const AstBinaryConstraint& cur) override {
179  // only target equality
180  if (cur.getOperator() != BinaryConstraintOp::EQ) {
181  return;
182  }
183 
184  // if equal, link right and left side
185  auto lhs = getVar(cur.getLHS());
186  auto rhs = getVar(cur.getRHS());
187 
188  addConstraint(imply(lhs, rhs));
189  addConstraint(imply(rhs, lhs));
190  }
191 
192  // record init nodes
193  void visitRecordInit(const AstRecordInit& init) override {
194  auto cur = getVar(init);
195 
196  std::vector<BoolDisjunctVar> vars;
197 
198  // if record is grounded, so are all its arguments
199  for (const auto& arg : init.getArguments()) {
200  auto arg_var = getVar(arg);
201  addConstraint(imply(cur, arg_var));
202  vars.push_back(arg_var);
203  }
204 
205  // if all arguments are grounded, so is the record
206  addConstraint(imply(vars, cur));
207  }
208 
209  // constants are also sources of grounded values
210  void visitConstant(const AstConstant& c) override {
211  addConstraint(isTrue(getVar(c)));
212  }
213 
214  // aggregators are grounding values
215  void visitAggregator(const AstAggregator& c) override {
216  addConstraint(isTrue(getVar(c)));
217  }
218 
219  // intrinsic functors with grounded values are grounded values
220  void visitIntrinsicFunctor(const AstIntrinsicFunctor& cur) override {
221  auto fun = getVar(cur);
222  std::vector<BoolDisjunctVar> varArgs;
223  for (const auto& arg : cur.getArguments()) {
224  varArgs.push_back(getVar(arg));
225  }
226  addConstraint(imply(varArgs, fun));
227  }
228 
229  // user-defined functors with grounded values are grounded values
230  void visitUserDefinedFunctor(const AstUserDefinedFunctor& cur) override {
231  auto fun = getVar(cur);
232  std::vector<BoolDisjunctVar> varArgs;
233  for (const auto& arg : cur.getArguments()) {
234  varArgs.push_back(getVar(arg));
235  }
236  addConstraint(imply(varArgs, fun));
237  }
238 
239  // casts propogate groundedness in and out
240  void visitTypeCast(const AstTypeCast& cast) override {
241  addConstraint(imply(getVar(cast.getValue()), getVar(cast)));
242  }
243  };
244 
245  // run analysis on given clause
246  return Analysis().analyse(clause);
247 }
