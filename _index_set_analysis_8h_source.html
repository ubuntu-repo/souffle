171  /** count the number of bits in key */
172  static size_t card(SearchColumns cols) {
173  size_t sz = 0, idx = 1;
174  for (size_t i = 0; i < sizeof(SearchColumns) * 8; i++) {

183  /** maps search columns to an lexicographical order (labeled by a number) */
184  int map(SearchColumns cols) const {
185  assert(orders.size() == chainToOrder.size() && "Order and Chain Sizes do not match!!");
186  int i = 0;
187  for (auto it = chainToOrder.begin(); it != chainToOrder.end(); ++it, ++i) {
188  if (it->find(cols) != it->end()) {
189  assert((size_t)i < orders.size());

202  /** insert an index based on the delta*/
203  void insertIndex(std::vector<int>& ids, SearchColumns delta) {
204  int pos = 0;
205  SearchColumns mask = 0;
206 
207  while (mask < delta) {
208  mask = SearchColumns(1 << (pos));
209  SearchColumns result = (delta) & (mask);
210  if (result) {
211  ids.push_back(pos);
212  }
213  pos++;
214  }
215  }
