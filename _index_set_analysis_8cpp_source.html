67 /** Breadth first search */
68 bool MaxMatching::bfSearch() {
69  SearchColumns u;
70  std::queue<SearchColumns> bfQueue;
71  // Build layers
72  for (auto& it : graph) {
73  if (getMatch(it.first) == NIL) {
74  distance[it.first] = 0;
75  bfQueue.push(it.first);
76  } else {
77  distance[it.first] = INF;
78  }
79  }
80  distance[NIL] = INF;
81  while (!bfQueue.empty()) {
82  u = bfQueue.front();
83  bfQueue.pop();
84  assert(u != NIL);
85  const Edges& children = graph[u];
86  for (auto it : children) {
87  SearchColumns mv = getMatch(it);
88  if (getDistance(mv) == INF) {
89  distance[mv] = getDistance(u) + 1;
90  if (mv != NIL) {
91  bfQueue.push(mv);
92  }
93  }
94  }
95  }
96  return (getDistance(0) != INF);
97 }

99 /** Depth first search */
100 bool MaxMatching::dfSearch(SearchColumns u) {
101  if (u != 0) {
102  Edges& children = graph[u];
103  for (auto v : children) {
104  if (getDistance(getMatch(v)) == getDistance(u) + 1) {
105  if (dfSearch(getMatch(v))) {
106  match[u] = v;
107  match[v] = u;
108  return true;
109  }
110  }
111  }
112 
113  distance[u] = INF;
114  return false;
115  }
116  return true;
117 }

135 /** map the keys in the key set to lexicographical order */
136 void IndexSet::solve() {
137  if (searches.empty()) {
138  return;
139  }
140 
141  // check whether one of the naive indexers should be used
142  // two conditions: either set by environment or relation is a hash map
143  static const char ENV_NAIVE_INDEX[] = "SOUFFLE_USE_NAIVE_INDEX";
144  if (std::getenv(ENV_NAIVE_INDEX)) {
145  static bool first = true;
146  // print a warning - only the first time
147  if (first) {
148  std::cout << "WARNING: auto index selection disabled, naive indexes are utilized!!\n";
149  first = false;
150  }
151 
152  // every search pattern gets its naive index
153  for (SearchColumns cur : searches) {
154  // obtain order
155  LexicographicalOrder order;
156  SearchColumns mask = cur;
157  for (int i = 0; mask != 0; i++) {
158  if (!(1 << i & mask)) {
159  continue;
160  }
161  order.push_back(i);
162  // clear bit
163  mask &= ~(1 << i);
164  }
165 
166  // add new order
167  orders.push_back(order);
168 
169  // register pseudo chain
170  chainToOrder.push_back(Chain());
171  chainToOrder.back().insert(cur);
172  }
173 
174  return;
175  }
176 
177  // Construct the matching poblem
178  for (auto search : searches) {
179  // For this node check if other nodes are strict subsets
180  for (auto itt : searches) {
181  if (isStrictSubset(search, itt)) {
182  matching.addEdge(search, toB(itt));
183  }
184  }
185  }
186 
187  // Perform the Hopcroft-Karp on the graph and receive matchings (mapped A->B and B->A)
188  // Assume: alg.calculate is not called on an empty graph
189  assert(!searches.empty());
190  const MaxMatching::Matchings& matchings = matching.solve();
191 
192  // Extract the chains given the nodes and matchings
193  const ChainOrderMap chains = getChainsFromMatching(matchings, searches);
194 
195  // Should never get no chains back as we never call calculate on an empty graph
196  assert(!chains.empty());
197 
198  for (const auto& chain : chains) {
199  std::vector<int> ids;
200  SearchColumns initDelta = *(chain.begin());
201  insertIndex(ids, initDelta);
202 
203  for (auto iit = chain.begin(); next(iit) != chain.end(); ++iit) {
204  SearchColumns delta = *(next(iit)) - *iit;
205  insertIndex(ids, delta);
206  }
207 
208  assert(!ids.empty());
209 
210  orders.push_back(ids);
211  }
212 
213  // Construct the matching poblem
214  for (auto search : searches) {
215  int idx = map(search);
216  size_t l = card(search);
217  SearchColumns k = 0;
218  for (size_t i = 0; i < l; i++) {
219  k = k + (1 << (orders[idx][i]));
220  }
221  assert(k == search && "incorrect lexicographical order");
222  }
223 }

225 /** given an unmapped node from set A we follow it from set B until it cannot be matched from B
226  if not mateched from B then umn is a chain*/
227 IndexSet::Chain IndexSet::getChain(const SearchColumns umn, const MaxMatching::Matchings& match) {
228  SearchColumns start = umn; // start at an unmateched node
229  Chain chain;
230  // Assume : no circular mappings, i.e. a in A -> b in B -> ........ -> a in A is not allowed.
231  // Given this, the loop will terminate
232  while (true) {
233  auto mit = match.find(toB(start)); // we start from B side
234  chain.insert(start);
235 
236  if (mit == match.end()) {
237  return chain;
238  }
239 
240  SearchColumns a = mit->second;
241  chain.insert(a);
242  start = a;
243  }
244 }

246 /** get all chains from the matching */
247 const IndexSet::ChainOrderMap IndexSet::getChainsFromMatching(
248  const MaxMatching::Matchings& match, const SearchSet& nodes) {
249  assert(!nodes.empty());
250 
251  // Get all unmatched nodes from A
252  const SearchSet& umKeys = getUnmatchedKeys(match, nodes);
253 
254  // Case: if no unmatched nodes then we have an anti-chain
255  if (umKeys.empty()) {
256  for (auto node : nodes) {
257  std::set<SearchColumns> a;
258  a.insert(node);
259  chainToOrder.push_back(a);
260  return chainToOrder;
261  }
262  }
263 
264  assert(!umKeys.empty());
265 
266  // A worklist of used nodes
267  SearchSet usedKeys;
268 
269  // Case: nodes < umKeys or if nodes == umKeys then anti chain - this is handled by this loop
270  for (auto umKey : umKeys) {
271  Chain c = getChain(umKey, match);
272  assert(!c.empty());
273  chainToOrder.push_back(c);
274  }
275 
276  assert(!chainToOrder.empty());
277

281 /** Compute indexes */
282 void IndexSetAnalysis::run(const RamTranslationUnit& translationUnit) {
283  const auto* indexKeysAnalysis = translationUnit.getAnalysis<RamIndexKeysAnalysis>();
284  const auto* existCheckAnalysis = translationUnit.getAnalysis<RamExistenceCheckAnalysis>();
285  const auto* provExistCheckAnalysis = translationUnit.getAnalysis<RamProvenanceExistenceCheckAnalysis>();
286 
287  // visit all nodes to collect searches of each relation
288  visitDepthFirst(*translationUnit.getProgram(), [&](const RamNode& node) {
289  if (const auto* indexSearch = dynamic_cast<const RamIndexRelationSearch*>(&node)) {
290  IndexSet& indexes = getIndexes(indexSearch->getRelation());
291  indexes.addSearch(indexKeysAnalysis->getRangeQueryColumns(indexSearch));
292  } else if (const auto* exists = dynamic_cast<const RamExistenceCheck*>(&node)) {
293  IndexSet& indexes = getIndexes(exists->getRelation());
294  indexes.addSearch(existCheckAnalysis->getKey(exists));
295  } else if (const auto* provExists = dynamic_cast<const RamProvenanceExistenceCheck*>(&node)) {
296  IndexSet& indexes = getIndexes(provExists->getRelation());
297  indexes.addSearch(provExistCheckAnalysis->getKey(provExists));
298  }
299  });
300 
301  // find optimal indexes for relations
302  for (auto& cur : data) {
303  IndexSet& indexes = cur.second;
304  indexes.solve();
305  }
306 }
