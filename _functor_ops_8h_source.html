60 inline bool isValidFunctorOpArity(FunctorOp op, size_t arity) {
61  switch (op) {
62  /** Unary Functor Operators */
63  case FunctorOp::ORD:
64  case FunctorOp::STRLEN:
65  case FunctorOp::NEG:
66  case FunctorOp::BNOT:
67  case FunctorOp::LNOT:
68  case FunctorOp::TONUMBER:
69  case FunctorOp::TOSTRING:
70  return arity == 1;
71 
72  /** Binary Functor Operators */
73  case FunctorOp::ADD:
74  case FunctorOp::SUB:
75  case FunctorOp::MUL:
76  case FunctorOp::DIV:
77  case FunctorOp::EXP:
78  case FunctorOp::MOD:
79  case FunctorOp::BAND:
80  case FunctorOp::BOR:
81  case FunctorOp::BXOR:
82  case FunctorOp::LAND:
83  case FunctorOp::LOR:
84  return arity == 2;
85 
86  /** Ternary Functor Operators */
87  case FunctorOp::SUBSTR:
88  return arity == 3;
89 
90  /** Non-fixed */
91  case FunctorOp::MAX:
92  case FunctorOp::MIN:
93  case FunctorOp::CAT:
94  return arity >= 2;
95 
96  /** Undefined */
97  default:
98  break;
99  }
100 
101  assert(false && "unsupported operator");
102  return false;
103 }

223 inline bool functorOpAcceptsNumbers(size_t arg, const FunctorOp op) {

227  case FunctorOp::BNOT:
228  case FunctorOp::LNOT:

232  case FunctorOp::ORD:
233  case FunctorOp::STRLEN:
234  case FunctorOp::TONUMBER:
235  assert(arg < 1 && "unary functor argument out of bounds");
236  return false;
237 
238  /** Binary Functor Operators */
239  case FunctorOp::ADD:
240  case FunctorOp::SUB:
241  case FunctorOp::MUL:
242  case FunctorOp::DIV:
243  case FunctorOp::EXP:
244  case FunctorOp::BAND:
245  case FunctorOp::BOR:
246  case FunctorOp::BXOR:
247  case FunctorOp::LAND:
248  case FunctorOp::LOR:
249  case FunctorOp::MOD:
250  assert(arg < 2 && "binary functor argument out of bounds");
251  return true;
252 
253  /** Ternary Functor Operators */
254  case FunctorOp::SUBSTR:
255  assert(arg < 3 && "ternary functor argument out of bounds");
256  return arg == 1 || arg == 2;
257 
258  /** Non-fixed Functor Operators */
259  case FunctorOp::MAX:

261  return true;
262  case FunctorOp::CAT:
