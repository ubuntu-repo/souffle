37 void ComponentLookup::run(const AstTranslationUnit& translationUnit) {
38  const AstProgram* program = translationUnit.getProgram();
39  for (AstComponent* component : program->getComponents()) {
40  globalScopeComponents.insert(component);
41  enclosingComponent[component] = nullptr;
42  }
43  visitDepthFirst(*program, [&](const AstComponent& cur) {
44  nestedComponents[&cur];
45  for (AstComponent* nestedComponent : cur.getComponents()) {
46  nestedComponents[&cur].insert(nestedComponent);
47  enclosingComponent[nestedComponent] = &cur;
48  }
49  });
50 }

52 const AstComponent* ComponentLookup::getComponent(
53  const AstComponent* scope, const std::string& name, const TypeBinding& activeBinding) const {
54  // forward according to binding (we do not do this recursively on purpose)
55  AstTypeIdentifier boundName = activeBinding.find(name);
56  if (boundName.empty()) {
57  // compName is not bound to anything => just just compName
58  boundName = name;
59  }
60 
61  // search nested scopes bottom up
62  const AstComponent* searchScope = scope;
63  while (searchScope != nullptr) {
64  for (const AstComponent* cur : searchScope->getComponents()) {
65  if (cur->getComponentType()->getName() == toString(boundName)) {
66  return cur;
67  }
68  }
69  auto found = enclosingComponent.find(searchScope);
70  if (found != enclosingComponent.end()) {
71  searchScope = found->second;
72  } else {
73  searchScope = nullptr;
74  break;
75  }
76  }
77 
78  // check global scope
79  for (const AstComponent* cur : globalScopeComponents) {
80  if (cur->getComponentType()->getName() == toString(boundName)) {
81  return cur;
82  }
83  }
84 
85  // no such component in scope
86  return nullptr;
87 }

103  void add(std::unique_ptr<AstType>& type, ErrorReport& report) {
104  // add to result content (check existence first)
105  auto foundItem =
106  std::find_if(types.begin(), types.end(), [&](const std::unique_ptr<AstType>& element) {
107  return (element->getName() == type->getName());
108  });
109  if (foundItem != types.end()) {
110  Diagnostic err(Diagnostic::ERROR,
111  DiagnosticMessage("Redefinition of type " + toString(type->getName()), type->getSrcLoc()),
112  {DiagnosticMessage("Previous definition", (*foundItem)->getSrcLoc())});
113  report.addDiagnostic(err);
114  }
115  types.push_back(std::move(type));
116  }
117 
118  void add(std::unique_ptr<AstRelation>& rel, ErrorReport& report) {
119  // add to result content (check existence first)
120  auto foundItem = std::find_if(
121  relations.begin(), relations.end(), [&](const std::unique_ptr<AstRelation>& element) {
122  return (element->getName() == rel->getName());
123  });
124  if (foundItem != relations.end()) {
125  Diagnostic err(Diagnostic::ERROR,
126  DiagnosticMessage(
127  "Redefinition of relation " + toString(rel->getName()), rel->getSrcLoc()),
128  {DiagnosticMessage("Previous definition", (*foundItem)->getSrcLoc())});
129  report.addDiagnostic(err);
130  }
131  relations.push_back(std::move(rel));
132  }

134  void add(std::unique_ptr<AstLoad>& directive, ErrorReport& report) {
135  // Check if load directive already exists
136  auto foundItem = std::find_if(loads.begin(), loads.end(), [&](const std::unique_ptr<AstLoad>& load) {
137  return load->getName() == directive->getName();

139  // if yes, add error

144  {DiagnosticMessage("Previous definition", (*foundItem)->getSrcLoc())});
145  report.addDiagnostic(err);
146  }
147  // if not, add it

186 void collectContent(const AstComponent& component, const TypeBinding& binding,
187  const AstComponent* enclosingComponent, const ComponentLookup& componentLookup, ComponentContent& res,
188  std::vector<std::unique_ptr<AstClause>>& orphans, const std::set<std::string>& overridden,
189  ErrorReport& report, unsigned int maxInstantiationDepth) {
190  // start with relations and clauses of the base components
191  for (const auto& base : component.getBaseComponents()) {
192  const AstComponent* comp = componentLookup.getComponent(enclosingComponent, base->getName(), binding);
193  if (comp) {
194  // link formal with actual type parameters
195  const auto& formalParams = comp->getComponentType()->getTypeParameters();
196  const auto& actualParams = base->getTypeParameters();
197 
198  // update type binding
199  TypeBinding activeBinding = binding.extend(formalParams, actualParams);
200 
201  for (const auto& cur : comp->getInstantiations()) {
202  // instantiate sub-component
203  ComponentContent content = getInstantiatedContent(*cur, enclosingComponent, componentLookup,
204  orphans, report, activeBinding, maxInstantiationDepth - 1);
205 
206  // process types
207  for (auto& type : content.types) {
208  res.add(type, report);
209  }
210 
211  // process relations
212  for (auto& rel : content.relations) {
213  res.add(rel, report);
214  }
215 
216  // process io directives
217  for (auto& io : content.loads) {
218  res.add(io, report);
219  }
220  for (auto& io : content.printSizes) {
221  res.add(io, report);
222  }
223  for (auto& io : content.stores) {
224  res.add(io, report);
225  }
226  }
227 
228  // collect definitions from base type
229  std::set<std::string> superOverridden;
230  superOverridden.insert(overridden.begin(), overridden.end());
231  superOverridden.insert(component.getOverridden().begin(), component.getOverridden().end());
232  collectContent(*comp, activeBinding, comp, componentLookup, res, orphans, superOverridden, report,
233  maxInstantiationDepth);
234  }
235  }
236 
237  // and continue with the local types
238  for (const auto& cur : component.getTypes()) {
239  // create a clone
240  std::unique_ptr<AstType> type(cur->clone());
241 
242  // instantiate elements of union types
243  visitDepthFirst(*type, [&](const AstUnionType& type) {
244  for (const auto& name : type.getTypes()) {
245  AstTypeIdentifier newName = binding.find(name);
246  if (!newName.empty()) {
247  const_cast<AstTypeIdentifier&>(name) = newName;
248  }
249  }
250  });
251 
252  // instantiate elements of record types
253  visitDepthFirst(*type, [&](const AstRecordType& type) {
254  for (const auto& field : type.getFields()) {
255  AstTypeIdentifier newName = binding.find(field.type);
256  if (!newName.empty()) {
257  const_cast<AstTypeIdentifier&>(field.type) = newName;
258  }
259  }
260  });
261 
262  // add to result list (check existence first)
263  res.add(type, report);
264  }
265 
266  // and the local relations
267  for (const auto& cur : component.getRelations()) {
268  // create a clone
269  std::unique_ptr<AstRelation> rel(cur->clone());
270 
271  // update attribute types
272  for (AstAttribute* attr : rel->getAttributes()) {
273  AstTypeIdentifier forward = binding.find(attr->getTypeName());
274  if (!forward.empty()) {
275  attr->setTypeName(forward);
276  }
277  }
278 
279  // add to result list (check existence first)
280  res.add(rel, report);
281  }
282 
283  // and the local io directives
284  for (const auto& cur : component.getLoads()) {
285  // create a clone
286  std::unique_ptr<AstLoad> io(cur->clone());
287 
288  res.add(io, report);
289  }
290  for (const auto& cur : component.getPrintSizes()) {
291  // create a clone
292  std::unique_ptr<AstPrintSize> io(cur->clone());
293 
294  res.add(io, report);
295  }
296  for (const auto& cur : component.getStores()) {
297  // create a clone
298  std::unique_ptr<AstStore> io(cur->clone());
299 
300  res.add(io, report);
301  }
302 
303  // index the available relations
304  std::map<AstRelationIdentifier, AstRelation*> index;
305  for (const auto& cur : res.relations) {
306  index[cur->getName()] = cur.get();
307  }
308 
309  // add the local clauses
310  for (const auto& cur : component.getClauses()) {
311  if (overridden.count(cur->getHead()->getName().getNames()[0]) == 0) {
312  AstRelation* rel = index[cur->getHead()->getName()];
313  if (rel) {
314  rel->addClause(std::unique_ptr<AstClause>(cur->clone()));
315  } else {
316  orphans.emplace_back(cur->clone());
317  }
318  }
319  }
320 
321  // add orphan clauses at the current level if they can be resolved
322  for (auto iter = orphans.begin(); iter != orphans.end();) {
323  auto& cur = *iter;
324  AstRelation* rel = index[cur->getHead()->getName()];
325  if (rel) {
326  // add orphan to current instance and delete from orphan list
327  rel->addClause(std::unique_ptr<AstClause>(cur->clone()));
328  iter = orphans.erase(iter);
329  } else {
330  ++iter;
331  }
332  }
333 }

335 ComponentContent getInstantiatedContent(const AstComponentInit& componentInit,
336  const AstComponent* enclosingComponent, const ComponentLookup& componentLookup,
337  std::vector<std::unique_ptr<AstClause>>& orphans, ErrorReport& report, const TypeBinding& binding,
338  unsigned int maxDepth) {
339  // start with an empty list
340  ComponentContent res;
341 
342  if (maxDepth == 0) {
343  report.addError("Component instantiation limit reached", componentInit.getSrcLoc());
344  return res;
345  }
346 
347  // get referenced component
348  const AstComponent* component = componentLookup.getComponent(
349  enclosingComponent, componentInit.getComponentType()->getName(), binding);
350  if (!component) {
351  // this component is not defined => will trigger a semantic error
352  return res;
353  }
354 
355  // update type biding
356  const auto& formalParams = component->getComponentType()->getTypeParameters();
357  const auto& actualParams = componentInit.getComponentType()->getTypeParameters();
358  TypeBinding activeBinding = binding.extend(formalParams, actualParams);
359 
360  // instantiated nested components
361  for (const auto& cur : component->getInstantiations()) {
362  // get nested content
363  ComponentContent nestedContent = getInstantiatedContent(
364  *cur, component, componentLookup, orphans, report, activeBinding, maxDepth - 1);
365 
366  // add types
367  for (auto& type : nestedContent.types) {
368  res.add(type, report);
369  }
370 
371  // add relations
372  for (auto& rel : nestedContent.relations) {
373  res.add(rel, report);
374  }
375 
376  // add IO directives
377  for (auto& io : nestedContent.loads) {
378  res.add(io, report);
379  }
380  for (auto& io : nestedContent.printSizes) {
381  res.add(io, report);
382  }
383  for (auto& io : nestedContent.stores) {
384  res.add(io, report);
385  }
386  }
387 
388  // collect all content in this component
389  std::set<std::string> overridden;
390  collectContent(*component, activeBinding, enclosingComponent, componentLookup, res, orphans, overridden,
391  report, maxDepth);
392 
393  // update type names
394  std::map<AstTypeIdentifier, AstTypeIdentifier> typeNameMapping;
395  for (const auto& cur : res.types) {
396  auto newName = componentInit.getInstanceName() + cur->getName();
397  typeNameMapping[cur->getName()] = newName;
398  cur->setName(newName);
399  }
400 
401  // update relation names
402  std::map<AstRelationIdentifier, AstRelationIdentifier> relationNameMapping;
403  for (const auto& cur : res.relations) {
404  auto newName = componentInit.getInstanceName() + cur->getName();
405  relationNameMapping[cur->getName()] = newName;
406  cur->setName(newName);
407  }
408 
409  // create a helper function fixing type and relation references
410  auto fixNames = [&](const AstNode& node) {
411  // rename attribute types in headers
412  visitDepthFirst(node, [&](const AstAttribute& attr) {
413  auto pos = typeNameMapping.find(attr.getTypeName());
414  if (pos != typeNameMapping.end()) {
415  const_cast<AstAttribute&>(attr).setTypeName(pos->second);
416  }
417  });
418 
419  // rename atoms in clauses
420  visitDepthFirst(node, [&](const AstAtom& atom) {
421  auto pos = relationNameMapping.find(atom.getName());
422  if (pos != relationNameMapping.end()) {
423  const_cast<AstAtom&>(atom).setName(pos->second);
424  }
425  });
426 
427  // rename IO directives
428  visitDepthFirst(node, [&](const AstLoad& load) {
429  auto pos = relationNameMapping.find(load.getName());
430  if (pos != relationNameMapping.end()) {
431  const_cast<AstLoad&>(load).setName(pos->second);
432  }
433  });
434  visitDepthFirst(node, [&](const AstPrintSize& printSize) {
435  auto pos = relationNameMapping.find(printSize.getName());
436  if (pos != relationNameMapping.end()) {
437  const_cast<AstPrintSize&>(printSize).setName(pos->second);
438  }
439  });
440  visitDepthFirst(node, [&](const AstStore& store) {
441  auto pos = relationNameMapping.find(store.getName());
442  if (pos != relationNameMapping.end()) {
443  const_cast<AstStore&>(store).setName(pos->second);
444  }
445  });
446 
447  // rename field types in records
448  visitDepthFirst(node, [&](const AstRecordType& recordType) {
449  auto& fields = recordType.getFields();
450  for (size_t i = 0; i < fields.size(); i++) {
451  auto& field = fields[i];
452  auto pos = typeNameMapping.find(field.type);
453  if (pos != typeNameMapping.end()) {
454  const_cast<AstRecordType&>(recordType).setFieldType(i, pos->second);
455  }
456  }
457  });
458 
459  // rename variant types in unions
460  visitDepthFirst(node, [&](const AstUnionType& unionType) {
461  auto& variants = unionType.getTypes();
462  for (size_t i = 0; i < variants.size(); i++) {
463  auto pos = typeNameMapping.find(variants[i]);
464  if (pos != typeNameMapping.end()) {
465  const_cast<AstUnionType&>(unionType).setVariantType(i, pos->second);
466  }
467  }
468  });
469 
470  // rename type information in typecast
471  visitDepthFirst(node, [&](const AstTypeCast& cast) {
472  auto pos = typeNameMapping.find(cast.getType());
473  if (pos != typeNameMapping.end()) {
474  const_cast<AstTypeCast&>(cast).setType(pos->second);
475  }
476  });
477  };
478 
479  // rename attribute type in headers and atoms in clauses of the relation
480  for (const auto& cur : res.relations) {
481  fixNames(*cur);
482  }
483 
484  // rename orphans
485  for (const auto& cur : orphans) {
486  fixNames(*cur);
487  }
488 
489  // rename io directives
490  for (const auto& cur : res.loads) {
491  fixNames(*cur);
492  }
493  for (const auto& cur : res.printSizes) {
494  fixNames(*cur);
495  }
496  for (const auto& cur : res.stores) {
497  fixNames(*cur);
498  }
499 
500  // rename subtypes
501  for (const auto& cur : res.types) {
502  fixNames(*cur);
503  }
504 
505  // done
506  return res;
507 }

510 bool ComponentInstantiationTransformer::transform(AstTranslationUnit& translationUnit) {
511  // TODO: Do this without being a friend class of AstProgram
512 
513  // unbound clauses with no relation defined
514  std::vector<std::unique_ptr<AstClause>> unbound;
515 
516  AstProgram& program = *translationUnit.getProgram();
517 
518  auto* componentLookup = translationUnit.getAnalysis<ComponentLookup>();
519 
520  for (const auto& cur : program.instantiations) {
521  std::vector<std::unique_ptr<AstClause>> orphans;
522 
523  ComponentContent content = getInstantiatedContent(
524  *cur, nullptr, *componentLookup, orphans, translationUnit.getErrorReport());
525  for (auto& type : content.types) {
526  program.types.insert(std::make_pair(type->getName(), std::move(type)));
527  }
528  for (auto& rel : content.relations) {
529  program.relations.insert(std::make_pair(rel->getName(), std::move(rel)));
530  }
531  for (auto& io : content.loads) {
532  program.loads.push_back(std::move(io));
533  }
534  for (auto& io : content.printSizes) {
535  program.printSizes.push_back(std::move(io));
536  }
537  for (auto& io : content.stores) {
538  program.stores.push_back(std::move(io));
539  }
540  for (auto& cur : orphans) {
541  auto pos = program.relations.find(cur->getHead()->getName());
542  if (pos != program.relations.end()) {
543  pos->second->addClause(std::move(cur));
544  } else {
545  unbound.push_back(std::move(cur));
546  }
547  }
548  }
549 
550  // add clauses
551  for (auto& cur : program.clauses) {
552  auto pos = program.relations.find(cur->getHead()->getName());
553  if (pos != program.relations.end()) {
554  pos->second->addClause(std::move(cur));
555  } else {
556  unbound.push_back(std::move(cur));
557  }
558  }
559  // remember the remaining orphan clauses
560  program.clauses.clear();
561  program.clauses.swap(unbound);
562 
563  // unbound directives with no relation defined
564  std::vector<std::unique_ptr<AstLoad>> unboundLoads;
565  std::vector<std::unique_ptr<AstPrintSize>> unboundPrintSizes;
566  std::vector<std::unique_ptr<AstStore>> unboundStores;
567 
568  // add IO directives
569  for (auto& cur : program.loads) {
570  auto pos = program.relations.find(cur->getName());
571  if (pos != program.relations.end()) {
572  pos->second->addLoad(std::move(cur));
573  } else {
574  unboundLoads.push_back(std::move(cur));
575  }
576  }
577  // remember the remaining orphan directives
578  program.loads.clear();
579  program.loads.swap(unboundLoads);
580 
581  for (auto& cur : program.stores) {
582  auto pos = program.relations.find(cur->getName());
583  if (pos != program.relations.end()) {
584  pos->second->addStore(std::move(cur));
585  } else {
586  unboundStores.push_back(std::move(cur));
587  }
588  }
589  // remember the remaining orphan directives
590  program.stores.clear();
591  program.stores.swap(unboundStores);
592 
593  // delete components and instantiations
594  program.instantiations.clear();
595  program.components.clear();
596 
597  return true;
598 }
