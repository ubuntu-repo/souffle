27 std::string SrcLocation::extloc() const {
28  std::ifstream in(filename);
29  std::stringstream s;
30  if (in.is_open()) {
31  s << "file " << baseName(filename) << " at line " << start.line << "\n";
32  for (int i = 0; i < start.line - 1; ++i) {
33  in.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
34  }
35  int c;
36  int lineLen = 0;
37  int offsetColumn = start.column;
38  bool prevWhitespace = false;
39  bool afterFirstNonSpace = false;
40  while ((c = in.get()) != '\n' && c != EOF) {
41  s << (char)c;
42  lineLen++;
43 
44  // Offset column to account for C preprocessor having reduced
45  // consecutive non-leading whitespace chars to a single space.
46  if (std::isspace(c)) {
47  if (afterFirstNonSpace && prevWhitespace && offsetColumn >= lineLen) {
48  offsetColumn++;
49  }
50  prevWhitespace = true;
51  } else {
52  prevWhitespace = false;
53  afterFirstNonSpace = true;
54  }
55  }
56  lineLen++; // Add new line
57  in.close();
58  s << "\n";
59  for (int i = 1; i <= lineLen; i++) {
60  char ch = (i == offsetColumn) ? '^' : '-';
61  s << ch;
62  }
63  in.close();
64  } else {
65  s << "unknown source location.";
66  }
