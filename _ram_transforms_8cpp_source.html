33 bool ExpandFilterTransformer::expandFilters(RamProgram& program) {
34  // flag to determine whether the RAM program has changed
35  bool changed = false;
36 
37  visitDepthFirst(program, [&](const RamQuery& query) {
38  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
39  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
40  if (const RamFilter* filter = dynamic_cast<RamFilter*>(node.get())) {
41  const RamCondition* condition = &filter->getCondition();
42  std::vector<std::unique_ptr<RamCondition>> conditionList = toConjunctionList(condition);
43  if (conditionList.size() > 1) {
44  changed = true;
45  std::vector<std::unique_ptr<RamFilter>> filters;
46  for (auto iter = conditionList.rbegin(); iter != conditionList.rend(); ++iter) {
47  auto& cond = *iter;
48  auto tempCond = cond->clone();
49  if (filters.empty()) {
50  filters.emplace_back(std::make_unique<RamFilter>(
51  std::unique_ptr<RamCondition>(std::move(tempCond)),
52  std::unique_ptr<RamOperation>(filter->getOperation().clone())));
53  } else {
54  filters.emplace_back(std::make_unique<RamFilter>(
55  std::unique_ptr<RamCondition>(std::move(tempCond)),
56  std::move(filters.back())));
57  }
58  }
59  node = std::move(filters.back());
60  }
61  }
62  node->apply(makeLambdaRamMapper(filterRewriter));
63  return node;
64  };
65  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(filterRewriter));
66  });
67  return changed;
68 }

70 bool HoistConditionsTransformer::hoistConditions(RamProgram& program) {
71  // flag to determine whether the RAM program has changed
72  bool changed = false;
73 
74  // helper for collecting conditions from filter operations
75  auto addCondition = [](std::unique_ptr<RamCondition> condition,
76  RamCondition* c) -> std::unique_ptr<RamCondition> {
77  if (condition == nullptr) {
78  return std::unique_ptr<RamCondition>(c);
79  } else {
80  return std::make_unique<RamConjunction>(std::move(condition), std::unique_ptr<RamCondition>(c));
81  }
82  };
83 
84  // insert a new filter
85  auto insertFilter = [](RamOperation* op, std::unique_ptr<RamCondition>& condition) {
86  op->apply(makeLambdaRamMapper([&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
87  if (nullptr != dynamic_cast<RamOperation*>(node.get())) {
88  return std::make_unique<RamFilter>(std::move(condition),
89  std::unique_ptr<RamOperation>(dynamic_cast<RamOperation*>(node.release())));
90  }
91  return node;
92  }));
93  };
94 
95  // hoist conditions to the most outer scope if they
96  // don't depend on RamSearches
97  visitDepthFirst(program, [&](const RamQuery& query) {
98  std::unique_ptr<RamCondition> newCondition;
99  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
100  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
101  if (auto* filter = dynamic_cast<RamFilter*>(node.get())) {
102  const RamCondition& condition = filter->getCondition();
103  // if filter condition is independent of any RamSearch,
104  // delete the filter operation and collect condition
105  if (rla->getLevel(&condition) == -1) {
106  changed = true;
107  newCondition = addCondition(std::move(newCondition), condition.clone());
108  node->apply(makeLambdaRamMapper(filterRewriter));
109  return std::unique_ptr<RamOperation>(filter->getOperation().clone());
110  }
111  }
112  node->apply(makeLambdaRamMapper(filterRewriter));
113  return node;
114  };
115  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(filterRewriter));
116  if (newCondition != nullptr) {
117  // insert new filter operation at outer-most level of the query
118  changed = true;
119  insertFilter((RamOperation*)&query, newCondition);
120  }
121  });
122 
123  // hoist conditions for each RamSearch operation
124  visitDepthFirst(program, [&](const RamSearch& search) {
125  std::unique_ptr<RamCondition> newCondition;
126  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> filterRewriter =
127  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
128  if (auto* filter = dynamic_cast<RamFilter*>(node.get())) {
129  const RamCondition& condition = filter->getCondition();
130  // if filter condition matches level of RamSearch,
131  // delete the filter operation and collect condition
132  if (rla->getLevel(&condition) == search.getTupleId()) {
133  changed = true;
134  newCondition = addCondition(std::move(newCondition), condition.clone());
135  node->apply(makeLambdaRamMapper(filterRewriter));
136  return std::unique_ptr<RamOperation>(filter->getOperation().clone());
137  }
138  }
139  node->apply(makeLambdaRamMapper(filterRewriter));
140  return node;
141  };
142  const_cast<RamSearch*>(&search)->apply(makeLambdaRamMapper(filterRewriter));
143  if (newCondition != nullptr) {
144  // insert new filter operation after the search operation
145  changed = true;
146  insertFilter((RamOperation*)&search, newCondition);
147  }
148  });
149  return changed;
150 }

152 std::unique_ptr<RamExpression> MakeIndexTransformer::getExpression(
153  RamCondition* c, size_t& element, int identifier) {
154  if (auto* binRelOp = dynamic_cast<RamConstraint*>(c)) {
155  if (binRelOp->getOperator() == BinaryConstraintOp::EQ) {
156  if (const RamElementAccess* lhs = dynamic_cast<RamElementAccess*>(binRelOp->getLHS())) {
157  RamExpression* rhs = binRelOp->getRHS();
158  if (lhs->getTupleId() == identifier && rla->getLevel(rhs) < identifier) {
159  element = lhs->getElement();
160  return std::unique_ptr<RamExpression>(rhs->clone());
161  }
162  }
163  if (const RamElementAccess* rhs = dynamic_cast<RamElementAccess*>(binRelOp->getRHS())) {
164  RamExpression* lhs = binRelOp->getLHS();
165  if (rhs->getTupleId() == identifier && rla->getLevel(lhs) < identifier) {
166  element = rhs->getElement();
167  return std::unique_ptr<RamExpression>(lhs->clone());
168  }
169  }
170  }
171  }
172  return nullptr;
173 }

175 std::unique_ptr<RamCondition> MakeIndexTransformer::constructPattern(
176  std::vector<std::unique_ptr<RamExpression>>& queryPattern, bool& indexable,
177  std::vector<std::unique_ptr<RamCondition>> conditionList, int identifier) {
178  // Remaining conditions which cannot be handled by an index
179  std::unique_ptr<RamCondition> condition;
180  auto addCondition = [&](std::unique_ptr<RamCondition> c) {
181  if (condition != nullptr) {
182  condition = std::make_unique<RamConjunction>(std::move(condition), std::move(c));
183  } else {
184  condition = std::move(c);
185  }
186  };
187 
188  // Build query pattern and remaining condition
189  for (auto& cond : conditionList) {
190  size_t element = 0;
191  if (std::unique_ptr<RamExpression> value = getExpression(cond.get(), element, identifier)) {
192  if (queryPattern[element] == nullptr) {
193  indexable = true;
194  queryPattern[element] = std::move(value);
195  } else {
196  // TODO: This case is a recursive case introducing a new filter operation
197  // at upper level, i.e., if queryPattern[element] == value ...
198  // and apply indexing recursively to the rewritten program.
199  // At the moment we just another local condition which is sub-optimal
200  // Note sure whether there are cases in practice that would improve the performance
201  addCondition(std::make_unique<RamConstraint>(BinaryConstraintOp::EQ, std::move(value),
202  std::unique_ptr<RamExpression>(queryPattern[element]->clone())));
203  }
204  } else {
205  addCondition(std::move(cond));
206  }
207  }
208  return condition;
209 }

278 std::unique_ptr<RamOperation> IfConversionTransformer::rewriteIndexScan(const RamIndexScan* indexScan) {
279  // check whether tuple is used in subsequent operations
280  bool tupleNotUsed = true;
281  visitDepthFirst(*indexScan, [&](const RamNode& node) {
282  if (const RamElementAccess* element = dynamic_cast<const RamElementAccess*>(&node)) {
283  if (element->getTupleId() == indexScan->getTupleId()) {
284  tupleNotUsed = false;
285  }
286  } else if (const RamUnpackRecord* unpack = dynamic_cast<const RamUnpackRecord*>(&node)) {
287  if (unpack->getReferenceLevel() == indexScan->getTupleId()) {
288  tupleNotUsed = false;
289  }
290  }
291  });
292 
293  // if not used, transform the IndexScan operation to an existence check
294  if (tupleNotUsed) {
295  // replace IndexScan with an Filter/Existence check
296  std::vector<std::unique_ptr<RamExpression>> newValues;
297  for (auto& cur : indexScan->getRangePattern()) {
298  RamExpression* val = nullptr;
299  if (cur != nullptr) {
300  val = cur->clone();
301  }
302  newValues.emplace_back(val);
303  }
304 
305  // check if there is a break statement nested in the Scan - if so, remove it
306  RamOperation* newOp;
307  if (const RamBreak* breakOp = dynamic_cast<const RamBreak*>(&indexScan->getOperation())) {
308  newOp = breakOp->getOperation().clone();
309  } else {
310  newOp = indexScan->getOperation().clone();
311  }
312 
313  return std::make_unique<RamFilter>(
314  std::make_unique<RamExistenceCheck>(
315  std::make_unique<RamRelationReference>(&indexScan->getRelation()),
316  std::move(newValues)),
317  std::unique_ptr<RamOperation>(newOp), indexScan->getProfileText());
318  }
319  return nullptr;
320 }

341 std::unique_ptr<RamOperation> ChoiceConversionTransformer::rewriteScan(const RamScan* scan) {
342  bool transformTuple = false;
343 
344  // Check that RamFilter follows the Scan in the loop nest
345  if (const auto* filter = dynamic_cast<const RamFilter*>(&scan->getOperation())) {
346  // Check that the Filter uses the identifier in the Scan
347  if (rla->getLevel(&filter->getCondition()) == scan->getTupleId()) {
348  transformTuple = true;
349 
350  // Check that the filter is not referred to after
351  const auto* nextNode = dynamic_cast<const RamNode*>(&filter->getOperation());
352 
353  visitDepthFirst(*nextNode, [&](const RamElementAccess& element) {
354  if (element.getTupleId() == scan->getTupleId()) {
355  transformTuple = false;
356  }
357  });
358 
359  visitDepthFirst(*nextNode, [&](const RamUnpackRecord& unpack) {
360  if (unpack.getReferenceLevel() == scan->getTupleId()) {
361  transformTuple = false;
362  }
363  });
364  }
365  }
366 
367  // Convert the Scan/If pair into a Choice
368  if (transformTuple) {
369  std::vector<std::unique_ptr<RamExpression>> newValues;
370  const auto* filter = dynamic_cast<const RamFilter*>(&scan->getOperation());
371  const int identifier = scan->getTupleId();
372 
373  return std::make_unique<RamChoice>(std::make_unique<RamRelationReference>(&scan->getRelation()),
374  identifier, std::unique_ptr<RamCondition>(filter->getCondition().clone()),
375  std::unique_ptr<RamOperation>(scan->getOperation().clone()), scan->getProfileText());
376  }
