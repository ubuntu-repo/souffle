176 class InterpreterProgInterface : public SouffleProgram {
177 public:
178  InterpreterProgInterface(Interpreter& interp)
179  : prog(*interp.getTranslationUnit().getProgram()), exec(interp),
180  symTable(interp.getTranslationUnit().getSymbolTable()) {
181  uint32_t id = 0;
182 
183  // Retrieve AST Relations and store them in a map
184  std::map<std::string, const RamRelation*> map;
185  visitDepthFirst(prog, [&](const RamRelation& rel) { map[rel.getName()] = &rel; });
186 
187  // Build wrapper relations for Souffle's interface
188  for (auto& rel_pair : exec.getRelationMap()) {
189  auto& name = rel_pair.first;
190  auto& interpreterRel = *rel_pair.second;
191  assert(map[name]);
192  const RamRelation& rel = *map[name];
193 
194  // construct types and names vectors
195  std::vector<std::string> types;
196  std::vector<std::string> attrNames;
197  for (size_t i = 0; i < rel.getArity(); i++) {
198  std::string t = rel.getArgTypeQualifier(i);
199  types.push_back(t);
200 
201  std::string n = rel.getArg(i);
202  attrNames.push_back(n);
203  }
204  InterpreterRelInterface* interface = new InterpreterRelInterface(
205  interpreterRel, symTable, rel.getName(), types, attrNames, id);
206  interfaces.push_back(interface);
207  bool input;
208  bool output;
209  visitDepthFirst(prog, [&](const RamStore& store) {
210  if (store.getRelation() == rel) {
211  output = true;
212  }
213  });
214  visitDepthFirst(prog, [&](const RamLoad& load) {
215  if (load.getRelation() == rel) {
216  input = true;
217  }
218  });
219  addRelation(rel.getName(), interface, input, output);
220  id++;
221  }
222  }
223  ~InterpreterProgInterface() override {
224  for (auto* interface : interfaces) {
225  delete interface;
226  }
227  }
