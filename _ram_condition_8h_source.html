417 /**
418  * @brief Convert terms of a conjunction to a list
419  * @param A RAM condition
420  * @param A list of RAM conditions
421  *
422  * Convert a condition of the format C1 /\ C2 /\ ... /\ Cn
423  * to a list {C1, C2, ..., Cn}.
424  */
425 inline std::vector<std::unique_ptr<RamCondition>> toConjunctionList(const RamCondition* condition) {
426  std::vector<std::unique_ptr<RamCondition>> list;
427  std::stack<const RamCondition*> stack;
428  if (condition != nullptr) {
429  stack.push(condition);
430  while (!stack.empty()) {
431  condition = stack.top();
432  stack.pop();
433  if (const auto* ramConj = dynamic_cast<const RamConjunction*>(condition)) {
434  stack.push(&ramConj->getLHS());
435  stack.push(&ramConj->getRHS());
436  } else {
437  list.emplace_back(condition->clone());
438  }
439  }
440  }
441  return list;
442 }

444 /**
445  * @brief Convert list of conditions to a conjunction
446  * @param A list of RAM conditions
447  * @param A RAM condition
448  *
449  * Convert a list {C1, C2, ..., Cn} to a condition of
450  * the format C1 /\ C2 /\ ... /\ Cn.
451  */
452 inline std::unique_ptr<RamCondition> toCondition(const std::vector<const RamCondition*>& list) {
453  std::unique_ptr<RamCondition> result;
454  for (const RamCondition* cur : list) {
455  if (result == nullptr) {
456  result = std::unique_ptr<RamCondition>(cur->clone());
457  } else {
458  result = std::make_unique<RamConjunction>(
459  std::move(result), std::unique_ptr<RamCondition>(cur->clone()));
460  }
461  }
462  return result;
463 }
