54  std::unique_ptr<RamDomain[]> readNextTuple() override {
55  if (sqlite3_step(selectStatement) != SQLITE_ROW) {
56  return nullptr;
57  }
58 
59  std::unique_ptr<RamDomain[]> tuple = std::make_unique<RamDomain[]>(arity + (isProvenance ? 2 : 0));
60 
61  uint32_t column;
62  for (column = 0; column < arity; column++) {
63  std::string element(reinterpret_cast<const char*>(sqlite3_column_text(selectStatement, column)));
64 
65  if (element.empty()) {
66  element = "n/a";
67  }
68  if (symbolMask.at(column)) {
69  tuple[column] = symbolTable.unsafeLookup(element);
70  } else {
71  try {
72 #if RAM_DOMAIN_SIZE == 64

128  void checkTableExists() {
129  sqlite3_stmt* tableStatement;
130  std::stringstream selectSQL;
131  selectSQL << "SELECT count(*) FROM sqlite_master WHERE type IN ('table', 'view') AND ";
132  selectSQL << " name IN ('" << relationName << "', '_" << relationName << "');";
133  const char* tail = nullptr;
134 
135  if (sqlite3_prepare_v2(db, selectSQL.str().c_str(), -1, &tableStatement, &tail) != SQLITE_OK) {
136  throwError("SQLite error in sqlite3_prepare_v2: ");
137  }
138 
139  if (sqlite3_step(tableStatement) == SQLITE_ROW) {
140  int count = sqlite3_column_int(tableStatement, 0);
141  if (count == 2) {
142  sqlite3_finalize(tableStatement);
143  return;
144  }
145  }
146  sqlite3_finalize(tableStatement);
147  throw std::invalid_argument("Required table and view does not exist for relation " + relationName);
148  }
