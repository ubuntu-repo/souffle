23 /**
24  * A wrapper for an iterator obtaining pointers of a certain type,
25  * dereferencing values before forwarding them to the consumer.
26  *
27  * @tparam Iter ... the type of wrapped iterator
28  * @tparam T ... the value to be accessed by the resulting iterator
29  */
30 template <typename Iter, typename T = typename std::remove_pointer<typename Iter::value_type>::type>
31 struct IterDerefWrapper : public std::iterator<std::forward_iterator_tag, T> {
32  /* The nested iterator. */
33  Iter iter;
34 
35 public:
36  // some constructores
37  IterDerefWrapper() = default;
38  IterDerefWrapper(const Iter& iter) : iter(iter) {}
39 
40  // defaulted copy and move constructors
41  IterDerefWrapper(const IterDerefWrapper&) = default;
42  IterDerefWrapper(IterDerefWrapper&&) = default;
43 
44  // default assignment operators
45  IterDerefWrapper& operator=(const IterDerefWrapper&) = default;
46  IterDerefWrapper& operator=(IterDerefWrapper&&) = default;
47 
48  /* The equality operator as required by the iterator concept. */
49  bool operator==(const IterDerefWrapper& other) const {
50  return iter == other.iter;
51  }
52 
53  /* The not-equality operator as required by the iterator concept. */
54  bool operator!=(const IterDerefWrapper& other) const {
55  return iter != other.iter;
56  }
57 
58  /* The deref operator as required by the iterator concept. */
59  const T& operator*() const {
60  return **iter;
61  }
62 
63  /* Support for the pointer operator. */
64  const T* operator->() const {
65  return &(**iter);
66  }
67 
68  /* The increment operator as required by the iterator concept. */
69  IterDerefWrapper& operator++() {
70  ++iter;
71  return *this;
72  }
73 };
