103  /**
104  * Abstract replacement method for a node.
105  *
106  * If the given nodes is to be replaced, the handed in node
107  * will be destroyed by the mapper and the returned node
108  * will become owned by the caller.
109  */
110  virtual std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const = 0;
111 
112  /**
113  * Wrapper for any subclass of the AST node hierarchy performing type casts.
114  */
115  template <typename T>
116  std::unique_ptr<T> operator()(std::unique_ptr<T> node) const {
117  std::unique_ptr<AstNode> resPtr =
118  (*this)(std::unique_ptr<AstNode>(static_cast<AstNode*>(node.release())));
119  assert(nullptr != dynamic_cast<T*>(resPtr.get()) && "Invalid target node!");
120  return std::unique_ptr<T>(dynamic_cast<T*>(resPtr.release()));
121  }
122 };
