51 class RamIntrinsicOperator : public RamExpression {
52 public:
53  template <typename... Args>
54  RamIntrinsicOperator(FunctorOp op, Args... args) : RamExpression(), operation(op) {
55  std::unique_ptr<RamExpression> tmp[] = {std::move(args)...};
56  for (auto& cur : tmp) {
57  arguments.push_back(std::move(cur));
58  }
59  }
60 
61  RamIntrinsicOperator(FunctorOp op, std::vector<std::unique_ptr<RamExpression>> args)
62  : RamExpression(), operation(op), arguments(std::move(args)) {}
63 
64  /** Print */
65  void print(std::ostream& os) const override {
66  if (isInfixFunctorOp(operation)) {
67  os << "(";
68  os << join(arguments, getSymbolForFunctorOp(operation),
69  print_deref<std::unique_ptr<RamExpression>>());
70  os << ")";
71  } else {
72  os << getSymbolForFunctorOp(operation);
73  os << "(";
74  os << join(arguments, ",", print_deref<std::unique_ptr<RamExpression>>());
75  os << ")";
76  }
77  }
78

80  FunctorOp getOperator() const {
81  return operation;
82  }

85  std::vector<RamExpression*> getArguments() const {
86  return toPtrVector(arguments);
87  }
88

90  const RamExpression* getArgument(size_t i) const {
91  assert(i >= 0 && i < arguments.size() && "argument index out of bounds");
92  return arguments[i].get();
93  }
94 
95  /** Get number of arguments */
96  size_t getArgCount() const {
97  return arguments.size();
98  }
99 
100  /** Obtain list of child nodes */
101  std::vector<const RamNode*> getChildNodes() const override {
102  std::vector<const RamNode*> res;
103  for (const auto& cur : arguments) {
104  res.push_back(cur.get());
105  }
106  return res;
107  }
108 
109  /* Clone */
110  RamIntrinsicOperator* clone() const override {
111  std::vector<std::unique_ptr<RamExpression>> argsCopy;
112  for (auto& arg : arguments) {
113  argsCopy.emplace_back(arg->clone());
114  }
115  auto res = new RamIntrinsicOperator(operation, std::move(argsCopy));
116  return res;
117  }
118 
119  /** Apply mapper */
120  void apply(const RamNodeMapper& map) override {
121  for (auto& arg : arguments) {
122  arg = map(std::move(arg));
123  }
124  }
125 
126 protected:
127  /** Operation symbol */
128  const FunctorOp operation;
129 
130  /** Arguments of the function */
131  std::vector<std::unique_ptr<RamExpression>> arguments;
132 
133  /** Check equality */
134  bool equal(const RamNode& node) const override {
135  assert(nullptr != dynamic_cast<const RamIntrinsicOperator*>(&node));
136  const auto& other = static_cast<const RamIntrinsicOperator&>(node);
137  return getOperator() == other.getOperator() && equal_targets(arguments, other.arguments);
138  }
