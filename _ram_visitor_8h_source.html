246 /**
247  * A utility function visiting all nodes within the RAM fragment rooted by the given node
248  * recursively in a depth-first pre-order fashion applying the given visitor to each
249  * encountered node.
250  *
251  * @param root the root of the RAM fragment to be visited
252  * @param visitor the visitor to be applied on each node
253  * @param args a list of extra parameters to be forwarded to the visitor
254  */
255 template <typename R, typename... Ps, typename... Args>
256 void visitDepthFirstPreOrder(const RamNode& root, RamVisitor<R, Ps...>& visitor, Args&... args) {
257  visitor(root, args...);
258  for (const RamNode* cur : root.getChildNodes()) {
259  if (cur != nullptr) {
260  visitDepthFirstPreOrder(*cur, visitor, args...);
261  }
262  }
263 }

283  * for visitor convenience functions.

285 template <typename R, typename N>
286 struct LambdaRamVisitor : public RamVisitor<void> {
287  std::function<R(const N&)> lambda;
288  LambdaRamVisitor(std::function<R(const N&)> lambda) : lambda(std::move(lambda)) {}
289  void visit(const RamNode& node) override {
290  if (const auto* n = dynamic_cast<const N*>(&node)) {
291  lambda(*n);
292  }
293  }
294 };
