52 /**
53  * An implementation of a meet operation between sets of types computing
54  * the set of pair-wise greatest common subtypes.
55  */
56 struct sub_type {
57  bool operator()(TypeSet& a, const TypeSet& b) const {
58  // compute result set
59  TypeSet res = getGreatestCommonSubtypes(a, b);
60 
61  // check whether a should change
62  if (res == a) {
63  return false;
64  }
65 
66  // update a
67  a = res;
68  return true;
69  }
70 };

104  * A constraint factory ensuring that all the types associated to the variable
105  * a are subtypes of type b.
106  */
107 TypeConstraint isSubtypeOf(const TypeVar& a, const Type& b) {
108  struct C : public Constraint<TypeVar> {
109  TypeVar a;
110  const Type& b;
111 
112  C(TypeVar a, const Type& b) : a(std::move(a)), b(b) {}
113 
114  bool update(Assignment<TypeVar>& ass) const override {
115  // get current value of variable a
116  TypeSet& s = ass[a];
117 
118  // remove all types that are not sub-types of b
119  if (s.isAll()) {
120  s = TypeSet(b);
121  return true;
122  }
123 
124  TypeSet res;
125  for (const Type& t : s) {
126  res.insert(getGreatestCommonSubtypes(t, b));
127  }
128 
129  // check whether there was a change
130  if (res == s) {
131  return false;
132  }
133  s = res;
134  return true;
135  }
136 
137  void print(std::ostream& out) const override {
138  out << a << " <: " << b.getName();
139  }
140  };
141 
142  return std::make_shared<C>(a, b);
143 }

158  // don't continually update super type constraints

194 TypeConstraint isSubtypeOfComponent(const TypeVar& a, const TypeVar& b, int index) {
195  struct C : public Constraint<TypeVar> {
196  TypeVar a;
197  TypeVar b;
198  unsigned index;
199 
200  C(TypeVar a, TypeVar b, int index) : a(std::move(a)), b(std::move(b)), index(index) {}
201 
202  bool update(Assignment<TypeVar>& ass) const override {
203  // get list of types for b
204  const TypeSet& recs = ass[b];
205 
206  // if it is (not yet) constrainted => skip
207  if (recs.isAll()) {
208  return false;
209  }
210 
211  // compute new types for a and b
212  TypeSet typesA;
213  TypeSet typesB;
214 
215  // iterate through types of b
216  for (const Type& t : recs) {
217  // only retain records
218  if (!isRecordType(t)) {
219  continue;
220  }
221  const auto& rec = static_cast<const RecordType&>(t);
222 
223  // of proper size
224  if (rec.getFields().size() <= index) {
225  continue;
226  }
227 
228  // this is a valid type for b
229  typesB.insert(rec);
230 
231  // and its corresponding field for a
232  typesA.insert(rec.getFields()[index].type);
233  }
234 
235  // combine with current types assigned to a
236  typesA = getGreatestCommonSubtypes(ass[a], typesA);
237 
238  // update values
239  bool changed = false;
240  if (recs != typesB) {
241  ass[b] = typesB;
242  changed = true;
243  }
244 
245  if (ass[a] != typesA) {
246  ass[a] = typesA;
247  changed = true;
248  }
249 
250  // done
251  return changed;
252  }
253 
254  void print(std::ostream& out) const override {
255  out << a << " <: " << b << "::" << index;
256  }
257  };
258 
259  return std::make_shared<C>(a, b, index);
260 }

264 AstClause* createAnnotatedClause(
265  const AstClause* clause, const std::map<const AstArgument*, TypeSet> argumentTypes) {
266  // Annotates each variable with its type based on a given type analysis result
267  struct TypeAnnotator : public AstNodeMapper {
268  const std::map<const AstArgument*, TypeSet>& types;
269 
270  TypeAnnotator(const std::map<const AstArgument*, TypeSet>& types) : types(types) {}
271 
272  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
273  if (auto* var = dynamic_cast<AstVariable*>(node.get())) {
274  std::stringstream newVarName;
275  newVarName << var->getName() << "&isin;" << types.find(var)->second;
276  return std::make_unique<AstVariable>(newVarName.str());
277  } else if (auto* var = dynamic_cast<AstUnnamedVariable*>(node.get())) {
278  std::stringstream newVarName;
279  newVarName << "_"
280  << "&isin;" << types.find(var)->second;
281  return std::make_unique<AstVariable>(newVarName.str());
282  }

284  return node;
285  }

288  /* Note:
289  * Because the type of each argument is stored in the form [address -> type-set],
290  * the type-analysis result does not immediately apply to the clone due to differing
291  * addresses.
292  * Two ways around this:
293  * (1) Perform the type-analysis again for the cloned clause
294  * (2) Keep track of the addresses of equivalent arguments in the cloned clause
295  * Method (2) was chosen to avoid having to recompute the analysis each time.
296  */
297  AstClause* annotatedClause = clause->clone();
298 
299  // Maps x -> y, where x is the address of an argument in the original clause, and y
300  // is the address of the equivalent argument in the clone.
301  std::map<const AstArgument*, const AstArgument*> memoryMap;
302 
303  std::vector<const AstArgument*> originalAddresses;
304  visitDepthFirst(*clause, [&](const AstArgument& arg) { originalAddresses.push_back(&arg); });
305 
306  std::vector<const AstArgument*> cloneAddresses;
307  visitDepthFirst(*annotatedClause, [&](const AstArgument& arg) { cloneAddresses.push_back(&arg); });
308 
309  assert(cloneAddresses.size() == originalAddresses.size());
310 
311  for (size_t i = 0; i < originalAddresses.size(); i++) {
312  memoryMap[originalAddresses[i]] = cloneAddresses[i];
313  }
314 
315  // Map the types to the clause clone
316  std::map<const AstArgument*, TypeSet> cloneArgumentTypes;
317  for (auto& pair : argumentTypes) {
318  cloneArgumentTypes[memoryMap[pair.first]] = pair.second;
319  }
320 
321  // Create the type-annotated clause
322  TypeAnnotator annotator(cloneArgumentTypes);
323  annotatedClause->apply(annotator);
324  return annotatedClause;
325 }

327 void TypeAnalysis::run(const AstTranslationUnit& translationUnit) {
328  // Check if debugging information is being generated and note where logs should be sent
329  std::ostream* debugStream = nullptr;
330  if (!Global::config().get("debug-report").empty()) {
331  debugStream = &analysisLogs;
332  }
333  auto* typeEnvAnalysis = translationUnit.getAnalysis<TypeEnvironmentAnalysis>();
334  for (const AstRelation* rel : translationUnit.getProgram()->getRelations()) {
335  for (const AstClause* clause : rel->getClauses()) {
336  // Perform the type analysis
337  std::map<const AstArgument*, TypeSet> clauseArgumentTypes =
338  analyseTypes(typeEnvAnalysis->getTypeEnvironment(), *clause, translationUnit.getProgram(),
339  debugStream);
340  argumentTypes.insert(clauseArgumentTypes.begin(), clauseArgumentTypes.end());
341 
342  if (debugStream != nullptr) {
343  // Store an annotated clause for printing purposes
344  AstClause* annotatedClause = createAnnotatedClause(clause, clauseArgumentTypes);
345  annotatedClauses.emplace_back(annotatedClause);
346  }
347  }
348  }

364 std::map<const AstArgument*, TypeSet> TypeAnalysis::analyseTypes(
365  const TypeEnvironment& env, const AstClause& clause, const AstProgram* program, std::ostream* logs) {
366  struct Analysis : public AstConstraintAnalysis<TypeVar> {
367  const TypeEnvironment& env;
368  const AstProgram* program;
369  std::set<const AstAtom*> negated;
370 
371  Analysis(const TypeEnvironment& env, const AstProgram* program) : env(env), program(program) {}
372 
373  // predicate
374  void visitAtom(const AstAtom& atom) override {
375  // get relation
376  auto rel = getAtomRelation(&atom, program);
377  if (!rel) {
378  return; // error in input program
379  }
380 
381  auto atts = rel->getAttributes();
382  auto args = atom.getArguments();
383  if (atts.size() != args.size()) {
384  return; // error in input program
385  }
386 
387  // set upper boundary of argument types
388  for (unsigned i = 0; i < atts.size(); i++) {
389  const auto& typeName = atts[i]->getTypeName();
390  if (env.isType(typeName)) {
391  // check whether atom is not negated
392  if (negated.find(&atom) == negated.end()) {
393  addConstraint(isSubtypeOf(getVar(args[i]), env.getType(typeName)));
394  } else {
395  addConstraint(isSupertypeOf(getVar(args[i]), env.getType(typeName)));
396  }
397  }
398  }
399  }
400 
401  // negations need to be skipped
402  void visitNegation(const AstNegation& cur) override {
403  // add nested atom to black-list
404  negated.insert(cur.getAtom());
405  }
406 
407  // symbol
408  void visitStringConstant(const AstStringConstant& cnst) override {
409  // this type has to be a sub-type of symbol
410  addConstraint(isSubtypeOf(getVar(cnst), env.getSymbolType()));
411  }
412 
413  // number
414  void visitNumberConstant(const AstNumberConstant& cnst) override {
415  // this type has to be a sub-type of number
416  addConstraint(isSubtypeOf(getVar(cnst), env.getNumberType()));
417  }
418 
419  // binary constraint
420  void visitBinaryConstraint(const AstBinaryConstraint& rel) override {
421  auto lhs = getVar(rel.getLHS());
422  auto rhs = getVar(rel.getRHS());
423  addConstraint(isSubtypeOf(lhs, rhs));
424  addConstraint(isSubtypeOf(rhs, lhs));
425  }
426 
427  // intrinsic functor
428  void visitIntrinsicFunctor(const AstIntrinsicFunctor& fun) override {
429  auto cur = getVar(fun);
430 
431  // add a constraint for the return type of the functor
432  if (fun.isNumerical()) {
433  addConstraint(isSubtypeOf(cur, env.getNumberType()));
434  }
435  if (fun.isSymbolic()) {
436  addConstraint(isSubtypeOf(cur, env.getSymbolType()));
437  }
438 
439  // add a constraint for each argument of the functor
440  if (fun.getFunction() == FunctorOp::ORD) {
441  return;
442  }
443 
444  for (size_t i = 0; i < fun.getArity(); i++) {
445  auto arg = getVar(fun.getArg(i));
446  if (fun.acceptsNumbers(i)) {
447  addConstraint(isSubtypeOf(arg, env.getNumberType()));
448  }
449  if (fun.acceptsSymbols(i)) {
450  addConstraint(isSubtypeOf(arg, env.getSymbolType()));
451  }
452  }
453  }
454 
455  // user-defined functors
456  void visitUserDefinedFunctor(const AstUserDefinedFunctor& fun) override {
457  auto cur = getVar(fun);
458 
459  // get functor declaration
460  const AstFunctorDeclaration* funDecl = program->getFunctorDeclaration(fun.getName());
461  // check whether functor declaration exists
462  if (funDecl != nullptr) {
463  // add a constraint for the return type
464  if (funDecl->isNumerical()) {
465  addConstraint(isSubtypeOf(cur, env.getNumberType()));
466  }
467  if (funDecl->isSymbolic()) {
468  addConstraint(isSubtypeOf(cur, env.getSymbolType()));
469  }
470 
471  // add constraints for arguments
472  for (size_t i = 0; i < fun.getArgCount(); i++) {
473  auto arg = getVar(fun.getArg(i));
474 
475  // check that usage does not exceed
476  // number of arguments in declaration
477  if (i < funDecl->getArgCount()) {
478  // add constraints for the i-th argument
479  if (funDecl->acceptsNumbers(i)) {
480  addConstraint(isSubtypeOf(arg, env.getNumberType()));
481  }
482  if (funDecl->acceptsSymbols(i)) {
483  addConstraint(isSubtypeOf(arg, env.getSymbolType()));
484  }
485  }
486  }
487  }
488  }
489 
490  // counter
491  void visitCounter(const AstCounter& counter) override {
492  // this value must be a number value
493  addConstraint(isSubtypeOf(getVar(counter), env.getNumberType()));
494  }
495 
496  // components of records
497  void visitRecordInit(const AstRecordInit& init) override {
498  // link element types with sub-values
499  auto rec = getVar(init);
500  int i = 0;
501 
502  for (const AstArgument* value : init.getArguments()) {
503  addConstraint(isSubtypeOfComponent(getVar(value), rec, i++));
504  }
505  }
506 
507  // visit aggregates
508  void visitAggregator(const AstAggregator& agg) override {
509  // this value must be a number value
510  addConstraint(isSubtypeOf(getVar(agg), env.getNumberType()));
511 
512  // also, the target expression needs to be a number
513  if (auto expr = agg.getTargetExpression()) {
514  addConstraint(isSubtypeOf(getVar(expr), env.getNumberType()));
515  }
516  }
517  };
518 
519  // run analysis
520  return Analysis(env, program).analyse(clause, logs);
521 }
