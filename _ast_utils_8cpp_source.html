57 std::set<const AstRelation*> getBodyRelations(const AstClause* clause, const AstProgram* program) {
58  std::set<const AstRelation*> bodyRelations;
59  for (const auto& lit : clause->getBodyLiterals()) {
60  visitDepthFirst(
61  *lit, [&](const AstAtom& atom) { bodyRelations.insert(getAtomRelation(&atom, program)); });
62  }
63  for (const auto& arg : clause->getHead()->getArguments()) {
64  visitDepthFirst(
65  *arg, [&](const AstAtom& atom) { bodyRelations.insert(getAtomRelation(&atom, program)); });
66  }
