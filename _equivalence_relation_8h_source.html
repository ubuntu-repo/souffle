86  /**
87  * inserts all nodes from the other relation into this one
88  * @param other the binary relation from which to add elements from

90  void insertAll(const EquivalenceRelation<TupleType>& other) {
91  other.genAllDisjointSetLists();
92 
93  // iterate over partitions at a time
94  for (typename StatesMap::chunk it : other.equivalencePartition.getChunks(MAX_THREADS)) {
95  for (auto& p : it) {
96  value_type rep = p.first;

104  // invalidate iterators unconditionally
105  this->statesMapStale.store(true, std::memory_order_relaxed);
106  }

108  /**
109  * Extend this relation with another relation, expanding this equivalence relation
110  * The supplied relation is the old knowledge, whilst this relation only contains
111  * explicitly new knowledge. After this operation the "implicitly new tuples" are now
112  * explicitly inserted this relation.
113  */
114  void extend(const EquivalenceRelation<TupleType>& other) {
115  // nothing to extend if there's no new/original knowledge
116  if (other.size() == 0 || this->size() == 0) return;
117 
118  this->genAllDisjointSetLists();
119  other.genAllDisjointSetLists();
120 
121  std::set<value_type> repsCovered;
122 
123  // find all the disjoint sets that need to be added to this relation
124  // that exist in other (and exist in this)
125  {
126  auto it = this->sds.sparseToDenseMap.begin();
127  auto end = this->sds.sparseToDenseMap.end();
128  value_type el;
129  for (; it != end; ++it) {
130  std::tie(el, std::ignore) = *it;
131  if (other.containsElement(el)) {
132  value_type rep = other.sds.findNode(el);
133  if (repsCovered.count(rep) == 0) {
134  repsCovered.emplace(rep);
135  }
136  }
137  }
138  }
139 
140  // add the intersecting dj sets into this one
141  {
142  value_type el;
143  value_type rep;
144  auto it = other.sds.sparseToDenseMap.begin();
145  auto end = other.sds.sparseToDenseMap.end();
146  for (; it != end; ++it) {
147  std::tie(el, std::ignore) = *it;
148  rep = other.sds.findNode(el);
149  if (repsCovered.count(rep) != 0) {
150  this->insert(el, rep);
151  }
152  }
153  }
154  }

538  /**
539  * Generate an approximate number of iterators for parallel iteration
540  * The iterators returned are not necessarily equal in size, but in practise are approximately similarly
541  * sized
542  * Depending on the structure of the data, there can be more or less partitions returned than requested.
543  * @param chunks the number of requested partitions
544  * @return a list of the iterators as ranges
545  */
546  std::vector<souffle::range<iterator>> partition(size_t chunks) const {
547  // generate all reps
548  genAllDisjointSetLists();
549 
550  size_t numPairs = this->size();
551  if (numPairs == 0) return {};
552  if (numPairs == 1 || chunks <= 1) return {souffle::make_range(begin(), end())};
553 
554  // if there's more dj sets than requested chunks, then just return an iter per dj set
555  std::vector<souffle::range<iterator>> ret;
556  if (chunks <= equivalencePartition.size()) {
557  for (auto& p : equivalencePartition) {
558  ret.push_back(souffle::make_range(closure(p.first), end()));
559  }
560  return ret;
561  }
562 
563  // keep it simple stupid
564  // just go through and if the size of the binrel is > numpairs/chunks, then generate an anteriorIt for
565  // each
566  const size_t perchunk = numPairs / chunks;
567  for (const auto& itp : equivalencePartition) {
568  const size_t s = itp.second->size();
569  if (s * s > perchunk) {
570  for (const auto& i : *itp.second) {
571  ret.push_back(souffle::make_range(anteriorIt(i), end()));
572  }
573  } else {
574  ret.push_back(souffle::make_range(closure(itp.first), end()));
575  }
576  }
577 
578  return ret;
579  }

608  /**
609  * Generate a cache of the sets such that they can be iterated over efficiently.
610  * Each set is partitioned into a PiggyList.
611  */
612  void genAllDisjointSetLists() const {
613  statesLock.lock();
614 
615  // no need to generate again, already done.
616  if (!this->statesMapStale.load(std::memory_order_acquire)) {
617  statesLock.unlock();
618  return;
619  }
620 
621  // btree version
622  emptyPartition();
623 
624  size_t dSetSize = this->sds.ds.a_blocks.size();
625  for (size_t i = 0; i < dSetSize; ++i) {
626  typename TupleType::value_type sparseVal = this->sds.toSparse(i);
627  parent_t rep = this->sds.findNode(sparseVal);
628 
629  StorePair p = {rep, nullptr};
630  StatesList* mapList = equivalencePartition.insert(p, [&](StorePair& sp) {
631  auto* r = new StatesList(1);
632  sp.second = r;
633  return r;
634  });
635  mapList->append(sparseVal);
636  }
637 
638  statesMapStale.store(false, std::memory_order_release);
639  statesLock.unlock();
640  }
