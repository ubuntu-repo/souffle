169 /* Print program in textual format */
170 void AstProgram::print(std::ostream& os) const {
171  /* Print types */
172  os << "// ----- Types -----\n";
173  for (const auto& cur : types) {
174  os << *cur.second << "\n";
175  }
176 
177  /* Print components */
178  if (!components.empty()) {
179  os << "\n// ----- Components -----\n";
180  for (const auto& cur : components) {
181  os << *cur << "\n";
182  }
183  }
184 
185  /* Print instantiations */
186  if (!instantiations.empty()) {
187  os << "\n";
188  for (const auto& cur : instantiations) {
189  os << *cur << "\n";
190  }
191  }
192 
193  /* Print functors */
194  os << "\n// ----- Functors -----\n";
195  for (const auto& cur : functors) {
196  const std::unique_ptr<AstFunctorDeclaration>& f = cur.second;
197  os << "\n\n// -- " << f->getName() << " --\n";
198  f->print(os);
199  os << "\n";
200  }
201 
202  /* Print relations */
203  os << "\n// ----- Relations -----\n";
204  for (const auto& cur : relations) {
205  const std::unique_ptr<AstRelation>& rel = cur.second;
206  os << "\n\n// -- " << rel->getName() << " --\n";
207  os << *rel << "\n\n";
208  for (const auto clause : rel->getClauses()) {
209  os << *clause << "\n\n";
210  }
211  for (const auto ioDirective : rel->getLoads()) {
212  os << *ioDirective << "\n\n";
213  }
214  for (const auto ioDirective : rel->getStores()) {
215  os << *ioDirective << "\n\n";
216  }
217  }
218 
219  if (!clauses.empty()) {
220  os << "\n// ----- Orphan Clauses -----\n";
221  os << join(clauses, "\n\n", print_deref<std::unique_ptr<AstClause>>()) << "\n";
222  }
223 
224  if (!loads.empty()) {
225  os << "\n// ----- Orphan Load directives -----\n";
226  os << join(loads, "\n\n", print_deref<std::unique_ptr<AstLoad>>()) << "\n";
227  }
228  if (!stores.empty()) {
229  os << "\n// ----- Orphan Store directives -----\n";
230  os << join(stores, "\n\n", print_deref<std::unique_ptr<AstStore>>()) << "\n";
231  }
232 
233  if (!pragmaDirectives.empty()) {
234  os << "\n// ----- Pragma -----\n";
235  for (const auto& cur : pragmaDirectives) {
236  os << *cur << "\n";
237  }
238  }

241 AstProgram* AstProgram::clone() const {
242  // create copy
243  auto res = new AstProgram();
244 
245  // move types
246  for (const auto& cur : types) {
247  res->types.insert(std::make_pair(cur.first, std::unique_ptr<AstType>(cur.second->clone())));
248  }
249 
250  // move relations
251  for (const auto& cur : relations) {
252  res->relations.insert(std::make_pair(cur.first, std::unique_ptr<AstRelation>(cur.second->clone())));
253  }
254 
255  // move components
256  for (const auto& cur : components) {
257  res->components.emplace_back(cur->clone());
258  }
259 
260  // move component instantiations
261  for (const auto& cur : instantiations) {
262  res->instantiations.emplace_back(cur->clone());
263  }
264 
265  // move ioDirectives
266  for (const auto& cur : loads) {
267  res->loads.emplace_back(cur->clone());
268  }
269  for (const auto& cur : stores) {
270  res->stores.emplace_back(cur->clone());
271  }
272 
273  ErrorReport errors;
274 
275  res->finishParsing();
276 
277  // done
278  return res;

306 void AstProgram::finishParsing() {
307  // unbound clauses with no relation defined
308  std::vector<std::unique_ptr<AstClause>> unbound;
309 
310  // add clauses
311  for (auto& cur : clauses) {
312  auto pos = relations.find(cur->getHead()->getName());
313  if (pos != relations.end()) {
314  pos->second->addClause(std::move(cur));
315  } else {
316  unbound.push_back(std::move(cur));
317  }
318  }
319  // remember the remaining orphan clauses
320  clauses.clear();
321  clauses.swap(unbound);
322 
323  // unbound directives with no relation defined
324  std::vector<std::unique_ptr<AstLoad>> unboundLoads;
325  std::vector<std::unique_ptr<AstStore>> unboundStores;
326 
327  // add IO directives
328  for (auto& cur : loads) {
329  auto pos = relations.find(cur->getName());
330  if (pos != relations.end()) {
331  pos->second->addLoad(std::move(cur));
332  } else {
333  unboundLoads.push_back(std::move(cur));
334  }
335  }
336  // remember the remaining orphan directives
337  loads.clear();
338  loads.swap(unboundLoads);
339 
340  for (auto& cur : stores) {
341  auto pos = relations.find(cur->getName());
342  if (pos != relations.end()) {
343  pos->second->addStore(std::move(cur));
344  } else {
345  unboundStores.push_back(std::move(cur));
346  }
347  }
348  // remember the remaining orphan directives
349  stores.clear();
350  stores.swap(unboundStores);
351 }
