101  friend std::ostream& operator<<(std::ostream& out, const AdornedClause& arg) {
102  size_t currpos = 0;
103  bool firstadded = true;
104  out << arg.clause->getHead()->getName() << "{" << arg.headAdornment << "} :- ";
105 
106  std::vector<AstLiteral*> bodyLiterals = arg.clause->getBodyLiterals();
107  for (AstLiteral* lit : bodyLiterals) {
108  if (dynamic_cast<AstAtom*>(lit) == nullptr) {
109  const AstAtom* corresAtom = lit->getAtom();
110  if (corresAtom != nullptr) {
111  if (firstadded) {
112  firstadded = false;
113  out << corresAtom->getName() << "{_}";
114  } else {
115  out << ", " << corresAtom->getName() << "{_}";
116  }
117  } else {
118  continue;
119  }
120  } else {
121  if (firstadded) {
122  firstadded = false;
123  out << lit->getAtom()->getName() << "{" << arg.bodyAdornment[currpos] << "}";
124  } else {
125  out << ", " << lit->getAtom()->getName() << "{" << arg.bodyAdornment[currpos] << "}";
126  }
