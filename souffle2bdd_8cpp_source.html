96  /**
97  * The entry point for the conversion of a program, converting the basic top-level structure.
98  */
99  void visitProgram(const AstProgram& program, std::ostream& out) override {
100  // type definition
101  out << "N " << std::numeric_limits<RamDomain>::max() << "\n\n";
102 
103  // variable order
104  int max_attributes = 0;
105  for (const auto& rel : program.getRelations()) {
106  max_attributes = std::max<int>(max_attributes, rel->getAttributes().size());
107  }
108  out << ".bddvarorder ";
109  for (int i = 0; i < max_attributes; i++) {
110  out << "N" << i;
111  if (i + 1 != max_attributes) {
112  out << "_";
113  }
114  }
115  out << "\n\n";
116 
117  // declarations
118  for (const auto& rel : program.getRelations()) {
119  // process the relation declaration
120  visit(*rel, out);
121  }
122  out << "\n";
123 
124  // rules
125  for (const auto& rel : program.getRelations()) {
126  for (const auto& clause : rel->getClauses()) {
127  visit(*clause, out);
128  }
129  }
130  out << "\n";
131  }
132 
133  /**
134  * Converting a relation by creating its declaration.
135  */
136  void visitRelation(const AstRelation& rel, std::ostream& out) override {
137  visitRelationIdentifier(rel.getName(), out);
138 
139  // make nullary relations single-element relations
140  out << "(";
141  if (rel.getAttributes().empty()) {
142  out << "dummy:N0";
143  }
144  int i = 0;
145  out << join(rel.getAttributes(), ",", [&](std::ostream& out, AstAttribute* cur) {
146  out << cur->getAttributeName() << ":N" << (i++);
147  });
148  out << ")";
149 
150  if (ioTypes->isInput(&rel)) {
151  out << " inputtuples";
152  }
153  if (ioTypes->isOutput(&rel)) {
154  out << " outputtuples";
155  }
156 
157  out << "\n";
158  }
159 
160  /**
161  * Converting a clause.
162  */
163  void visitClause(const AstClause& clause, std::ostream& out) override {
164  visit(*clause.getHead(), out);
165 
166  // if it is a fact, that's it
167  if (clause.isFact()) {
168  // there must not be any additons
169  if (!extra_literals.empty()) {
170  throw UnsupportedConstructException("Unsupported fact: " + toString(clause));
171  }
172  out << ".\n";
173  return;
174  }
175 
176  // convert the body
177  out << " :- ";
178  out << join(
179  clause.getBodyLiterals(), ",", [&](std::ostream& out, AstLiteral* cur) { visit(*cur, out); });
180 
181  // add extra_literals
182  for (const auto& cur : extra_literals) {
183  out << "," << cur;
184  }
185  extra_literals.clear();
186 
187  // done
188  out << ".\n";
189  }
190 
191  void visitAtom(const AstAtom& atom, std::ostream& out) override {
192  visitRelationIdentifier(atom.getName(), out);
193 
194  // since no nullary-relations are allowed, we add a dummy value
195  if (atom.getArguments().empty()) {
196  out << "(0)";
197  return;
198  }
199 
200  out << "(";
201  out << join(atom.getArguments(), ",", [&](std::ostream& out, AstArgument* cur) { visit(*cur, out); });
202  out << ")";
203  }

229  void visitFunctor(const AstFunctor& fun, std::ostream& out) override {
230  // create a new variable
231  std::string var = "aux_var_" + toString(varCounter++);
232 
233  // write variable
234  out << var;
235 
236  // assign variable in extra clause
237  std::stringstream binding;
238  binding << var << "=";
239 
240  // only intrinsic binary operators supported
241  const auto* inf = dynamic_cast<const AstIntrinsicFunctor*>(&fun);
242  if (inf != nullptr && inf->getArity() == 2) {
243  visit(*inf->getArg(0), binding);
244  binding << getSymbolForFunctorOp(inf->getFunction());
245  visit(*inf->getArg(1), binding);
246  } else {
247  throw UnsupportedConstructException("Unsupported function: " + toString(fun));
248  }
249 
250  extra_literals.push_back(binding.str());
251  }
252 
253  void visitVariable(const AstVariable& var, std::ostream& out) override {
254  out << var;
255  }

258  out << "_";

268  }
