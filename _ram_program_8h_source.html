109  /** Create clone */
110  RamProgram* clone() const override {
111  std::map<const RamRelation*, const RamRelation*> refMap;
112  RamProgram* res = new RamProgram(std::unique_ptr<RamStatement>(main->clone()));
113  for (auto& cur : relations) {
114  RamRelation* newRel = cur.second->clone();
115  refMap[cur.second.get()] = newRel;
116  res->addRelation(std::unique_ptr<RamRelation>(newRel));
117  }
118  for (auto& cur : subroutines) {
119  res->addSubroutine(cur.first, std::unique_ptr<RamStatement>(cur.second->clone()));
120  }
121  res->apply(makeLambdaRamMapper([&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
122  // check whether it is a unnamed variable
123  if (const RamRelationReference* relRef = dynamic_cast<RamRelationReference*>(node.get())) {
124  const RamRelation* rel = refMap[relRef->get()];
125  assert(rel != nullptr && "dangling RAM relation reference");
126  return std::unique_ptr<RamRelationReference>(new RamRelationReference(rel));
127  } else {
128  return node;
129  }
130  }));

132  }

159  if (relations.size() != other.relations.size() || subroutines.size() != other.subroutines.size()) {
160  return false;
