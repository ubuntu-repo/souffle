46 void RuleBody::conjunct(RuleBody&& other) {

48 
49  for (const auto& clauseA : dnf) {
50  for (const auto& clauseB : other.dnf) {
51  // create a new clause in result
52  clause cur;
53 
54  // it is the concatenation of the two clauses
55  for (const auto& lit : clauseA) {

78 std::vector<AstClause*> RuleBody::toClauseBodies() const {
79  // collect clause results
80  std::vector<AstClause*> bodies;
81  for (const clause& cur : dnf) {
82  bodies.push_back(new AstClause());
83  AstClause& clause = *bodies.back();
84 
85  for (const literal& lit : cur) {
86  // extract literal
87  AstLiteral* base = lit.atom->clone();
88  // negate if necessary
89  if (lit.negated) {
90  // negate
91  if (auto* atom = dynamic_cast<AstAtom*>(base)) {
92  base = new AstNegation(std::unique_ptr<AstAtom>(atom));
93  base->setSrcLoc(atom->getSrcLoc());
94  } else if (auto* cstr = dynamic_cast<AstConstraint*>(base)) {
95  cstr->negate();
96  }
97  }
98 
99  // add to result
100  clause.addToBody(std::unique_ptr<AstLiteral>(base));
101  }
102  }
103 
104  // done
105  return bodies;
106 }

200 void RuleBody::insert(std::vector<clause>& cnf, clause&& cls) {
201  for (const auto& cur : cnf) {
202  if (isSubsetOf(cur, cls)) {
203  return;
204  }
205  }
206  std::vector<clause> res;
207  for (auto& cur : cnf) {
208  if (!isSubsetOf(cls, cur)) {
209  res.push_back(std::move(cur));
210  }
211  }
212  res.swap(cnf);
213  cnf.push_back(std::move(cls));
214 }
