530  if (aggregate.getFunction() == souffle::COUNT && aggregate.getCondition() == nullptr) {
531  code->push_back(LVM_Aggregate_COUNT);
532  code->push_back(LVM_ITER_TypeScan);
533  code->push_back(counterLabel);
534  } else {
535  // Init value
536  switch (aggregate.getFunction()) {
537  case souffle::MIN:
538  code->push_back(LVM_Number);
539  code->push_back(MAX_RAM_DOMAIN);
540  break;
541  case souffle::MAX:
542  code->push_back(LVM_Number);
543  code->push_back(MIN_RAM_DOMAIN);
544  break;
545  case souffle::COUNT:
546  code->push_back(LVM_Number);
547  code->push_back(0);
548  break;
549  case souffle::SUM:
550  code->push_back(LVM_Number);
551  code->push_back(0);
552  break;

556

558  code->push_back(LVM_ITER_NotAtEnd);
559  code->push_back(counterLabel);
560  code->push_back(LVM_ITER_TypeScan);
561  code->push_back(LVM_Jmpez);
562  code->push_back(lookupAddress(L1));
563 
564  // Select the element pointed by iter
565  code->push_back(LVM_ITER_Select);
566  code->push_back(counterLabel);
567  code->push_back(LVM_ITER_TypeScan);
568  code->push_back(aggregate.getTupleId());

576  }

579  visit(aggregate.getExpression(), exitAddress);
580  }
581 
582  switch (aggregate.getFunction()) {
583  case souffle::MIN:
584  code->push_back(LVM_OP_MIN);

586  break;
587  case souffle::MAX:
588  code->push_back(LVM_OP_MAX);
589  code->push_back(2);
590  break;
591  case souffle::COUNT:
592  code->push_back(LVM_Number);
593  code->push_back(1);
594  code->push_back(LVM_OP_ADD);
595  break;
596  case souffle::SUM:
597  code->push_back(LVM_OP_ADD);
598  break;
599  }
600  setAddress(endOfLoop, code->size());
601  code->push_back(LVM_ITER_Inc);
602  code->push_back(counterLabel);
603  code->push_back(LVM_ITER_TypeScan);

606  }
607

611  code->push_back(LVM_Aggregate_Return);
612  code->push_back(aggregate.getTupleId());
613

616 
617  // Retrieve the result we just saved.
618  code->push_back(LVM_ElementAccess);
619  code->push_back(aggregate.getTupleId());
620  code->push_back(0);
621  code->push_back(LVM_Number);
622 
623  code->push_back(aggregate.getFunction() == souffle::MIN ? MAX_RAM_DOMAIN : MIN_RAM_DOMAIN);
624  code->push_back(LVM_OP_EQ);
625  code->push_back(LVM_Jmpnz); // If init == result, does not visit nested search
626  code->push_back(lookupAddress(L2));
627  }
628  visitSearch(aggregate, exitAddress);
629  setAddress(L2, code->size());
630  }
