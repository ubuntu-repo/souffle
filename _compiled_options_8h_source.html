132  /**
133  * Parses the given command line parameters, handles -h help requests or errors
134  * and returns whether the parsing was successful or not.
135  */
136  bool parse(int argc, char** argv) {
137  // get executable name
138  std::string exec_name = "analysis";
139  if (argc > 0) {
140  exec_name = argv[0];
141  }
142 
143  // local options
144  std::string fact_dir = input_dir;
145  std::string out_dir = output_dir;
146 
147 // avoid warning due to Solaris getopt.h
148 #pragma GCC diagnostic push
149 #pragma GCC diagnostic ignored "-Wwrite-strings"
150  // long options
151  option longOptions[] = {{"facts", true, nullptr, 'F'}, {"output", true, nullptr, 'D'},
152  {"profile", true, nullptr, 'p'}, {"jobs", true, nullptr, 'j'}, {"index", true, nullptr, 'i'},
153  // the terminal option -- needs to be null
154  {nullptr, false, nullptr, 0}};
155 #pragma GCC diagnostic pop
156 
157  // check whether all options are fine
158  bool ok = true;
159 
160  int c; /* command-line arguments processing */
161  while ((c = getopt_long(argc, argv, "D:F:hp:j:i:", longOptions, nullptr)) != EOF) {
162  switch (c) {
163  /* Fact directories */
164  case 'F':
165  if (!existDir(optarg)) {
166  printf("Fact directory %s does not exists!\n", optarg);
167  ok = false;
168  }
169  fact_dir = optarg;
170  break;
171  /* Output directory for resulting .csv files */
172  case 'D':
173  if (*optarg && !existDir(optarg)) {
174  printf("Output directory %s does not exists!\n", optarg);
175  ok = false;
176  }
177  out_dir = optarg;
178  break;
179  case 'p':
180  if (!profiling) {
181  std::cerr << "\nError: profiling was not enabled in compilation\n\n";
182  printHelpPage(exec_name);
183  exit(1);
184  }
185  profile_name = optarg;
186  break;
187  case 'j':
188 #ifdef _OPENMP
189  if (std::string(optarg) == "auto") {
190  num_jobs = 0;
191  } else {
192  int num = atoi(optarg);
193  if (num > 0) {
194  num_jobs = num;
195  } else {
196  std::cerr << "Invalid number of jobs [-j]: " << optarg << "\n";
197  ok = false;
198  }
199  }
200 #else
201  std::cerr << "\nWarning: OpenMP was not enabled in compilation\n\n";
202 #endif
203  break;
204  case 'i':
205  stratumIndex = (size_t)std::stoull(optarg);
206  break;
207  default:
208  printHelpPage(exec_name);
209  return false;
210  }
211  }
212 
213  // update member fields
214  input_dir = fact_dir;
215  output_dir = out_dir;
216 
217 #ifdef _OPENMP
218  if (num_jobs > 0) {
219  omp_set_num_threads(num_jobs);
220  }
221 #endif
222 
223  // return success state
224  return ok;
225  }
