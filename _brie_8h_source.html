82 /**
83  * A sparse array simulates an array associating to every element
84  * of uint32_t an element of a generic type T. Any non-defined element
85  * will be default-initialized utilizing the detail::default_factory
86  * functor.
87  *
88  * Internally the array is organized as a balanced tree. The leaf
89  * level of the tree corresponds to the elements of the represented
90  * array. Inner nodes utilize individual bits of the indices to reference
91  * sub-trees. For efficiency reasons, only the minimal sub-tree required
92  * to cover all non-null / non-default values stored in the array is
93  * maintained. Furthermore, several levels of nodes are aggreated in a
94  * B-tree like fashion to inprove cache utilization and reduce the number
95  * of steps required for lookup and insert operations.
96  *
97  * @tparam T the type of the stored elements
98  * @tparam BITS the number of bits consumed per node-level
99  * e.g. if it is set to 3, the resulting tree will be of a degree of
100  * 2^3=8, and thus 8 child-pointers will be stored in each inner node
101  * and as many values will be stored in each leaf node.
102  * @tparam merge_op the functor to be utilized when merging the content of two
103  * instances of this type.
104  * @tparam copy_op a functor to be applied to each stored value when copying an
105  * instance of this array. For instance, this is utilized by the
106  * trie implementation to create a clone of each sub-tree instead
107  * of preserving the original pointer.
108  */

386  RootInfoSnapshot getRootInfo() const {
387  RootInfoSnapshot res;
388  do {

390  do {

392  res.version = getRootVersion();

407  /**
408  * Updates the current root information based on the handed in modified
409  * snapshot instance if the version number of the snapshot still corresponds
410  * to the current version. Otherwise a concurrent update took place and the
411  * operation is aborted.
412  *
413  * @param info the updated information to be assigned to the active root-info data
414  * @return true if successfully updated, false if aborted
415  */
416  bool tryUpdateRootInfo(const RootInfoSnapshot& info) {
417  // check mod counter
418  uintptr_t version = info.version;
419 
420  // update root to invalid pointer (ending with 1)
421  if (!__sync_bool_compare_and_swap(&synced.root, (Node*)version, (Node*)(version + 1))) {
422  return false;
423  }
424 
425  // conduct update
426  synced.levels = info.levels;
427  synced.offset = info.offset;
428 
429  // update root (and thus the version to enable future retrievals)
430  __sync_synchronize();

549  /**

551  * If the cell does not exist yet it will be created as a side-effect.
552  *
553  * @param i the index of the requested cell
554  * @param ctxt a operation context to exploit state-less temporal locality
555  * @return a reference to the requested cell
556  */
557  inline Cell& getLeaf(index_type i, op_context& ctxt) {
558  // check context
559  if (ctxt.lastNode && (ctxt.lastIndex == (i & ~INDEX_MASK))) {
560  // return reference to referenced
561  return ctxt.lastNode->cell[i & INDEX_MASK];
562  }
563 
564  // get snapshot of root
565  auto info = getRootInfo();
566 
567  // check for emptiness
568  if (info.root == nullptr) {
569  // build new root node
570  info.root = newNode();
571 
572  // initialize the new node
573  info.root->parent = nullptr;
574  info.offset = i & ~(INDEX_MASK);
575 
576  // try updating root information atomically
577  if (tryUpdateRootInfo(info)) {
578  // success -- finish get call
579 
580  // update first

583  firstInfo.node = info.root;
584  firstInfo.offset = info.offset;

586  // there was some concurrent update => check again
587  firstInfo = getFirstInfo();
588  }
589  }
590

592  return info.root->cell[i & INDEX_MASK];
593  }

595  // somebody else was faster => use standard insertion procedure
596  delete info.root;
597 
598  // retrieve new root info
599  info = getRootInfo();
600 
601  // make sure there is a root
602  assert(info.root);
603  }

606  // - check boundary
607  // - navigate to node
608  // - insert value
609 
610  // check boundaries
611  while (!inBoundaries(i, info.levels, info.offset)) {

614  // update root info
615  info = getRootInfo();

621  while (level != 0) {
622  // get X coordinate
623  auto x = getIndex(i, level);
624

768  * @param trg a reference to the pointer the cloned node should be stored to
769  * @param src the node to be cloned
