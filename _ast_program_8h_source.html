171 protected:
172  /** Implements the node comparison for this node type */
173  bool equal(const AstNode& node) const override {
174  assert(nullptr != dynamic_cast<const AstProgram*>(&node));
175  const auto& other = static_cast<const AstProgram&>(node);
176 
177  // check list sizes
178  if (types.size() != other.types.size()) {
179  return false;
180  }
181  if (relations.size() != other.relations.size()) {
182  return false;
183  }
184 
185  // check types
186  for (const auto& cur : types) {
187  auto pos = other.types.find(cur.first);
188  if (pos == other.types.end()) {
189  return false;
190  }
191  if (*cur.second != *pos->second) {
192  return false;
193  }
194  }
195 
196  // check relations
197  for (const auto& cur : relations) {
198  auto pos = other.relations.find(cur.first);
199  if (pos == other.relations.end()) {
200  return false;
201  }
202  if (*cur.second != *pos->second) {
203  return false;
204  }
205  }
206 
207  // check components
208  if (!equal_targets(components, other.components)) {
209  return false;
210  }
211  if (!equal_targets(instantiations, other.instantiations)) {
212  return false;
213  }
214  if (!equal_targets(clauses, other.clauses)) {
215  return false;
216  }
217  if (!equal_targets(loads, other.loads)) {
218  return false;
219  }
220  if (!equal_targets(printSizes, other.printSizes)) {
221  return false;
222  }
223  if (!equal_targets(stores, other.stores)) {
224  return false;
225  }
226 
227  // no different found => programs are equal
228  return true;
229  }
