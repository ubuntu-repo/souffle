75 /**
76  * A bidirectional mapping between tuples and reference indices.
77  */
78 template <typename Tuple>
79 class RecordMap {
80  // create blocks of a million entries
81  static const std::size_t BLOCK_SIZE = 1 << 20;
82 
83  /** The definition of the tuple type handled by this instance */
84  using tuple_type = Tuple;
85 
86  /** The definition of the type of a tuple block */
87  using block_type = std::array<tuple_type, BLOCK_SIZE>;
88 
89  /** The type utilized for the block index */
90  using block_index_type = std::vector<std::unique_ptr<block_type>>;
91 
92  /** The mapping from tuples to references/indices */
93  std::unordered_map<tuple_type, RamDomain> r2i;
94 
95  /** The mapping from indices to tuples */
96  block_index_type i2r;
97 
98  /** a lock for the pack operation */
99  Lock pack_lock;
100 
101 public:
102  RecordMap() = default;
103 
104  /**
105  * Packs the given tuple -- and may create a new reference if necessary.
106  */
107  RamDomain pack(const tuple_type& tuple) {
108  RamDomain index;
109 
110  {
111  // lock pack operation
112  auto leas = pack_lock.acquire(); // lock hold till end of scope
113  (void)leas; // avoid warning
114 
115  // try lookup
116  auto pos = r2i.find(tuple);
117  if (pos != r2i.end()) {
118  // take the previously assigned value
119  index = pos->second;
120  } else {
121  // add tuple to index
122  index = r2i.size() + 1; // since 0 is skipped for the Nil element
123  r2i[tuple] = index;
124 
125  // assert that new index is smaller than the range
126  assert(index != std::numeric_limits<RamDomain>::max());
127 
128  if (index / BLOCK_SIZE == i2r.size()) {
129  i2r.push_back(std::make_unique<block_type>());
130  }
131 
132  // create entry for unpacking
133  auto& list = i2r[index / BLOCK_SIZE];
134 
135  // insert tuple
136  (*list)[index % BLOCK_SIZE] = tuple;
137  }
138  }
139 
140  // done
141  return index;
142  }
143 
144  /**
145  * Obtains a pointer to the tuple addressed by the given index.
146  */
147  const tuple_type& unpack(RamDomain index) {
148  // just look up the right spot
149  return (*(i2r[index / BLOCK_SIZE]))[index % BLOCK_SIZE];
150  }
151 };
