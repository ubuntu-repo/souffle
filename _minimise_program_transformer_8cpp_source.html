34 /**
35  * Extract valid permutations from a given permutation matrix of valid moves.
36  */
37 std::vector<std::vector<unsigned int>> extractPermutations(
38  const std::vector<std::vector<unsigned int>>& permutationMatrix) {
39  size_t clauseSize = permutationMatrix.size();
40  // keep track of the possible end-positions of each atom in the first clause
41  std::vector<std::vector<unsigned int>> validMoves;
42  for (size_t i = 0; i < clauseSize; i++) {
43  std::vector<unsigned int> currentRow;
44  for (size_t j = 0; j < clauseSize; j++) {
45  if (permutationMatrix[i][j] == 1) {
46  currentRow.push_back(j);
47  }
48  }
49  validMoves.push_back(currentRow);
50  }
51 
52  // extract the possible permutations, DFS style
53  std::vector<std::vector<unsigned int>> permutations;
54  std::vector<unsigned int> seen(clauseSize);
55  std::vector<unsigned int> currentPermutation;
56  std::stack<std::vector<unsigned int>> todoStack;
57 
58  todoStack.push(validMoves[0]);
59 
60  size_t currentIdx = 0;
61  while (!todoStack.empty()) {
62  if (currentIdx == clauseSize) {
63  // permutation is complete
64  permutations.push_back(currentPermutation);
65 
66  if (currentIdx == 0) {
67  // already at starting position, so no more permutations possible
68  break;
69  }
70 
71  // undo the last number added to the permutation
72  currentIdx--;
73  seen[currentPermutation[currentIdx]] = 0;
74  currentPermutation.pop_back();
75 
76  // see if we can pick up other permutations
77  continue;
78  }
79 
80  // pull out the possibilities for the current point of the permutation
81  std::vector<unsigned int> possibilities = todoStack.top();
82  todoStack.pop();
83  if (possibilities.empty()) {
84  // no more possibilities at this point, so undo our last move
85 
86  if (currentIdx == 0) {
87  // already at starting position, so no more permutations possible
88  break;
89  }
90 
91  currentIdx--;
92  seen[currentPermutation[currentIdx]] = 0;
93  currentPermutation.pop_back();
94 
95  // continue looking for permutations
96  continue;
97  }
98 
99  // try the next possibility
100  unsigned int nextNum = possibilities[0];
101 
102  // update the possibility vector for the current position
103  possibilities.erase(possibilities.begin());
104  todoStack.push(possibilities);
105 
106  if (seen[nextNum]) {
107  // number already seen in this permutation
108  continue;
109  } else {
110  // number can be used
111  seen[nextNum] = 1;
112  currentPermutation.push_back(nextNum);
113  currentIdx++;
114 
115  // if we havent reached the end of the permutation,
116  // push up the valid moves for the next position
117  if (currentIdx < clauseSize) {
118  todoStack.push(validMoves[currentIdx]);
119  }
120  }
121  }
122 
123  // found all permutations
124  return permutations;
125 }

127 /**
128  * Check if the atom at leftIdx in the left clause can potentially be matched up
129  * with the atom at rightIdx in the right clause.
130  * NOTE: index 0 refers to the head atom, index 1 to the first body atom, and so on.
131  */
132 bool isValidMove(const AstClause* left, size_t leftIdx, const AstClause* right, size_t rightIdx) {
133  // invalid indices
134  if (leftIdx < 0 || rightIdx < 0) {
135  return false;
136  }
137 
138  // handle the case where one of the indices refers to the head
139  if (leftIdx == 0 && rightIdx == 0) {
140  const AstAtom* leftHead = left->getHead()->getAtom();
141  const AstAtom* rightHead = right->getHead()->getAtom();
142  return leftHead->getName() == rightHead->getName();
143  } else if (leftIdx == 0 || rightIdx == 0) {
144  return false;
145  }
146 
147  // both must hence be body atoms
148  int leftBodyAtomIdx = leftIdx - 1;
149  const AstAtom* leftAtom = left->getBodyLiteral(leftBodyAtomIdx)->getAtom();
150 
151  int rightBodyAtomIdx = rightIdx - 1;
152  const AstAtom* rightAtom = right->getBodyLiteral(rightBodyAtomIdx)->getAtom();
153 
154  return leftAtom->getName() == rightAtom->getName();
155 }

157 /**
158  * Check whether a valid variable mapping exists for the given permutation.
159  */
160 bool isValidPermutation(
161  const AstClause* left, const AstClause* right, const std::vector<unsigned int>& permutation) {
162  // --- perform the permutation ---
163 
164  auto reorderedLeft = std::unique_ptr<AstClause>(left->clone());
165 
166  // deduce the body atom permutation from the full clause permutation
167  std::vector<unsigned int> bodyPermutation(permutation.begin() + 1, permutation.end());
168  for (size_t i = 0; i < bodyPermutation.size(); i++) {
169  bodyPermutation[i] -= 1;
170  }
171 
172  // currently, <permutation[i] == j> indicates that atom i should map to position j
173  // internally, for the clause class' reordering function, <permutation[i] == j> indicates
174  // that position i should contain atom j
175  // rearrange the permutation to match the internals
176  // TODO (azreika): perhaps change the internal reordering strategy because this came up in MST too
177  std::vector<unsigned int> reorderedPermutation(bodyPermutation.size());
178  for (size_t i = 0; i < bodyPermutation.size(); i++) {
179  reorderedPermutation[bodyPermutation[i]] = i;
180  }
181 
182  // perform the permutation
183  reorderedLeft->reorderAtoms(reorderedPermutation);
184 
185  // --- check if a valid variable exists corresponding to this permutation ---
186 
187  std::map<std::string, std::string> variableMap;
188  visitDepthFirst(*reorderedLeft, [&](const AstVariable& var) { variableMap[var.getName()] = ""; });
189 
190  // need to match the variables in the body
191  std::vector<AstLiteral*> leftAtoms = reorderedLeft->getBodyLiterals();
192  std::vector<AstLiteral*> rightAtoms = right->getBodyLiterals();
193 
194  // need to match the variables in the head
195  leftAtoms.push_back(reorderedLeft->getHead());
196  rightAtoms.push_back(right->getHead());
197 
198  // check if a valid variable mapping exists
199  auto isVariable = [&](const AstArgument* arg) {
200  return dynamic_cast<const AstVariable*>(arg) != nullptr;
201  };
202 
203  auto isConstant = [&](const AstArgument* arg) {
204  return dynamic_cast<const AstConstant*>(arg) != nullptr;
205  };
206 
207  bool validMapping = true;
208  for (size_t i = 0; i < leftAtoms.size() && validMapping; i++) {
209  // match arguments
210  std::vector<AstArgument*> leftArgs = leftAtoms[i]->getAtom()->getArguments();
211  std::vector<AstArgument*> rightArgs = rightAtoms[i]->getAtom()->getArguments();
212 
213  for (size_t j = 0; j < leftArgs.size(); j++) {
214  AstArgument* leftArg = leftArgs[j];
215  AstArgument* rightArg = rightArgs[j];
216  if (isVariable(leftArg) && isVariable(rightArg)) {
217  // both variables, their names should match to each other through the clause
218  std::string leftVarName = dynamic_cast<AstVariable*>(leftArg)->getName();
219  std::string rightVarName = dynamic_cast<AstVariable*>(rightArg)->getName();
220 
221  std::string currentMap = variableMap[leftVarName];
222  if (currentMap.empty()) {
223  // unassigned yet, so assign it appropriately
224  variableMap[leftVarName] = rightVarName;
225  } else if (currentMap != rightVarName) {
226  // mapping is inconsistent!
227  // clauses cannot be equivalent under this permutation
228  validMapping = false;
229  break;
230  }
231  } else if (isConstant(leftArg) && isConstant(rightArg)) {
232  // check if its the same constant
233  auto leftCst = dynamic_cast<AstConstant*>(leftArg)->getIndex();
234  auto rightCst = dynamic_cast<AstConstant*>(rightArg)->getIndex();
235 
236  if (leftCst != rightCst) {
237  // constants don't match, failed!
238  validMapping = false;
239  break;
240  }
241  } else {
242  // not the same type, failed!
243  validMapping = false;
244  break;
245  }
246  }
247  }
248 
249  // return whether a valid variable mapping exists for this permutation
250  return validMapping;
251 }

253 /**
254  * Check whether two clauses are bijectively equivalent.
255  */
256 bool areBijectivelyEquivalent(const AstClause* left, const AstClause* right) {
257  // only check bijective equivalence for a subset of the possible clauses
258  auto isValidClause = [&](const AstClause* clause) {
259  // check that all body literals are atoms
260  // i.e. avoid clauses with constraints or negations
261  // TODO (azreika): extend to constraints and negations
262  for (AstLiteral* lit : clause->getBodyLiterals()) {
263  if (dynamic_cast<AstAtom*>(lit) == nullptr) {
264  return false;
265  }
266  }
267 
268  // check that all arguments are either constants or variables
269  // i.e. only allow primitive arguments
270  bool valid = true;
271  visitDepthFirst(*clause, [&](const AstArgument& arg) {
272  if (dynamic_cast<const AstVariable*>(&arg) == nullptr &&
273  dynamic_cast<const AstConstant*>(&arg) == nullptr) {
274  valid = false;
275  }
276  });
277  return valid;
278  };
279 
280  if (!isValidClause(left) || !isValidClause(right)) {
281  return false;
282  }
283 
284  // rules must be the same length to be equal
285  if (left->getBodyLiterals().size() != right->getBodyLiterals().size()) {
286  return false;
287  }
288 
289  // set up the n x n permutation matrix, where n is the number of
290  // atoms in the clause, including the head atom
291  size_t size = left->getBodyLiterals().size() + 1;
292  auto permutationMatrix = std::vector<std::vector<unsigned int>>(size);
293  for (size_t i = 0; i < permutationMatrix.size(); i++) {
294  permutationMatrix[i] = std::vector<unsigned int>(size);
295  }
296 
297  // create permutation matrix
298  for (size_t i = 0; i < size; i++) {
299  for (size_t j = 0; j < size; j++) {
300  if (isValidMove(left, i, right, j)) {
301  permutationMatrix[i][j] = 1;
302  }
303  }
304  }
305 
306  // check if any of these permutations have valid variable mappings associated with them
307  std::vector<std::vector<unsigned int>> permutations = extractPermutations(permutationMatrix);
308  for (auto permutation : permutations) {
309  if (isValidPermutation(left, right, permutation)) {
310  // valid permutation with valid corresponding variable mapping exists
311  // therefore, the two clauses are equivalent!
312  return true;
313  }
314  }
315 
316  return false;
317 }

319 bool MinimiseProgramTransformer::transform(AstTranslationUnit& translationUnit) {
320  AstProgram& program = *translationUnit.getProgram();
321 
322  std::vector<AstClause*> clausesToDelete;
323 
324  // split up each relation's rules into equivalene classes
325  // TODO (azreika): consider turning this into an ast analysis instead
326  for (AstRelation* rel : program.getRelations()) {
327  std::vector<std::vector<AstClause*>> equivalenceClasses;
328 
329  for (AstClause* clause : rel->getClauses()) {
330  bool added = false;
331 
332  for (std::vector<AstClause*>& eqClass : equivalenceClasses) {
333  AstClause* rep = eqClass[0];
334 
335  if (areBijectivelyEquivalent(rep, clause)) {
336  // clause belongs to an existing equivalence class, so delete it
337  eqClass.push_back(clause);
338  clausesToDelete.push_back(clause);
339  added = true;
340  break;
341  }
342  }
343 
344  if (!added) {
345  // clause does not belong to any existing equivalence class, so keep it
346  std::vector<AstClause*> clauseToAdd = {clause};
347  equivalenceClasses.push_back(clauseToAdd);
348  }
349  }
350  }
351 
352  // remove non-representative clauses
353  for (auto clause : clausesToDelete) {
354  program.removeClause(clause);
355  }
356 
357  // changed iff any clauses were deleted
358  return !clausesToDelete.empty();
359 }
