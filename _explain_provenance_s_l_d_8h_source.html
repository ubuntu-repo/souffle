71  std::unique_ptr<TreeNode> explain(
72  std::string relName, std::vector<RamDomain> tuple, int ruleNum, int levelNum, size_t depthLimit) {
73  std::stringstream joinedArgs;
74  joinedArgs << join(numsToArgs(relName, tuple), ", ");
75  auto joinedArgsStr = joinedArgs.str();
76 
77  // if fact
78  if (levelNum == 0) {
79  return std::make_unique<LeafNode>(relName + "(" + joinedArgsStr + ")");
80  }
81 
82  assert(info.find(std::make_pair(relName, ruleNum)) != info.end() && "invalid rule for tuple");
83 
84  // if depth limit exceeded
85  if (depthLimit <= 1) {
86  tuple.push_back(ruleNum);
87  tuple.push_back(levelNum);
88 
89  // find if subproof exists already
90  size_t idx = 0;
91  auto it = std::find(subproofs.begin(), subproofs.end(), tuple);
92  if (it != subproofs.end()) {
93  idx = it - subproofs.begin();
94  } else {
95  subproofs.push_back(tuple);
96  idx = subproofs.size() - 1;
97  }
98 
99  return std::make_unique<LeafNode>("subproof " + relName + "(" + std::to_string(idx) + ")");
100  }
101 
102  auto internalNode = std::make_unique<InnerNode>(
103  relName + "(" + joinedArgsStr + ")", "(R" + std::to_string(ruleNum) + ")");
104 
105  // make return vector pointer
106  std::vector<RamDomain> ret;
107  std::vector<bool> err;
108 
109  // add level number to tuple
110  tuple.push_back(levelNum);
111 
112  // execute subroutine to get subproofs
113  prog.executeSubroutine(relName + "_" + std::to_string(ruleNum) + "_subproof", tuple, ret, err);
114 
115  // recursively get nodes for subproofs
116  size_t tupleCurInd = 0;
117  auto bodyRelations = info[std::make_pair(relName, ruleNum)];
118 
119  // start from begin + 1 because the first element represents the head atom
120  for (auto it = bodyRelations.begin() + 1; it < bodyRelations.end(); it++) {
121  std::string bodyLiteral = *it;
122  // split bodyLiteral since it contains relation name plus arguments
123  std::string bodyRel = splitString(bodyLiteral, ',')[0];
124 
125  // check whether the current atom is a constraint
126  bool isConstraint =
127  std::find(constraintList.begin(), constraintList.end(), bodyRel) != constraintList.end();
128 
129  // handle negated atom names
130  auto bodyRelAtomName = bodyRel;
131  if (bodyRel[0] == '!') {
132  bodyRelAtomName = bodyRel.substr(1);
133  }
134 
135  // traverse subroutine return
136  size_t arity;
137  if (isConstraint) {
138  // we only handle binary constraints, and assume arity is 4 to account for hidden provenance
139  // annotations
140  arity = 4;
141  } else {
142  arity = prog.getRelation(bodyRelAtomName)->getArity();
143  }
144  auto tupleEnd = tupleCurInd + arity;
145 
146  // store current tuple and error
147  std::vector<RamDomain> subproofTuple;
148  std::vector<bool> subproofTupleError;
149 
150  for (; tupleCurInd < tupleEnd - 2; tupleCurInd++) {
151  subproofTuple.push_back(ret[tupleCurInd]);
152  subproofTupleError.push_back(err[tupleCurInd]);
153  }
154 
155  int subproofRuleNum = ret[tupleCurInd];
156  int subproofLevelNum = ret[tupleCurInd + 1];
157 
158  // for a negation, display the corresponding tuple and do not recurse
159  if (bodyRel[0] == '!') {
160  std::stringstream joinedTuple;
161  joinedTuple << join(numsToArgs(bodyRelAtomName, subproofTuple, &subproofTupleError), ", ");
162  auto joinedTupleStr = joinedTuple.str();
163  internalNode->add_child(std::make_unique<LeafNode>(bodyRel + "(" + joinedTupleStr + ")"));
164  internalNode->setSize(internalNode->getSize() + 1);
165  // for a binary constraint, display the corresponding values and do not recurse
166  } else if (isConstraint) {
167  std::stringstream joinedConstraint;
168 
169  if (isNumericBinaryConstraintOp(toBinaryConstraintOp(bodyRel))) {
170  joinedConstraint << subproofTuple[0] << " " << bodyRel << " " << subproofTuple[1];
171  } else {
172  joinedConstraint << bodyRel << "(\"" << symTable.resolve(subproofTuple[0]) << "\", \""
173  << symTable.resolve(subproofTuple[1]) << "\")";
174  }
175 
176  internalNode->add_child(std::make_unique<LeafNode>(joinedConstraint.str()));
177  internalNode->setSize(internalNode->getSize() + 1);
178  // otherwise, for a normal tuple, recurse
179  } else {
180  auto child =
181  explain(bodyRel, subproofTuple, subproofRuleNum, subproofLevelNum, depthLimit - 1);
182  internalNode->setSize(internalNode->getSize() + child->getSize());
183  internalNode->add_child(std::move(child));
184  }
185 
186  tupleCurInd = tupleEnd;
187  }
188 
189  return std::move(internalNode);
190  }

226  std::string relName, std::vector<std::string> args, size_t ruleNum) override {
227  std::vector<std::string> variables;
228 
229  // check that the tuple actually doesn't exist
230  if (findTuple(relName, argsToNums(relName, args)) != std::make_pair(-1, -1)) {
231  // return a sentinel value
232  return std::vector<std::string>({"@"});
233  }
234 
235  // atom meta information stored for the current rule
236  auto atoms = info[std::make_pair(relName, ruleNum)];
237 
238  // atoms[0] represents variables in the head atom
239  auto headVariables = splitString(atoms[0], ',');
240 
241  // check that head variable bindings make sense, i.e. for a head like a(x, x), make sure both x are
242  // the same value
243  std::map<std::string, std::string> headVariableMapping;
244  for (size_t i = 0; i < headVariables.size(); i++) {
245  if (headVariableMapping.find(headVariables[i]) == headVariableMapping.end()) {
246  headVariableMapping[headVariables[i]] = args[i];
247  } else {
248  if (headVariableMapping[headVariables[i]] != args[i]) {
249  return std::vector<std::string>({"@non_matching"});
250  }
251  }
252  }
253 
254  // get body variables
255  std::vector<std::string> uniqueBodyVariables;
256  for (auto it = atoms.begin() + 1; it < atoms.end(); it++) {
257  auto atomRepresentation = splitString(*it, ',');
258 
259  // atomRepresentation.begin() + 1 because the first element is the relation name of the atom
260  // which is not relevant for finding variables
261  for (auto atomIt = atomRepresentation.begin() + 1; atomIt < atomRepresentation.end(); atomIt++) {
262  if (!contains(uniqueBodyVariables, *atomIt) && !contains(headVariables, *atomIt)) {
263  uniqueBodyVariables.push_back(*atomIt);
264  }
265  }
266  }
267 
268  return uniqueBodyVariables;

271  std::unique_ptr<TreeNode> explainNegation(std::string relName, size_t ruleNum,
272  const std::vector<std::string>& tuple,
273  std::map<std::string, std::string>& bodyVariables) override {
274  // construct a vector of unique variables that occur in the rule
275  std::vector<std::string> uniqueVariables;
276 
277  // we also need to know the type of each variable
278  std::map<std::string, char> variableTypes;
279 
280  // atom meta information stored for the current rule
281  auto atoms = info[std::make_pair(relName, ruleNum)];
282 
283  // atoms[0] represents variables in the head atom
284  auto headVariables = splitString(atoms[0], ',');
285 
286  uniqueVariables.insert(uniqueVariables.end(), headVariables.begin(), headVariables.end());
287 
288  // get body variables
289  for (auto it = atoms.begin() + 1; it < atoms.end(); it++) {
290  auto atomRepresentation = splitString(*it, ',');
291 
292  // atomRepresentation.begin() + 1 because the first element is the relation name of the atom
293  // which is not relevant for finding variables
294  for (auto atomIt = atomRepresentation.begin() + 1; atomIt < atomRepresentation.end(); atomIt++) {
295  if (!contains(uniqueVariables, *atomIt) && !contains(headVariables, *atomIt)) {
296  uniqueVariables.push_back(*atomIt);
297 
298  // store type of variable
299  auto currentRel = prog.getRelation(atomRepresentation[0]);
300  assert(currentRel != nullptr &&
301  ("relation " + atomRepresentation[0] + " doesn't exist").c_str());
302  variableTypes[*atomIt] =
303  *currentRel->getAttrType(atomIt - atomRepresentation.begin() - 1);
304  }
305  }
306  }
307 
308  std::vector<RamDomain> args;
309 
310  size_t varCounter = 0;
311 
312  // construct arguments to pass in to the subroutine
313  // - this contains the variable bindings selected by the user
314 
315  // add number representation of tuple
316  auto tupleNums = argsToNums(relName, tuple);
317  args.insert(args.end(), tupleNums.begin(), tupleNums.end());
318  varCounter += tuple.size();
319 
320  while (varCounter < uniqueVariables.size()) {
321  auto var = uniqueVariables[varCounter];
322  auto varValue = bodyVariables[var];
323  if (variableTypes[var] == 's') {
324  if (varValue.size() >= 2 && varValue[0] == '"' && varValue[varValue.size() - 1] == '"') {
325  auto originalStr = varValue.substr(1, varValue.size() - 2);
326  args.push_back(symTable.lookup(originalStr));
327  } else {
328  // assume no quotation marks
329  args.push_back(symTable.lookup(varValue));
330  }
331  } else {
332  args.push_back(std::stoi(varValue));
333  }
334 
335  varCounter++;
336  }
337 
338  // set up return and error vectors for subroutine calling
339  std::vector<RamDomain> ret;
340  std::vector<bool> err;
341 
342  // execute subroutine to get subproofs
343  prog.executeSubroutine(
344  relName + "_" + std::to_string(ruleNum) + "_negation_subproof", args, ret, err);
345 
346  // construct tree nodes
347  std::stringstream joinedArgsStr;
348  joinedArgsStr << join(tuple, ",");
349  auto internalNode = std::make_unique<InnerNode>(
350  relName + "(" + joinedArgsStr.str() + ")", "(R" + std::to_string(ruleNum) + ")");
351 
352  // traverse return vector and construct child nodes
353  // making sure we display existent and non-existent tuples correctly
354  int literalCounter = 1;
355  for (size_t returnCounter = 0; returnCounter < ret.size(); returnCounter++) {
356  // check what the next contained atom is
357  bool atomExists;
358  if (err[returnCounter]) {
359  atomExists = false;
360  returnCounter++;
361  } else {
362  atomExists = true;
363  assert(err[returnCounter + 1] && "there should be a separator for literals in return");
364  returnCounter += 2;
365  }
366 
367  // get the relation of the current atom
368  auto atomRepresentation = splitString(atoms[literalCounter], ',');
369  std::string bodyRel = atomRepresentation[0];
370 
371  // check whether the current atom is a constraint
372  bool isConstraint =
373  std::find(constraintList.begin(), constraintList.end(), bodyRel) != constraintList.end();
374 
375  // handle negated atom names
376  auto bodyRelAtomName = bodyRel;
377  if (bodyRel[0] == '!') {
378  bodyRelAtomName = bodyRel.substr(1);
379  }
380 
381  // traverse subroutine return
382  size_t arity;
383  if (isConstraint) {
384  // we only handle binary constraints, and assume arity is 4 to account for hidden provenance
385  // annotations
386  arity = 4;
387  } else {
388  arity = prog.getRelation(bodyRelAtomName)->getArity();
389  }
390 
391  // process current literal
392  std::vector<RamDomain> atomValues;
393  std::vector<bool> atomErrs;
394  size_t j = returnCounter;
395  for (; j < returnCounter + arity - 2; j++) {
396  atomValues.push_back(ret[j]);
397  atomErrs.push_back(err[j]);
398  }
399 
400  // add child nodes to the proof tree
401  std::stringstream childLabel;
402  if (isConstraint) {
403  assert(atomValues.size() == 2 && "not a binary constraint");
404 
405  if (isNumericBinaryConstraintOp(toBinaryConstraintOp(bodyRel))) {
406  childLabel << atomValues[0] << " " << bodyRel << " " << atomValues[1];
407  } else {
408  childLabel << bodyRel << "(\"" << symTable.resolve(atomValues[0]) << "\", \""
409  << symTable.resolve(atomValues[1]) << "\")";
410  }
411  } else {
412  childLabel << bodyRel << "(";
413  childLabel << join(numsToArgs(bodyRelAtomName, atomValues, &atomErrs), ", ");
414  childLabel << ")";
415  }
416 
417  if (atomExists) {
418  childLabel << " ✓";
419  } else {
420  childLabel << " x";
421  }
422 
423  internalNode->add_child(std::make_unique<LeafNode>(childLabel.str()));
424  internalNode->setSize(internalNode->getSize() + 1);
425 
426  returnCounter = j - 1;
427  literalCounter++;
428  }
429 
430  return std::move(internalNode);
431  }

544  std::pair<int, int> findTuple(const std::string& relName, std::vector<RamDomain> tup) {
545  auto rel = prog.getRelation(relName);
546 
547  if (rel == nullptr) {
548  return std::make_pair(-1, -1);
549  }
550 
551  // find correct tuple
552  for (auto& tuple : *rel) {
553  bool match = true;
554  std::vector<RamDomain> currentTuple;
555 
556  for (size_t i = 0; i < rel->getArity() - 2; i++) {
557  RamDomain n;
558  if (*rel->getAttrType(i) == 's') {
559  std::string s;
560  tuple >> s;
561  n = symTable.lookupExisting(s);
562  } else {
563  tuple >> n;
564  }
565 
566  currentTuple.push_back(n);
567 
568  if (n != tup[i]) {
569  match = false;
570  break;
571  }
572  }
573 
574  if (match) {
575  RamDomain ruleNum;
576  tuple >> ruleNum;
577 
578  RamDomain levelNum;
579  tuple >> levelNum;
580 
581  return std::make_pair(ruleNum, levelNum);
582  }
583  }
584 
585  // if no tuple exists
586  return std::make_pair(-1, -1);
