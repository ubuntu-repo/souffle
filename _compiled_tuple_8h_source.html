26 /**
27  * The type of object stored within relations representing the actual
28  * tuple value. Each tuple consists of a constant number of components.
29  *
30  * @tparam Domain the domain of the component values
31  * @tparam arity the number of components within an instance
32  */
33 template <typename Domain, std::size_t _arity>
34 struct Tuple {

55  bool operator==(const Tuple& other) const {
56  for (std::size_t i = 0; i < arity; i++) {
57  if (data[i] != other.data[i]) return false;

59  return true;

64  return !(*this == other);

66 
67  // required to put tuples into e.g. a std::set container
68  bool operator<(const Tuple& other) const {
69  for (std::size_t i = 0; i < arity; ++i) {
70  if (data[i] < other.data[i]) return true;
71  if (data[i] > other.data[i]) return false;
72  }
73  return false;

77  bool operator>(const Tuple& other) const {
78  for (std::size_t i = 0; i < arity; ++i) {
79  if (data[i] > other.data[i]) return true;
80  if (data[i] < other.data[i]) return false;
81  }

87  if (arity == 0) return out << "[]";
88  out << "[";
