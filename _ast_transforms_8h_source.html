350 class PipelineTransformer : public MetaTransformer {
351 public:
352  template <typename... Args>
353  PipelineTransformer(Args... args) {
354  std::unique_ptr<AstTransformer> tmp[] = {std::move(args)...};
355  for (auto& cur : tmp) {
356  pipeline.push_back(std::move(cur));
357  }
358  }
359 
360  void setDebugReport() override {
361  for (auto& i : pipeline) {
362  if (auto* mt = dynamic_cast<MetaTransformer*>(i.get())) {
363  mt->setDebugReport();
364  } else {
365  i = std::make_unique<DebugReporter>(std::move(i));
366  }
367  }
368  }
369 
370  void setVerbosity(bool verbose) override {
371  this->verbose = verbose;
372  for (auto& cur : pipeline) {
373  if (auto* mt = dynamic_cast<MetaTransformer*>(cur.get())) {
374  mt->setVerbosity(verbose);
375  }
376  }
377  }
378 
379  void disableTransformers(const std::set<std::string>& transforms) override {
380  for (auto& i : pipeline) {
381  if (auto* mt = dynamic_cast<MetaTransformer*>(i.get())) {
382  mt->disableTransformers(transforms);
383  } else if (transforms.find(i->getName()) != transforms.end()) {
384  i = std::make_unique<NullTransformer>();
385  }
386  }
387  }
388 
389  std::string getName() const override {
390  return "PipelineTransformer";
391  }
392 
393 private:
394  std::vector<std::unique_ptr<AstTransformer>> pipeline;
395  bool transform(AstTranslationUnit& translationUnit) override;
396 };
