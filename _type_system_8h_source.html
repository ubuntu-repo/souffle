165 /**
166  * A collection to represent sets of types. In addition to ordinary set capabilities
167  * it may also represent the set of all types -- without being capable of iterating over those.
168  *
169  * It is the basic entity to conduct sub- and super-type computations.
170  */
171 struct TypeSet {
172 public:
173  using const_iterator = IterDerefWrapper<typename std::set<const Type*>::const_iterator>;
174 
175  // -- constructors, destructors and assignment operations --
176 
177  TypeSet(bool all = false) : all(all) {}
178 
179  TypeSet(const TypeSet& other) = default;
180 
181  TypeSet(TypeSet&& other) noexcept : all(other.all), types() {
182  types.swap(other.types);
183  }
184 
185  template <typename... Types>
186  TypeSet(const Types&... types) : all(false) {
187  for (const Type* cur : toVector<const Type*>(&types...)) {
188  this->types.insert(cur);
189  }
190  }
191 
192  TypeSet& operator=(const TypeSet& other) = default;
193 
194  /** A factory function for the all-types set */
195  static TypeSet getAllTypes() {
196  return TypeSet(true);
197  }
198 
199  /** Emptiness check */
200  bool empty() const {
201  return !all && types.empty();
202  }
203 
204  /** Universality check */
205  bool isAll() const {
206  return all;
207  }
208 
209  /** Determines the size of this set unless it is the universal set */
210  std::size_t size() const {
211  assert(!all && "Unable to give size of universe.");
212  return types.size();
213  }
214 
215  /** Determines whether a given type is included or not */
216  bool contains(const Type& type) const {
217  return all || types.find(&type) != types.end();
218  }
219 
220  /** Adds the given type to this set */
221  void insert(const Type& type) {
222  if (all) {
223  return;
224  }
225  types.insert(&type);
226  }
227 
228  /** Inserts all the types of the given set into this set */
229  void insert(const TypeSet& set) {
230  if (all) {
231  return;
232  }
233 
234  // if the other set is universal => make this one universal
235  if (set.isAll()) {
236  all = true;
237  types.clear();
238  return;
239  }
240 
241  // add types one by one
242  for (const auto& t : set) {
243  insert(t);
244  }
245  }
246 
247  /** Allows to iterate over the types contained in this set (only if not universal) */
248  const_iterator begin() const {
249  assert(!all && "Unable to enumerate universe.");
250  return derefIter(types.begin());
251  }
252 
253  /** Allows to iterate over the types contained in this set (only if not universal) */
254  const_iterator end() const {
255  assert(!all && "Unable to enumerate universe.");
256  return derefIter(types.end());
257  }
258 
259  /** Determines whether this set is a subset of the given set */
260  bool isSubsetOf(const TypeSet& b) const {
261  if (all) {
262  return b.isAll();
263  }
264  return all_of(*this, [&](const Type& cur) { return b.contains(cur); });
265  }
266 
267  /** Determines equality between type sets */
268  bool operator==(const TypeSet& other) const {
269  return all == other.all && types == other.types;
270  }
271 
272  /** Determines inequality between type sets */
273  bool operator!=(const TypeSet& other) const {
274  return !(*this == other);
275  }
276 
277  /** Adds print support for type sets */
278  void print(std::ostream& out) const {
279  if (all) {
280  out << "{ - all types - }";
281  } else {
282  out << "{"
283  << join(types, ",", [](std::ostream& out, const Type* type) { out << type->getName(); })
284  << "}";
285  }
286  }
287 
288  friend std::ostream& operator<<(std::ostream& out, const TypeSet& set) {
289  set.print(out);
290  return out;
291  }
292 
293 private:
294  /** True if it is the all-types set, false otherwise */
295  bool all;
296 
297  /** The enumeration of types in case it is not the all-types set */
298  std::set<const Type*, deref_less<Type>> types;
299 };
